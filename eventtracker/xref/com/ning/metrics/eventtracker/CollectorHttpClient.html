17  package com.ning.metrics.eventtracker;
18  
19  import com.ning.metrics.serialization.event.Event;
20  import org.apache.commons.httpclient.Header;
21  import org.apache.commons.httpclient.HttpClient;
22  import org.apache.commons.httpclient.methods.ByteArrayRequestEntity;
23  import org.apache.commons.httpclient.methods.PostMethod;
24  
25  import java.io.ByteArrayOutputStream;
26  import java.io.IOException;
27  import java.io.ObjectOutputStream;
28  
29  /**
30   * Client class to post Events to the Event collector.
31   */
32  public class CollectorHttpClient
33  {
34      public static final String URI_PATH = "/rest/1.0/event";
35  
36      private final CollectorUriBuilder uriBuilder;
37      private final HttpClient httpClient;
38      private final Header header = new Header("Content-type", "ning/thrift");
39  
40      /**
41       * Creates a new Collector Client that uses a fixed host and port as target.
42       *
43       * @param host   Collector hostname or SLB VIP
44       * @param port   Collector port
45       * @param client HttpClient to use
46       */
47      public CollectorHttpClient(String host, int port, HttpClient client)
48      {
49          this(new SimpleUriBuilder(String.format("http://%s:%d%s", host, port, URI_PATH)), client);
50      }
51  
52      /**
53       * Creates a new Collector Client that uses an URI builder implementation to find
54       * the host to talk to.
55       *
56       * @param uriBuilder CollectorUriBuilder to use
57       * @param httpClient HttpClient to use
58       */
59      public CollectorHttpClient(final CollectorUriBuilder uriBuilder, final HttpClient httpClient)
60      {
61          this.uriBuilder = uriBuilder;
62          this.httpClient = httpClient;
63      }
64  
65      /**
66       * Logs an event to the collector. The <code>eventDate</code> is used to figure out what path to store the event under, e.g., <code>/events/ning/Foo/2009/07/12/20</code>
67       *
68       * @param event Event to send
69       * @return true if event was accepted, false if it was rejected
70       * @throws IOException for I/O exceptions
71       */
72      public boolean postThrift(Event event) throws IOException
73      {
74          ByteArrayOutputStream out = new ByteArrayOutputStream();
75          event.writeExternal(new ObjectOutputStream(out));
76          byte[] payload = out.toByteArray();
77  
78          // Post to collector
79  
80          final CollectorUriBuilder requestUriBuilder = uriBuilder.clone()
81              .addQueryParam("name", event.getName());
82  
83          if (event.getEventDateTime() != null) {
84              requestUriBuilder.addQueryParam("date", event.getEventDateTime().toString());
85          }
86  
87          if (event.getGranularity() != null) {
88              requestUriBuilder.addQueryParam("granularity", event.getGranularity().toString());
89          }
90  
91          PostMethod method = new PostMethod(requestUriBuilder.build().toString());
92          method.setRequestEntity(new ByteArrayRequestEntity(payload));
93          method.addRequestHeader(header);
94  
95          boolean success = false;
96          try {
97              int result = httpClient.executeMethod(method);
98              success = (result == 202); //accepted;
99          }
100         finally {
101             method.releaseConnection();
102         }
103 
104         return success;
105     }
