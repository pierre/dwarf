17  package com.ning.metrics.eventtracker;

19  import com.google.inject.AbstractModule;
20  import com.google.inject.Singleton;
21  import com.ning.metrics.serialization.writer.DiskSpoolEventWriter;
22  import com.ning.metrics.serialization.writer.EventWriter;
23  import org.apache.commons.httpclient.HttpClient;
24  import org.apache.log4j.Logger;
25  import org.skife.config.ConfigurationObjectFactory;
26  
27  import java.util.concurrent.Executors;
28  import java.util.concurrent.ScheduledExecutorService;
29  import java.util.concurrent.ScheduledThreadPoolExecutor;

31  /**
32   * Wires all pieces related to talking to the Collector core.
33   * See http://github.com/pierre/collector
34   * <p/>
35   * Note that Guice injection is optional, you can directly instantiate a CollectorController
36   * via the factories.
37   *
38   * @see com.ning.metrics.eventtracker.ScribeCollectorFactory
39   */
40  public class CollectorControllerModule extends AbstractModule
41  {
42      private static final Logger log = Logger.getLogger(CollectorControllerModule.class);
43  
44      public static enum Type
45      {
46          SCRIBE,
47          COLLECTOR,
48          NO_LOGGING
49      }
50  
51      protected Type type;
52  
53      @Override
54      protected void configure()
55      {
56          final CollectorConfig collectorConfig = new ConfigurationObjectFactory(System.getProperties()).build(CollectorConfig.class);
57          bind(CollectorConfig.class).toInstance(collectorConfig);
58  
59          type = Type.valueOf(collectorConfig.getType());
60  
61          switch (type) {
62              case COLLECTOR:
63                  bind(HttpClient.class)
64                      .to(HttpClient.class) // despite what Idea thinks, the "to(HttpClient.class)" is necessary
65                      .in(Singleton.class);
66                  bind(CollectorUriBuilder.class).to(SimpleUriBuilder.class).in(Singleton.class);
67                  bind(EventSender.class).to(HttpSender.class);
68                  log.info("Enabled Collector Event Logging");
69                  break;
70              case SCRIBE:
71                  bind(ScribeSenderProvider.class).in(new FixedManagedJmxExportScope(log, "eventtracker:name=ScribeSender"));
72                  bind(EventSender.class).toProvider(ScribeSenderProvider.class);
73                  log.info("Enabled Scribe Event Logging");
74                  break;
75              case NO_LOGGING:
76                  bind(EventSender.class).toInstance(new NoLoggingSender());
77                  log.info("Disabled Event Logging");
78                  break;
79              default:
80                  throw new IllegalStateException("Unknown type " + type);
81          }
82  
83          bind(ScheduledExecutorService.class).toInstance(new ScheduledThreadPoolExecutor(1, Executors.defaultThreadFactory()));
84          bind(CollectorController.class).in(new FixedManagedJmxExportScope(log, "eventtracker:name=CollectorController"));
85          bind(DiskSpoolEventWriter.class).toProvider(DiskSpoolEventWriterProvider.class);
86          bind(EventWriter.class).toProvider(ThresholdEventWriterProvider.class);
87      }
88  }
