017    package com.ning.metrics.eventtracker;
018    
019    import com.ning.metrics.serialization.event.Event;
020    import org.apache.commons.httpclient.Header;
021    import org.apache.commons.httpclient.HttpClient;
022    import org.apache.commons.httpclient.methods.ByteArrayRequestEntity;
023    import org.apache.commons.httpclient.methods.PostMethod;
024    
025    import java.io.ByteArrayOutputStream;
026    import java.io.IOException;
027    import java.io.ObjectOutputStream;
028    
029    /**
030     * Client class to post Events to the Event collector.
031     */
032    public class CollectorHttpClient
033    {
034        public static final String URI_PATH = "/rest/1.0/event";
035    
036        private final CollectorUriBuilder uriBuilder;
037        private final HttpClient httpClient;
038        private final Header header = new Header("Content-type", "ning/thrift");
039    
040        /**
041         * Creates a new Collector Client that uses a fixed host and port as target.
042         *
043         * @param host   Collector hostname or SLB VIP
044         * @param port   Collector port
045         * @param client HttpClient to use
046         */
047        public CollectorHttpClient(String host, int port, HttpClient client)
048        {
049            this(new SimpleUriBuilder(String.format("http://%s:%d%s", host, port, URI_PATH)), client);
050        }
051    
052        /**
053         * Creates a new Collector Client that uses an URI builder implementation to find
054         * the host to talk to.
055         *
056         * @param uriBuilder CollectorUriBuilder to use
057         * @param httpClient HttpClient to use
058         */
059        public CollectorHttpClient(final CollectorUriBuilder uriBuilder, final HttpClient httpClient)
060        {
061            this.uriBuilder = uriBuilder;
062            this.httpClient = httpClient;
063        }
064    
065        /**
066         * Logs an event to the collector. The <code>eventDate</code> is used to figure out what path to store the event under, e.g., <code>/events/ning/Foo/2009/07/12/20</code>
067         *
068         * @param event Event to send
069         * @return true if event was accepted, false if it was rejected
070         * @throws IOException for I/O exceptions
071         */
072        public boolean postThrift(Event event) throws IOException
073        {
074            ByteArrayOutputStream out = new ByteArrayOutputStream();
075            event.writeExternal(new ObjectOutputStream(out));
076            byte[] payload = out.toByteArray();
077    
078            // Post to collector
079    
080            final CollectorUriBuilder requestUriBuilder = uriBuilder.clone()
081                .addQueryParam("name", event.getName());
082    
083            if (event.getEventDateTime() != null) {
084                requestUriBuilder.addQueryParam("date", event.getEventDateTime().toString());
085            }
086    
087            if (event.getGranularity() != null) {
088                requestUriBuilder.addQueryParam("granularity", event.getGranularity().toString());
089            }
090    
091            PostMethod method = new PostMethod(requestUriBuilder.build().toString());
092            method.setRequestEntity(new ByteArrayRequestEntity(payload));
093            method.addRequestHeader(header);
094    
095            boolean success = false;
096            try {
097                int result = httpClient.executeMethod(method);
098                success = (result == 202); //accepted;
099            }
100            finally {
101                method.releaseConnection();
102            }
103    
104            return success;
105        }
106    
107    }
