017    package com.ning.metrics.eventtracker;
018    
019    import com.ning.metrics.serialization.event.Event;
020    import com.ning.metrics.serialization.util.Managed;
021    import org.apache.commons.codec.binary.Base64;
022    import org.apache.commons.logging.Log;
023    import org.apache.commons.logging.LogFactory;
024    import org.apache.thrift.transport.TTransportException;
025    import scribe.thrift.LogEntry;
026    import scribe.thrift.ResultCode;
027    
028    import java.io.ByteArrayOutputStream;
029    import java.io.IOException;
030    import java.io.ObjectOutputStream;
031    import java.nio.charset.Charset;
032    import java.util.ArrayList;
033    import java.util.List;
034    import java.util.concurrent.atomic.AtomicInteger;
035    
036    /**
037     * ScribeSender
038     * <p/>
039     * The class needs to be public for JMX.
040     */
041    public class ScribeSender implements EventSender
042    {
043        private static final Log log = LogFactory.getLog(ScribeSender.class);
044    
045        private final AtomicInteger connectionRetries = new AtomicInteger(0);
046        private ScribeClient scribeClient;
047    
048        private final AtomicInteger messagesSuccessfullySent = new AtomicInteger(0);
049        private final AtomicInteger messagesSuccessfullySentSinceLastReconnection = new AtomicInteger(0);
050        private int messagesToSendBeforeReconnecting = 0;
051    
052        public ScribeSender(ScribeClient scribeClient, int messagesToSendBeforeReconnecting)
053        {
054            this.scribeClient = scribeClient;
055            this.messagesToSendBeforeReconnecting = messagesToSendBeforeReconnecting;
056        }
057    
058        /**
059         * Re-initialize the connection with the Scribe endpoint.
060         */
061        public synchronized void createConnection()
062        {
063            if (scribeClient != null) {
064                try {
065                    connectionRetries.incrementAndGet();
066                    scribeClient.closeLogger();
067                    scribeClient.openLogger();
068    
069                    log.info("Connection to Scribe established");
070                }
071                catch (TTransportException e) {
072                    log.warn(String.format("Unable to connect to Scribe: %s", e.getLocalizedMessage()));
073                    scribeClient.closeLogger();
074                }
075            }
076            else {
077                log.warn("Scribe client has not been set up correctly.");
078            }
079        }
080    
081        /**
082         * Disconnect from Scribe for good.
083         */
084        public void shutdown()
085        {
086            if (scribeClient != null) {
087                scribeClient.closeLogger();
088            }
089        }
090    
091        @Override
092        public boolean send(Event event) throws IOException
093        {
094            if (scribeClient == null) {
095                log.warn("Scribe client has not been set up correctly.");
096                return false;
097            }
098    
099            ResultCode res;
100            // TODO: offer batch API?, see drainEvents
101            List<LogEntry> list = new ArrayList<LogEntry>(1);
102            // TODO: update Scribe to pass a Thrift directly instead of serializing it
103    
104            // Has the sender specified how to send the data?
105            byte[] payload = event.getSerializedEvent();
106    
107            // Nope, default to ObjectOutputStream
108            if (payload == null) {
109                ByteArrayOutputStream out = new ByteArrayOutputStream();
110                event.writeExternal(new ObjectOutputStream(out));
111                payload = out.toByteArray();
112    
113                // 64-bit encode the serialized object
114                payload = new Base64().encode(payload);
115            }
116    
117    
118            String scribePayload = new String(payload, Charset.forName("UTF-8"));
119            list.add(new LogEntry(event.getName(), String.format("%s:%s", event.getEventDateTime().getMillis(), scribePayload)));
120    
121            try {
122                res = scribeClient.log(list);
123                messagesSuccessfullySent.addAndGet(list.size());
124                messagesSuccessfullySentSinceLastReconnection.addAndGet(list.size());
125    
126                // For load balancing capabilities, we don't want to make sticky connections to Scribe.
127                // After a certain threshold, force a refresh of the connection.
128                if (messagesSuccessfullySentSinceLastReconnection.get() > messagesToSendBeforeReconnecting) {
129                    log.info("Recycling connection with Scribe");
130                    messagesSuccessfullySentSinceLastReconnection.set(0);
131                    createConnection();
132                }
133            }
134            catch (org.apache.thrift.TException e) {
135                // Connection flacky?
136                log.warn(String.format("Error while sending message to Scribe: %s", e.getLocalizedMessage()));
137                createConnection();
138    
139                // Throw the exception for the caller to log the exception before quarantining the events
140                throw new IOException(e);
141            }
142    
143            return res == ResultCode.OK;
144        }
145    
146        @Managed(description = "Get the number of messages successfully sent since startup to Scribe")
147        public long getMessagesSuccessfullySent()
148        {
149            return messagesSuccessfullySent.get();
150        }
151    
152        @Managed(description = "Get the number of messages successfully sent since last reconnection to Scribe")
153        public long getMessagesSuccessfullySentSinceLastReconnection()
154        {
155            return messagesSuccessfullySentSinceLastReconnection.get();
156        }
157    
158        @Managed(description = "Get the number of times we retried to connect to Scribe")
159        public long getConnectionRetries()
160        {
161            return connectionRetries.get();
162        }
163    }
