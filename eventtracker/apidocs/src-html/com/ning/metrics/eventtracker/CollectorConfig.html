017    package com.ning.metrics.eventtracker;
018    
019    import org.skife.config.Config;
020    
021    public abstract class CollectorConfig
022    {
023        /**
024         * Configure the type of the eventtracker. Valid values are SCRIBE, COLLECTOR, DUMMY
025         *
026         * @return the type of eventtracker to use
027         */
028        @Config(value = "eventtracker.type")
029        public String getType()
030        {
031            // config-magic doesn't support enums :(
032            return "COLLECTOR";
033        }
034    
035        //------------------- Spooling -------------------//
036    
037        /**
038         * Maximum number of events per file in the temporary spooling area. Past this threshold,
039         * buffered events are promoted to the final spool queue (where they are picked up by the final sender).
040         *
041         * @return the maximum number of events per file
042         * @see com.ning.metrics.serialization.writer.ThresholdEventWriter
043         */
044        @Config(value = "eventtracker.diskspool.flush-event-queue-size")
045        public long getFlushEventQueueSize()
046        {
047            return 10000;
048        }
049    
050        /**
051         * Maxixum number of seconds before events are promoted from the temporary spooling area to the final spool queue.
052         *
053         * @return maxixmum age of events in seconds in the temporary spool queue
054         * @see com.ning.metrics.serialization.writer.ThresholdEventWriter
055         */
056        @Config(value = "eventtracker.diskspool.refresh-delay-seconds")
057        public int getRefreshDelayInSeconds()
058        {
059            return 60;
060        }
061    
062        /**
063         * Directory for the Event Tracker to store events it can not send immediately
064         *
065         * @return the directory path
066         */
067        @Config(value = "eventtracker.diskspool.path")
068        public String getSpoolDirectoryName()
069        {
070            return "/tmp/eventtracker/diskspool";
071        }
072    
073        /**
074         * If false, events will not be periodically sent
075         *
076         * @return whether to send events buffered locally
077         */
078        @Config(value = "eventtracker.diskspool.enabled")
079        public boolean isFlushEnabled()
080        {
081            return true;
082        }
083    
084        /**
085         * Delay between flushes (in seconds)
086         *
087         * @return delay between flushes to the remote server
088         */
089        @Config(value = "eventtracker.diskspool.flush-interval-seconds")
090        public int getFlushIntervalInSeconds()
091        {
092            return 60;
093        }
094    
095        /**
096         * Type of outputter to use when spooling: NONE, FLUSH, or SYNC
097         *
098         * @return the String representation of the SyncType
099         */
100        @Config(value = "eventtracker.diskspool.synctype")
101        public String getSyncType()
102        {
103            return "NONE";
104        }
105    
106        /**
107         * Batch size to use for the outputter.
108         * A flush or sync is triggered after this amount of events have been written.
109         *
110         * @return the batch size for writes
111         */
112        @Config(value = "eventtracker.diskspool.batch-size")
113        public int getSyncBatchSize()
114        {
115            return 50;
116        }
117    
118        @Config(value = "eventtracker.event-end-point.rate-window-size-minutes")
119        public int getRateWindowSizeMinutes()
120        {
121            return 5;
122        }
123    
124        //------------------- Scribe Sender -------------------//
125    
126        /**
127         * Scribe port
128         *
129         * @return the Scribe port to use
130         */
131        @Config(value = "eventtracker.scribe.port")
132        public int getScribePort()
133        {
134            return 1463;
135        }
136    
137        /**
138         * Scribe host
139         *
140         * @return the hostname or IP of the Scribe host to use
141         */
142        @Config(value = "eventtracker.scribe.host")
143        public String getScribeHost()
144        {
145            return "127.0.0.1";
146        }
147    
148        /**
149         * Number of messages to send to Scribe before refreshing the connection
150         *
151         * @return the threshold before reconnecting to Scribe
152         */
153        @Config(value = "eventtracker.scribe.refresh_rate")
154        public int getMessagesToSendBeforeReconnectingToScribe()
155        {
156            return 1000000; // 1 Million
157        }
158    }
