017    package com.ning.metrics.eventtracker;
018    
019    import com.ning.metrics.serialization.event.Event;
020    import com.ning.metrics.serialization.util.FixedManagedJmxExport;
021    import com.ning.metrics.serialization.writer.DiskSpoolEventWriter;
022    import com.ning.metrics.serialization.writer.EventHandler;
023    import com.ning.metrics.serialization.writer.SyncType;
024    import com.ning.metrics.serialization.writer.ThresholdEventWriter;
025    
026    import java.io.IOException;
027    import java.io.ObjectInputStream;
028    import java.util.concurrent.Executors;
029    import java.util.concurrent.ScheduledThreadPoolExecutor;
030    
031    public class ScribeCollectorFactory
032    {
033        private static CollectorController singletonController;
034    
035        private final CollectorController controller;
036        private static ScribeSender eventSender;
037    
038        /**
039         * Initialize the Scribe controller without Guice
040         *
041         * @param scribeHost             Scribe hostname or IP
042         * @param scribePort             Scribe port
043         * @param scribeRefreshRate      Number of messages to send to Scribe before refreshing the connection
044         * @param spoolDirectoryName     Directory name for the spool queue
045         * @param isFlushEnabled         Whether to send events to remote agent
046         * @param flushIntervalInSeconds Delay between flushes (in seconds) to remote agent
047         * @param syncType               type of Sync (NONE, FLUSH or SYNC)
048         * @param rateWindowSizeMinutes  event rate window size
049         * @param flushEventQueueSize    Maximum queue size in the temporary spooling area
050         * @param refreshDelayInSeconds  Number of seconds before promoting events from tmp to final spool queue
051         * @return Scribe controller
052         * @throws java.io.IOException if an Exception occurs while trying to create the directory
053         * @see com.ning.metrics.eventtracker.CollectorController
054         */
055        @SuppressWarnings("unused")
056        public static synchronized CollectorController createScribeController(
057            String scribeHost,
058            int scribePort,
059            int scribeRefreshRate,
060            String spoolDirectoryName,
061            boolean isFlushEnabled,
062            long flushIntervalInSeconds,
063            String syncType,
064            int syncBatchSize,
065            int rateWindowSizeMinutes,
066            long flushEventQueueSize,
067            long refreshDelayInSeconds
068        ) throws IOException
069        {
070            if (singletonController == null) {
071                singletonController = new ScribeCollectorFactory(scribeHost, scribePort, scribeRefreshRate, spoolDirectoryName, isFlushEnabled, flushIntervalInSeconds, syncType, syncBatchSize, rateWindowSizeMinutes, flushEventQueueSize, refreshDelayInSeconds).get();
072            }
073    
074            return singletonController;
075        }
076    
077        ScribeCollectorFactory(
078            String scribeHost,
079            int scribePort,
080            int scribeRefreshRate,
081            String spoolDirectoryName,
082            boolean isFlushEnabled,
083            long flushIntervalInSeconds,
084            String syncType,
085            int syncBatchSize,
086            int rateWindowSizeMinutes,
087            long flushEventQueueSize,
088            long refreshDelayInSeconds
089        ) throws IOException
090        {
091            eventSender = new ScribeSender(new ScribeClientImpl(scribeHost, scribePort), scribeRefreshRate);
092            FixedManagedJmxExport.export("eventtracker:name=ScribeSender", eventSender);
093            eventSender.createConnection();
094    
095            DiskSpoolEventWriter eventWriter = new DiskSpoolEventWriter(new EventHandler()
096            {
097                @Override
098                public void handle(ObjectInputStream objectInputStream) throws ClassNotFoundException, IOException
099                {
100                    while (objectInputStream.read() != -1) {
101                        Event event = (Event) objectInputStream.readObject();
102                        eventSender.send(event);
103                    }
104    
105                    objectInputStream.close();
106                }
107    
108                @Override
109                public void rollback() throws IOException
110                {
111                    // no-op
112                }
113            }, spoolDirectoryName, isFlushEnabled, flushIntervalInSeconds, new ScheduledThreadPoolExecutor(1, Executors.defaultThreadFactory()), SyncType.valueOf(syncType), syncBatchSize, rateWindowSizeMinutes);
114            ThresholdEventWriter thresholdEventWriter = new ThresholdEventWriter(eventWriter, flushEventQueueSize, refreshDelayInSeconds);
115    
116            controller = new CollectorController(thresholdEventWriter);
117            FixedManagedJmxExport.export("eventtracker:name=CollectorController", controller);
118        }
119    
120        private CollectorController get()
121        {
122            return controller;
123        }
124    
125        public static void shutdown()
126        {
127            eventSender.shutdown();
128        }
129    }
