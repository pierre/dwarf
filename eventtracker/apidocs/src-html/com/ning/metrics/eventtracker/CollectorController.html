017    package com.ning.metrics.eventtracker;
018    
019    import com.google.inject.Inject;
020    import com.ning.metrics.serialization.event.Event;
021    import com.ning.metrics.serialization.util.Managed;
022    import com.ning.metrics.serialization.writer.EventWriter;
023    import org.apache.log4j.Logger;
024    
025    import java.io.IOException;
026    import java.util.concurrent.atomic.AtomicLong;
027    
028    /**
029     * Main entry point of the library. The CollectorController is responsible for forwarding events to the persister layer.
030     * <p/>
031     * The class needs to be public for JMX.
032     */
033    public class CollectorController
034    {
035        private static final Logger log = Logger.getLogger(CollectorController.class);
036    
037        private final EventWriter eventWriter;
038    
039        private final AtomicLong eventsReceived = new AtomicLong(0);
040        private final AtomicLong eventsLost = new AtomicLong(0);
041    
042        @Inject
043        public CollectorController(EventWriter eventWriter)
044        {
045            this.eventWriter = eventWriter;
046            log.debug(String.format("Initialized Collector Controller with file manager [%s]", eventWriter));
047        }
048    
049        /**
050         * Offer an event to the queue.
051         *
052         * @param event an event to collect
053         * @throws IOException if a serialization exception (to disk) occurs
054         */
055        public void offerEvent(final Event event) throws IOException
056        {
057            eventsReceived.incrementAndGet();
058    
059            try {
060                log.debug(String.format("Writing event: %s", event));
061                eventWriter.write(event);
062            }
063            catch (IOException e) {
064                log.error(String.format("Failed to write event: %s", event), e);
065                eventsLost.incrementAndGet();
066    
067                throw e;
068            }
069        }
070    
071        @Managed(description = "Number of events received")
072        public AtomicLong getEventsReceived()
073        {
074            return eventsReceived;
075        }
076    
077        @Managed(description = "Number of events lost (unable to serialize them to disk)")
078        public AtomicLong getEventsLost()
079        {
080            return eventsLost;
081        }
082    
083        @Managed(description = "Promote events to final spool area")
084        public void commit() throws IOException
085        {
086            eventWriter.forceCommit();
087        }
088    
089        @Managed(description = "Flush events to remote agent")
090        public void flush() throws IOException
091        {
092            eventWriter.flush();
093        }
094    }
