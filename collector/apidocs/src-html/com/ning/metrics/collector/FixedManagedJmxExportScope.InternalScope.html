001    package com.ning.metrics.collector;
002    
003    import com.google.inject.Key;
004    import com.google.inject.Provider;
005    import com.google.inject.Scope;
006    import com.google.inject.Scopes;
007    import com.ning.metrics.serialization.util.FixedManagedJmxExport;
008    import org.apache.log4j.Logger;
009    
010    /**
011     * A scope to do automatic exporting of injected objects.
012     * <p/>
013     * Use of this scope also binds the object as a singleton by design.
014     */
015    public class FixedManagedJmxExportScope implements Scope
016    {
017        private final InternalScope innerScope;
018    
019        public FixedManagedJmxExportScope(String objectName)
020        {
021            this(Logger.getLogger(FixedManagedJmxExportScope.class), objectName);
022        }
023    
024        public FixedManagedJmxExportScope(Logger log, String objectName)
025        {
026            this.innerScope = new InternalScope(log, objectName);
027        }
028    
029        /**
030         * Scopes a provider. The returned provider returns objects from this scope.
031         * If an object does not exist in this scope, the provider can use the given
032         * unscoped provider to retrieve one.
033         * <p/>
034         * <p>Scope implementations are strongly encouraged to override
035         * {@link Object#toString} in the returned provider and include the backing
036         * provider's {@code toString()} output.
037         *
038         * @param key      binding key
039         * @param unscoped locates an instance when one doesn't already exist in this
040         *                 scope.
041         * @return a new provider which only delegates to the given unscoped provider
042         *         when an instance of the requested object doesn't already exist in this
043         *         scope
044         */
045        @Override
046        public <T> Provider<T> scope(Key<T> key, Provider<T> unscoped)
047        {
048            return Scopes.SINGLETON.scope(key, innerScope.scope(key, unscoped));
049        }
050    
051        public static class InternalScope implements Scope
052        {
053            private final String objectName;
054            private final Logger log;
055    
056            public InternalScope(Logger log, String objectName)
057            {
058                this.objectName = objectName;
059                this.log = log;
060            }
061    
062            @Override
063            public <T> Provider<T> scope(Key<T> tKey, final Provider<T> tProvider)
064            {
065                return new Provider<T>()
066                {
067                    private volatile boolean exported = false;
068    
069                    @Override
070                    public synchronized T get()
071                    {
072                        final T object = tProvider.get();
073                        log.info(String.format("Exporting object[%s] at [%s]", object.getClass(), objectName));
074                        if (!exported) {
075                            FixedManagedJmxExport.export(objectName, object);
076                            exported = true;
077                        }
078                        return object;
079                    }
080                };
081            }
082        }
083    }
