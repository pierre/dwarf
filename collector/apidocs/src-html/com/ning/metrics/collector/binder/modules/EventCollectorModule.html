079    /**
080     * This module contains the minimal plumbing to instantiate the EventCollector.
081     */
082    public class EventCollectorModule implements Module
083    {
084        private static final Logger log = Logger.getLogger(EventCollectorModule.class);
085    
086        @SuppressWarnings("unchecked")
087        @Override
088        public void configure(Binder binder)
089        {
090            CollectorConfig config = new ConfigurationObjectFactory(System.getProperties()).build(CollectorConfig.class);
091            binder.bind(CollectorConfig.class).toInstance(config);
092    
093            if (config.isActiveMQEnabled()) {
094                binder.bind(ActiveMQSender.class)
095                    .to(ActiveMQSenderImpl.class)
096                    .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ActiveMQEventCollectorSender"));
097    
098                binder.bind(ActiveMQController.class)
099                    .to(ActiveMQControllerImpl.class)
100                    .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ActiveMQEventCollector"));
101            }
102            else {
103                binder.bind(ActiveMQSender.class)
104                    .toProvider(Providers.<ActiveMQSender>of(null));
105    
106                binder.bind(ActiveMQController.class)
107                    .toProvider(Providers.<ActiveMQController>of(null));
108            }
109    
110            binder.bind(FileSystem.class)
111                .toProvider(FileSystemProvider.class)
112                .asEagerSingleton();
113    
114            /*
115             * EventWriters
116             */
117            binder.bind(EventWriter.class)
118                .annotatedWith(HdfsEventWriter.class)
119                .to(HadoopFileEventWriter.class)
120                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=HadoopFileEventWriter"));
121    
122            binder.bind(DiskSpoolEventWriter.class)
123                .toProvider(DiskSpoolEventWriterProvider.class)
124                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=DiskSpoolEventWriter"));
125    
126            binder.bind(EventWriter.class)
127                .annotatedWith(BufferingEventCollectorEventWriter.class)
128                .toProvider(ThresholdEventWriterProvider.class)
129                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ThresholdEventWriter"));
130    
131            /*
132             * TaskQueue
133             */
134            binder.bind(TaskQueueService.class)
135                .to(TaskQueueServiceImpl.class);
136            /*
137                * ScheduledExecutorServices
138                */
139            binder.bind(ScheduledExecutorService.class)
140                .annotatedWith(DiskSpoolFlushExecutor.class)
141                .toInstance(new ScheduledThreadPoolExecutor(2, new NamedThreadFactory("spool to HDFS promoter")));
142    
143            binder.bind(ScheduledExecutorService.class)
144                .annotatedWith(BufferingEventCollectorExecutor.class)
145                .toInstance(new ScheduledThreadPoolExecutor(2, new NamedThreadFactory("tmp to spool promoter")));
146    
147            /*
148             * Filters
149             */
150            String filterListDelimiter = config.getFilters();
151    
152            binder.bind(Filter.class)
153                .annotatedWith(EventEndpointRequestFilter.class)
154                .to(OrFilter.class);
155    
156            binder.bind(Filter.class)
157                .annotatedWith(Names.named("host"))
158                .toProvider(new EventFilterProvider(FieldExtractors.HOST, config.getFiltersHost(), filterListDelimiter))
159                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=Host"));
160    
161            binder.bind(Filter.class)
162                .annotatedWith(Names.named("ip"))
163                .toProvider(new EventFilterProvider(FieldExtractors.IP, config.getFiltersIp(), filterListDelimiter))
164                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=IP"));
165    
166            binder.bind(Filter.class)
167                .annotatedWith(Names.named("user-agent"))
168                .toProvider(new EventFilterProvider(FieldExtractors.USERAGENT, config.getFiltersUserAgent(), filterListDelimiter))
169                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=UserAgent"));
170    
171            binder.bind(Filter.class)
172                .annotatedWith(Names.named("path"))
173                .toProvider(new EventFilterProvider(FieldExtractors.PATH, config.getFiltersPath(), filterListDelimiter))
174                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=Path"));
175    
176            binder.bind(Filter.class)
177                .annotatedWith(Names.named("event-type"))
178                .toProvider(new EventFilterProvider(FieldExtractors.EVENT_TYPE, config.getFiltersEventType(), filterListDelimiter))
179                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=EventType"));
180    
181            binder.bind(new TypeLiteral<List<Filter>>()
182            {
183            })
184                .toProvider(new ArrayListProvider<Filter>()
185                    .add(Names.named("host"), Filter.class)
186                    .add(Names.named("ip"), Filter.class)
187                    .add(Names.named("user-agent"), Filter.class)
188                    .add(Names.named("path"), Filter.class)
189                    .add(Names.named("event-type"), Filter.class)
190                )
191                .asEagerSingleton();
192    
193            /*
194             * Request handlers
195             */
196            binder.bind(EventRequestHandler.class)
197                .annotatedWith(ExternalEventRequestHandler.class)
198                .toProvider(new EventRequestHandlerProvider(
199                    QueryParameterEventExtractor.class,
200                    Names.named("base10"),
201                    ExternalEventEndPointStats.class
202                ))
203                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ExternalEventHandler"));
204    
205            binder.bind(ThriftEnvelopeEventParser.class)
206                .annotatedWith(Names.named("base64"))
207                .toProvider(new ThriftEnvelopeEventParserProvider(Base64NumberConverter.class))
208                .in(Scopes.SINGLETON);
209    
210            binder.bind(QueryParameterEventExtractor.class)
211                .annotatedWith(Names.named("base64"))
212                .toProvider(new ExternalEventExtractorProvider(Names.named("base64")))
213                .in(Scopes.SINGLETON);
214    
215            binder.bind(ThriftEnvelopeEventParser.class)
216                .annotatedWith(Names.named("base10"))
217                .toProvider(new ThriftEnvelopeEventParserProvider(DecimalNumberConverter.class))
218                .in(Scopes.SINGLETON);
219    
220            binder.bind(QueryParameterEventExtractor.class)
221                .annotatedWith(Names.named("base10"))
222                .toProvider(new ExternalEventExtractorProvider(Names.named("base10")))
223                .in(Scopes.SINGLETON);
224    
225            binder.bind(EventRequestHandler.class)
226                .annotatedWith(Base64ExternalEventRequestHandler.class)
227                .toProvider(new EventRequestHandlerProvider(
228                    QueryParameterEventExtractor.class,
229                    Names.named("base64"),
230                    ExternalEventEndPointStats.class
231                ))
232                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=Base64ExternalEventHandler"));
233    
234            binder.bind(EventRequestHandler.class)
235                .annotatedWith(InternalEventRequestHandler.class)
236                .toProvider(new EventRequestHandlerProvider(
237                    BodyEventExtractor.class,
238                    InternalEventEndPointStats.class
239                ))
240                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=InternalEventHandler"));
241    
242            binder.bind(EventEndPointStats.class)
243                .annotatedWith(ExternalEventEndPointStats.class)
244                .toProvider(EventEndPointStatsProvider.class)
245                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector:name=ExternalEventEndPointStats"));
246    
247            binder.bind(EventEndPointStats.class)
248                .annotatedWith(InternalEventEndPointStats.class)
249                .toProvider(EventEndPointStatsProvider.class)
250                .in(new FixedManagedJmxExportScope("com.ning.metrics.collector:name=InternalEventEndPointStats"));
251    
252            binder.bind(EventHandler.class)
253                .to(EventHandlerImpl.class)
254                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=EventHandler"));
255    
256            /*
257            * Final wiring
258            */
259            binder.bind(BufferingEventCollector.class)
260                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=BufferingEventCollector"));
261    
262            binder.bind(EventCollector.class)
263                .to(BufferingEventCollector.class);
264    
265            binder.bind(F5PoolMemberControl.class)
266                .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=F5poolMemberControl"));
267        }
268    }
