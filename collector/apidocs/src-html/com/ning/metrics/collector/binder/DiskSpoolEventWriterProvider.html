001    package com.ning.metrics.collector.binder;
002    
003    import com.google.inject.Inject;
004    import com.google.inject.Injector;
005    import com.google.inject.Provider;
006    import com.ning.metrics.collector.binder.annotations.DiskSpoolFlushExecutor;
007    import com.ning.metrics.collector.binder.annotations.HdfsEventWriter;
008    import com.ning.metrics.collector.binder.config.CollectorConfig;
009    import com.ning.metrics.serialization.event.Event;
010    import com.ning.metrics.serialization.writer.DiskSpoolEventWriter;
011    import com.ning.metrics.serialization.writer.EventHandler;
012    import com.ning.metrics.serialization.writer.EventWriter;
013    import com.ning.metrics.serialization.writer.SyncType;
014    
015    import java.io.IOException;
016    import java.io.ObjectInputStream;
017    import java.util.concurrent.ScheduledExecutorService;
018    
019    public class DiskSpoolEventWriterProvider implements Provider<DiskSpoolEventWriter>
020    {
021        private final Injector injector;
022        private final EventWriter hadoopEventWriter;
023        private final ScheduledExecutorService executor;
024    
025        @Inject
026        public DiskSpoolEventWriterProvider(
027            Injector injector,
028            @HdfsEventWriter EventWriter hadoopEventWriter,
029            @DiskSpoolFlushExecutor ScheduledExecutorService executor
030        )
031        {
032            this.injector = injector;
033            this.hadoopEventWriter = hadoopEventWriter;
034            this.executor = executor;
035        }
036    
037        /**
038         * Provides an instance of {@code T}. Must never return {@code null}.
039         *
040         * @throws com.google.inject.OutOfScopeException
041         *          when an attempt is made to access a scoped object while the scope
042         *          in question is not currently active
043         * @throws com.google.inject.ProvisionException
044         *          if an instance cannot be provided. Such exceptions include messages
045         *          and throwables to describe why provision failed.
046         */
047        @Override
048        public DiskSpoolEventWriter get()
049        {
050            CollectorConfig config = injector.getInstance(CollectorConfig.class);
051            return new DiskSpoolEventWriter(new EventHandler()
052            {
053                @Override
054                public void handle(ObjectInputStream objectInputStream) throws ClassNotFoundException, IOException
055                {
056                    while (objectInputStream.read() != -1) {
057                        Event event = (Event) objectInputStream.readObject();
058                        hadoopEventWriter.write(event);
059                    }
060    
061                    objectInputStream.close();
062                    hadoopEventWriter.forceCommit();
063                }
064    
065                @Override
066                public void rollback() throws IOException
067                {
068                    hadoopEventWriter.rollback();
069                }
070    
071                @Override
072                public String toString()
073                {
074                    return hadoopEventWriter.toString();
075                }
076            }, config.getSpoolDirectoryName(), config.isFlushEnabled(), config.getFlushIntervalInSeconds(), executor, SyncType.valueOf(config.getSyncType()), config.getRateWindowSizeMinutes());
077        }
078    }
