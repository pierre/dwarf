017    package com.ning.metrics.collector.binder;
018    
019    import com.google.inject.Inject;
020    import com.google.inject.Provider;
021    import com.ning.metrics.collector.endpoint.filters.FieldExtractor;
022    import com.ning.metrics.collector.endpoint.filters.PatternSetFilter;
023    
024    import java.util.HashSet;
025    import java.util.Set;
026    import java.util.regex.Pattern;
027    
028    public class EventFilterProvider implements Provider<PatternSetFilter>
029    {
030        private final FieldExtractor fieldExtractor;
031        private final Set<Pattern> patternSet = new HashSet<Pattern>();
032    
033        @Inject
034        public EventFilterProvider(FieldExtractor fieldExtractor, String patternListString, String delimiter)
035        {
036            this.fieldExtractor = fieldExtractor;
037    
038            if (!patternListString.isEmpty()) {
039                for (String str : patternListString.split(delimiter)) {
040                    patternSet.add(Pattern.compile(str));
041                }
042            }
043        }
044    
045        @Override
046        public PatternSetFilter get()
047        {
048            PatternSetFilter filter = new PatternSetFilter(fieldExtractor, patternSet);
049    
050            return filter;
051        }
052    }
