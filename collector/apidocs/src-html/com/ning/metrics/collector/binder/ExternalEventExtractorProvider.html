017    package com.ning.metrics.collector.binder;
018    
019    import com.google.inject.Inject;
020    import com.google.inject.Injector;
021    import com.google.inject.Key;
022    import com.google.inject.Provider;
023    import com.ning.metrics.collector.endpoint.extractors.QueryParameterEventExtractor;
024    import com.ning.metrics.collector.events.parsing.ThriftEnvelopeEventParser;
025    
026    import java.lang.annotation.Annotation;
027    
028    public class ExternalEventExtractorProvider implements Provider<QueryParameterEventExtractor>
029    {
030        private Injector injector;
031        private Key<ThriftEnvelopeEventParser> thriftEnvelopeEventParserKey;
032    
033        public ExternalEventExtractorProvider(Annotation thriftEnvelopeEventParserAnnotation)
034        {
035            this.thriftEnvelopeEventParserKey = Key.get(ThriftEnvelopeEventParser.class, thriftEnvelopeEventParserAnnotation);
036        }
037    
038        public ExternalEventExtractorProvider(Class<? extends Annotation> thriftEnvelopeEventParserAnnotation)
039        {
040            this.thriftEnvelopeEventParserKey = Key.get(ThriftEnvelopeEventParser.class, thriftEnvelopeEventParserAnnotation);
041        }
042    
043        @Inject
044        public void configure(Injector injector)
045        {
046            this.injector = injector;
047        }
048    
049        @Override
050        public QueryParameterEventExtractor get()
051        {
052            return new QueryParameterEventExtractor(injector.getInstance(thriftEnvelopeEventParserKey));
053        }
054    }
