017    package com.ning.metrics.collector.binder;
018    
019    import com.google.inject.Inject;
020    import com.google.inject.Provider;
021    import com.ning.metrics.collector.binder.config.CollectorConfig;
022    import com.ning.metrics.serialization.hadoop.HadoopThriftEnvelopeSerialization;
023    import com.ning.metrics.serialization.hadoop.HadoopThriftWritableSerialization;
024    import org.apache.hadoop.conf.Configuration;
025    import org.apache.hadoop.fs.FileSystem;
026    
027    import java.io.IOException;
028    
029    public class FileSystemProvider implements Provider<FileSystem>
030    {
031        private final FileSystem fileSystem;
032    
033        @Inject
034        public FileSystemProvider(CollectorConfig collectorConfig) throws IOException
035        {
036            String hfsHost = collectorConfig.getHfsHost();
037            Configuration hadoopConfig = new Configuration();
038    
039            if (hfsHost.isEmpty()) {
040                // Local filesystem, for testing
041                hadoopConfig.set("fs.default.name", "file:///");
042            }
043            else {
044                hadoopConfig.set("fs.default.name", hfsHost);
045            }
046    
047            hadoopConfig.setLong("dfs.block.size", collectorConfig.getHadoopBlockSize());
048            hadoopConfig.set("hadoop.job.ugi", collectorConfig.getHadoopUgi());
049            hadoopConfig.setStrings("io.serializations", HadoopThriftWritableSerialization.class.getName(), HadoopThriftEnvelopeSerialization.class.getName(), "org.apache.hadoop.io.serializer.WritableSerialization");
050    
051            fileSystem = FileSystem.get(hadoopConfig);
052        }
053    
054        public FileSystem get()
055        {
056            return fileSystem;
057        }
058    }
