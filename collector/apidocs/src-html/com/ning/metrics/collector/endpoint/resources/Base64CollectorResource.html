017    package com.ning.metrics.collector.endpoint.resources;
018    
019    import com.google.inject.Inject;
020    import com.ning.metrics.collector.binder.annotations.Base64ExternalEventRequestHandler;
021    import com.ning.metrics.collector.endpoint.EventStats;
022    import com.ning.metrics.collector.events.parsing.EventExtractorUtil;
023    import com.ning.metrics.collector.events.parsing.ParsedRequest;
024    import com.ning.metrics.serialization.event.Granularity;
025    import org.joda.time.DateTime;
026    
027    import javax.ws.rs.GET;
028    import javax.ws.rs.Path;
029    import javax.ws.rs.Produces;
030    import javax.ws.rs.QueryParam;
031    import javax.ws.rs.core.Context;
032    import javax.ws.rs.core.HttpHeaders;
033    import javax.ws.rs.core.MediaType;
034    import javax.ws.rs.core.Response;
035    
036    /**
037     * Version 2 of the collector's external API encodes numbers as in a custom base64 encoding.
038     */
039    @Path("/2")
040    public class Base64CollectorResource
041    {
042        private final EventRequestHandler requestHandler;
043        private final EventExtractorUtil extractorUtil;
044    
045        @Inject
046        public Base64CollectorResource(
047            @Base64ExternalEventRequestHandler EventRequestHandler requestHandler,
048            EventExtractorUtil extractorUtil
049        )
050        {
051            this.requestHandler = requestHandler;
052            this.extractorUtil = extractorUtil;
053        }
054    
055        @GET
056        @Produces(MediaType.TEXT_PLAIN)
057        public Response get(
058            @QueryParam("v") String event,
059            @QueryParam("date") String eventDateTimeString,
060            @QueryParam(Granularity.GRANULARITY_QUERY_PARAM) String eventGranularity,
061            @Context HttpHeaders httpHeaders
062        )
063        {
064            EventStats eventStats = new EventStats();
065            DateTime eventDateTime = new DateTime(eventDateTimeString);
066            return requestHandler.handleEventRequest(event, new ParsedRequest(httpHeaders, eventDateTime, eventGranularity, extractorUtil), eventStats);
067        }
068    }
