017    package com.ning.metrics.collector.endpoint;
018    
019    import com.google.inject.Inject;
020    import com.ning.metrics.serialization.util.Managed;
021    import com.ning.metrics.serialization.writer.EventRate;
022    import org.apache.log4j.Logger;
023    import org.joda.time.Period;
024    
025    import com.ning.metrics.collector.binder.config.CollectorConfig;
026    import com.ning.metrics.serialization.event.Event;
027    
028    import java.util.ArrayList;
029    import java.util.List;
030    import java.util.Map;
031    import java.util.concurrent.ConcurrentHashMap;
032    import java.util.concurrent.ConcurrentMap;
033    import java.util.concurrent.atomic.AtomicLong;
034    
035    public class EventEndPointStats
036    {
037            private static final int MAX_EVENT_TYPES = 100000;
038            private static final Logger log  = Logger.getLogger(EventEndPointStats.class);
039    
040            private final AtomicLong totalEvents = new AtomicLong(0);
041            private final AtomicLong filteredEvents = new AtomicLong(0);
042            private final AtomicLong succesfulEvents = new AtomicLong(0);
043            private final AtomicLong failedEvents = new AtomicLong(0);
044            private final AtomicLong rejectedEvents  = new AtomicLong(0);
045            private final ConcurrentMap<String, AtomicLong> successfulEventsByType = new ConcurrentHashMap<String, AtomicLong>();
046            private final EventRate eventParseRate;
047            private final EventRate successfulEventParseRate;
048            private final EventRate failedEventParseRate;
049            private final EventRate filteredEventParseRate;
050            private final EventRate rejectedEventRate;
051            private final int rateWindowSizeMinutes;
052    
053            @Inject
054        public EventEndPointStats(CollectorConfig config)
055        {
056            this(config.getRateWindowSizeMinutes());
057        }
058    
059            public EventEndPointStats(int rateWindowSizeMinutes)
060            {
061                    this.rateWindowSizeMinutes = rateWindowSizeMinutes;
062                    eventParseRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
063                    successfulEventParseRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
064                    failedEventParseRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
065                    filteredEventParseRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
066                    rejectedEventRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
067            }
068    
069            public void updateTotalEvents()
070            {
071                    totalEvents.incrementAndGet();
072                    eventParseRate.increment();
073            }
074    
075            public void updateSuccessfulEvents()
076            {
077                    succesfulEvents.incrementAndGet();
078                    successfulEventParseRate.increment();
079            }
080    
081            public void updateSuccesfulEventCounters(Event event)
082            {
083                    String eventTypeString = String.format("%s(%s)", event.getName(), event.getVersion());
084    
085                    updateSuccessfulEvents();
086                    updateSuccessfulEventsByType(eventTypeString);
087            }
088    
089            public void updateSuccessfulEventsByType(String eventType)
090            {
091                    AtomicLong zeroCounter = new AtomicLong(0);
092                    AtomicLong counter = successfulEventsByType.putIfAbsent(eventType, zeroCounter);
093    
094                    if (counter == null) {
095                            zeroCounter.incrementAndGet();
096                    }
097                    else {
098                            counter.incrementAndGet();
099                    }
100    
101                    if (successfulEventsByType.size() > MAX_EVENT_TYPES) {
102                            successfulEventsByType.clear();
103                            log.info(String.format("max number of distinct event types seen (%d),  clearing successful stats by event type", MAX_EVENT_TYPES));
104                    }
105            }
106    
107            public void updateFailedEvents()
108            {
109                    failedEvents.incrementAndGet();
110                    failedEventParseRate.increment();
111            }
112    
113            public void updateFilteredEvents()
114            {
115                    filteredEvents.incrementAndGet();
116                    filteredEventParseRate.increment();
117            }
118    
119            public void updateRejectedEvents()
120            {
121                    rejectedEvents.incrementAndGet();
122                    rejectedEventRate.increment();
123            }
124    
125            @Managed(description = "total events received while collector enabled")
126            public long getTotalEvents()
127            {
128                    return totalEvents.get();
129            }
130    
131            @Managed(description = "events received while collector enabled that do not pass any filters")
132            public long getFilteredEvents()
133            {
134                    return filteredEvents.get();
135            }
136    
137            @Managed(description = "events rejected due to collector buffer piling up")
138            public long getRejectedEvents()
139            {
140                    return rejectedEvents.get();
141            }
142    
143            @Managed(description = "events received while enabled that pass filters, but failed to parse")
144            public long getFailedToParseEvents()
145            {
146                    return failedEvents.get();
147            }
148    
149            @Managed(description = "events received while enabled that pass filters and parse successfully")
150            public long getSuccessfulParseEvents()
151            {
152                    return succesfulEvents.get();
153            }
154    
155            @Managed
156            public List<String> getSuccessfulParseEventsByType()
157            {
158                    List<String> resultStringList = new ArrayList<String>();
159                    
160                    for (Map.Entry<String, AtomicLong> entry : successfulEventsByType.entrySet()) {
161                            resultStringList.add(String.format("%s = %d", entry.getKey(), entry.getValue().get()));
162                    }
163    
164                    return resultStringList;
165            }
166    
167            @Managed(description = "succesful event parse rate per minute")
168            public long getSuccessfulEventsPerMinute()
169            {
170                    return successfulEventParseRate.getRate() / rateWindowSizeMinutes;
171            }
172    
173            @Managed(description = "failed event parse rate per minute")
174            public long getFailedEventsPerMinute()
175            {
176                    return failedEventParseRate.getRate() / rateWindowSizeMinutes;
177            }
178    
179            @Managed(description = "filtered event parse rate per minute")
180            public long getFilteredEventsPerMinute()
181            {
182                    return filteredEventParseRate.getRate() / rateWindowSizeMinutes;
183            }
184    
185            @Managed(description = "rejected event parse rate per minute")
186            public long getRejectedEventsPerMinute()
187            {
188                    return rejectedEventRate.getRate() / rateWindowSizeMinutes;
189            }
190    
191            @Managed(description = "total event parse rate per minute")
192            public long getEventsPerMinute()
193            {
194                    return eventParseRate.getRate() / rateWindowSizeMinutes;
195            }
196    }
