017    package com.ning.metrics.collector.endpoint.setup;
018    
019    import org.apache.commons.lang.ArrayUtils;
020    import org.slf4j.bridge.SLF4JBridgeHandler;
021    
022    import javax.servlet.ServletContextEvent;
023    import javax.servlet.ServletContextListener;
024    import java.util.logging.Handler;
025    import java.util.logging.LogManager;
026    import java.util.logging.Logger;
027    
028    /**
029     * Takes java.util.logging and redirects it into log4j.
030     */
031    public class SetupJULBridge implements ServletContextListener
032    {
033        private static final org.apache.log4j.Logger log = org.apache.log4j.Logger.getLogger(SetupJULBridge.class);
034    
035        @Override
036        public void contextInitialized(final ServletContextEvent event)
037        {
038            // we first remove the default handler(s)
039            final Logger rootLogger = LogManager.getLogManager().getLogger("");
040            final Handler[] handlers = rootLogger.getHandlers();
041    
042            if (!ArrayUtils.isEmpty(handlers)) {
043                for (Handler handler : handlers) {
044                    rootLogger.removeHandler(handler);
045                }
046            }
047            // and then we let jul-to-sfl4j do its magic so that jersey messages go to sfl4j (and thus log4j)
048            SLF4JBridgeHandler.install();
049    
050            log.info("Assimilated java.util Logging");
051        }
052    
053        @Override
054        public void contextDestroyed(final ServletContextEvent event)
055        {
056            SLF4JBridgeHandler.uninstall();
057        }
058    }
