017    package com.ning.metrics.collector.endpoint.servers;
018    
019    import com.ning.metrics.collector.util.NamedThreadFactory;
020    import org.apache.thrift.TProcessor;
021    import org.apache.thrift.TProcessorFactory;
022    import org.apache.thrift.protocol.TBinaryProtocol;
023    import org.apache.thrift.protocol.TProtocolFactory;
024    import org.apache.thrift.server.TNonblockingServer;
025    import org.apache.thrift.transport.TFramedTransport;
026    import org.apache.thrift.transport.TNonblockingServerTransport;
027    
028    import java.util.concurrent.ExecutorService;
029    import java.util.concurrent.LinkedBlockingQueue;
030    import java.util.concurrent.ThreadPoolExecutor;
031    import java.util.concurrent.TimeUnit;
032    
033    /**
034     * An extension of the TNonblockingServer to a Half-Sync/Half-Async server.
035     * Like TNonblockingServer, it relies on the use of TFramedTransport.
036     */
037    public class CollectorTHsHaServer extends TNonblockingServer
038    {
039    
040        // This wraps all the functionality of queueing and thread pool management
041        // for the passing of Invocations from the Selector to workers.
042        private ExecutorService invoker;
043    
044        protected final int MIN_WORKER_THREADS;
045        protected final int MAX_WORKER_THREADS;
046        protected final int STOP_TIMEOUT_VAL;
047        protected final TimeUnit STOP_TIMEOUT_UNIT;
048    
049        /**
050         * Create server with given processor, and server transport. Default server
051         * options, TBinaryProtocol for the protocol, and TFramedTransport.Factory on
052         * both input and output transports. A TProcessorFactory will be created that
053         * always returns the specified processor.
054         */
055        public CollectorTHsHaServer(TProcessor processor,
056                                    TNonblockingServerTransport serverTransport)
057        {
058            this(processor, serverTransport, new Options());
059        }
060    
061        /**
062         * Create server with given processor, server transport, and server options
063         * using TBinaryProtocol for the protocol, and TFramedTransport.Factory on
064         * both input and output transports. A TProcessorFactory will be created that
065         * always returns the specified processor.
066         */
067        public CollectorTHsHaServer(TProcessor processor,
068                                    TNonblockingServerTransport serverTransport,
069                                    Options options)
070        {
071            this(new TProcessorFactory(processor), serverTransport, options);
072        }
073    
074        /**
075         * Create server with specified processor factory, server transport, and server
076         * options. TBinaryProtocol is assumed. TFramedTransport.Factory is used on
077         * both input and output transports.
078         */
079        public CollectorTHsHaServer(TProcessorFactory processorFactory,
080                                    TNonblockingServerTransport serverTransport,
081                                    Options options)
082        {
083            this(processorFactory, serverTransport, new TFramedTransport.Factory(),
084                new TBinaryProtocol.Factory(), options);
085        }
086    
087        /**
088         * Create server with specified processor factory, server transport, in/out
089         * transport factory, in/out protocol factory, and server options.
090         */
091        public CollectorTHsHaServer(TProcessorFactory processorFactory,
092                                    TNonblockingServerTransport serverTransport,
093                                    TFramedTransport.Factory transportFactory,
094                                    TProtocolFactory protocolFactory,
095                                    Options options)
096        {
097            this(processorFactory, serverTransport,
098                transportFactory, transportFactory,
099                protocolFactory, protocolFactory,
100                options);
101        }
102    
103        /**
104         * Create server with every option fully specified.
105         */
106        public CollectorTHsHaServer(TProcessorFactory processorFactory,
107                                    TNonblockingServerTransport serverTransport,
108                                    TFramedTransport.Factory inputTransportFactory,
109                                    TFramedTransport.Factory outputTransportFactory,
110                                    TProtocolFactory inputProtocolFactory,
111                                    TProtocolFactory outputProtocolFactory,
112                                    Options options)
113        {
114            super(processorFactory, serverTransport,
115                inputTransportFactory, outputTransportFactory,
116                inputProtocolFactory, outputProtocolFactory,
117                options);
118    
119            MIN_WORKER_THREADS = options.minWorkerThreads;
120            MAX_WORKER_THREADS = options.maxWorkerThreads;
121            STOP_TIMEOUT_VAL = options.stopTimeoutVal;
122            STOP_TIMEOUT_UNIT = options.stopTimeoutUnit;
123        }
124    
125        /**
126         * @inheritDoc
127         */
128        @Override
129        public void serve()
130        {
131            if (!startInvokerPool()) {
132                return;
133            }
134    
135            // start listening, or exit
136            if (!startListening()) {
137                return;
138            }
139    
140            // start the selector, or exit
141            if (!startSelectorThread()) {
142                return;
143            }
144    
145            // this will block while we serve
146            joinSelector();
147    
148            gracefullyShutdownInvokerPool();
149    
150            // do a little cleanup
151            stopListening();
152    
153            // ungracefully shut down the invoker pool?
154        }
155    
156        protected boolean startInvokerPool()
157        {
158            // start the invoker pool
159            LinkedBlockingQueue<Runnable> queue = new LinkedBlockingQueue<Runnable>();
160            invoker = new ThreadPoolExecutor(MIN_WORKER_THREADS, MAX_WORKER_THREADS,
161                STOP_TIMEOUT_VAL, STOP_TIMEOUT_UNIT, queue, new NamedThreadFactory("CollectorTHsHaServer worker"));
162    
163            return true;
164        }
165    
166        protected void gracefullyShutdownInvokerPool()
167        {
168            // try to gracefully shut down the executor service
169            invoker.shutdown();
170    
171            // Loop until awaitTermination finally does return without a interrupted
172            // exception. If we don't do this, then we'll shut down prematurely. We want
173            // to let the executorService clear it's task queue, closing client sockets
174            // appropriately.
175            long timeoutMS = 10000;
176            long now = System.currentTimeMillis();
177            while (timeoutMS >= 0) {
178                try {
179                    invoker.awaitTermination(timeoutMS, TimeUnit.MILLISECONDS);
180                    break;
181                }
182                catch (InterruptedException ix) {
183                    long newnow = System.currentTimeMillis();
184                    timeoutMS -= (newnow - now);
185                    now = newnow;
186                }
187            }
188        }
189    
190        /**
191         * We override the standard invoke method here to queue the invocation for
192         * invoker service instead of immediately invoking. The thread pool takes care of the rest.
193         */
194        @Override
195        protected void requestInvoke(FrameBuffer frameBuffer)
196        {
197            invoker.execute(new Invocation(frameBuffer));
198        }
199    
200        /**
201         * An Invocation represents a method call that is prepared to execute, given
202         * an idle worker thread. It contains the input and output protocols the
203         * thread's processor should use to perform the usual Thrift invocation.
204         */
205        private class Invocation implements Runnable
206        {
207    
208            private final FrameBuffer frameBuffer;
209    
210            public Invocation(final FrameBuffer frameBuffer)
211            {
212                this.frameBuffer = frameBuffer;
213            }
214    
215            public void run()
216            {
217                frameBuffer.invoke();
218            }
219        }
220    
221        public static class Options extends TNonblockingServer.Options
222        {
223            public int minWorkerThreads = 5;
224            public int maxWorkerThreads = Integer.MAX_VALUE;
225            public int stopTimeoutVal = 60;
226            public TimeUnit stopTimeoutUnit = TimeUnit.SECONDS;
227        }
228    }
