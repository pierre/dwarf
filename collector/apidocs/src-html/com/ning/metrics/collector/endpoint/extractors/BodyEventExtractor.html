017    package com.ning.metrics.collector.endpoint.extractors;
018    
019    import com.ning.metrics.serialization.thrift.ThriftFieldListParser;
020    import com.ning.metrics.collector.events.parsing.EventParsingException;
021    import com.ning.metrics.collector.events.parsing.ExtractedAnnotation;
022    import com.ning.metrics.serialization.event.Event;
023    import com.ning.metrics.serialization.event.ThriftEnvelopeEvent;
024    import com.ning.metrics.serialization.thrift.ThriftEnvelope;
025    import com.ning.metrics.serialization.thrift.ThriftField;
026    import org.apache.log4j.Logger;
027    
028    import java.io.IOException;
029    import java.util.List;
030    
031    public class BodyEventExtractor implements EventExtractor
032    {
033        private static final Logger log = Logger.getLogger(BodyEventExtractor.class);
034    
035        @Override
036        public Event extractEvent(String eventType, ExtractedAnnotation annotation) throws EventParsingException
037        {
038            if (eventType != null) {
039                log.debug(String.format("receiving event of type %s", eventType));
040                List<ThriftField> thriftFieldList;
041    
042                try {
043                    thriftFieldList = new ThriftFieldListParser().parse(annotation.getContentLength(), annotation.getInputStream());
044                }
045                catch (IOException e) {
046                    throw new EventParsingException(String.format("IOException while trying to parse event from post body"), e);
047                }
048                catch (IllegalArgumentException e) {
049                    throw new EventParsingException(String.format("Parse exception while trying to parse event from post body"), e);
050                }
051    
052                ThriftEnvelope envelope = new ThriftEnvelope(eventType, thriftFieldList);
053    
054                return new ThriftEnvelopeEvent(annotation.getDateTime(), envelope, annotation.getBucketGranularity());
055            }
056    
057            return null;
058        }
059    }
