017    package com.ning.metrics.collector.endpoint.servers;
018    
019    import com.google.inject.Inject;
020    import com.google.inject.Singleton;
021    import com.google.inject.servlet.GuiceFilter;
022    import org.apache.log4j.Logger;
023    import org.mortbay.jetty.Connector;
024    import org.mortbay.jetty.Handler;
025    import org.mortbay.jetty.Server;
026    import org.mortbay.jetty.bio.SocketConnector;
027    import org.mortbay.jetty.security.SslSocketConnector;
028    import org.mortbay.jetty.servlet.Context;
029    import org.mortbay.jetty.servlet.DefaultServlet;
030    import org.mortbay.jetty.servlet.FilterHolder;
031    import org.mortbay.jetty.servlet.ServletHolder;
032    
033    import com.ning.metrics.collector.binder.config.CollectorConfig;
034    import com.ning.metrics.collector.binder.modules.JettyListener;
035    
036    import java.util.concurrent.Executors;
037    import java.util.concurrent.ScheduledExecutorService;
038    import java.util.concurrent.ScheduledThreadPoolExecutor;
039    
040    @Singleton
041    public class JettyServer
042    {
043        private final static Logger log = Logger.getLogger(JettyServer.class);
044        private final String ip;
045        private final int port;
046        private final int sslPort;
047        private final boolean sslEnabled;
048        private final String keystoreLocation;
049        private final String keystorePassword;
050    
051        @Inject
052        @SuppressWarnings("unused")
053        public JettyServer(
054            CollectorConfig config
055        )
056        {
057            this(config.getLocalIp(), config.getLocalPort(), config.isSSLEnabled(), config.getLocalSSLPort(), config.getSSLkeystoreLocation(), config.getSSLkeystorePassword());
058        }
059    
060        public JettyServer(
061            String ip,
062            int port,
063            boolean sslEnabled,
064            int sslPort,
065            String keystoreLocation,
066            String keystorePassword
067        )
068        {
069            this.ip = ip;
070            this.port = port;
071            this.sslEnabled = sslEnabled;
072            this.sslPort = sslPort;
073            this.keystoreLocation = keystoreLocation;
074            this.keystorePassword = keystorePassword;
075    
076            ScheduledExecutorService executor = new ScheduledThreadPoolExecutor(1, Executors.defaultThreadFactory());
077            executor.execute(new Runnable()
078            {
079                @Override
080                public void run()
081                {
082                    try {
083                        start();
084                    }
085                    catch (Exception e) {
086                        log.warn("Unable to start the Jetty server", e);
087                        Thread.currentThread().interrupt();
088                    }
089                }
090            });
091        }
092    
093        private void start() throws Exception
094        {
095            final long startTime = System.currentTimeMillis();
096    
097            final Server server = new Server();
098    
099            Connector connector = new SocketConnector();
100            connector.setHost(ip);
101            connector.setPort(port);
102            server.addConnector(connector);
103    
104            if (sslEnabled) {
105                SslSocketConnector sslConnector = new SslSocketConnector();
106                sslConnector.setPort(sslPort);
107                sslConnector.setKeystore(keystoreLocation);
108                sslConnector.setKeyPassword(keystorePassword);
109                sslConnector.setPassword(keystorePassword);
110                server.addConnector(sslConnector);
111            }
112    
113            server.setStopAtShutdown(true);
114    
115            Context context = new Context(server, "/", Context.SESSIONS);
116            context.addEventListener(new JettyListener());
117    
118            /* Make sure Guice filter all requests */
119            FilterHolder filterHolder = new FilterHolder(GuiceFilter.class);
120            context.addFilter(filterHolder, "/*", Handler.DEFAULT);
121    
122            ServletHolder sh = new ServletHolder(DefaultServlet.class);
123            sh.setInitParameter("com.sun.jersey.config.property.resourceConfigClass", "com.sun.jersey.api.core.PackagesResourceConfig");
124            sh.setInitParameter("com.sun.jersey.config.property.packages", "com.ning.metrics.collector.endpoint");
125            context.addServlet(sh, "/*");
126    
127            server.start();
128    
129            final long secondsToStart = (System.currentTimeMillis() - startTime) / 1000;
130            log.info(String.format("Jetty server started in %d:%02d", secondsToStart / 60, secondsToStart % 60));
131        }
132    }
