017    package com.ning.metrics.collector.endpoint.extractors;
018    
019    import com.google.inject.Inject;
020    import org.apache.log4j.Logger;
021    
022    import com.ning.metrics.serialization.event.Event;
023    import com.ning.metrics.collector.events.parsing.EventParser;
024    import com.ning.metrics.collector.events.parsing.EventParsingException;
025    import com.ning.metrics.collector.events.parsing.ExtractedAnnotation;
026    
027    /**
028     * API versions 1 and 2: query parameters-based API (via GET)
029     */
030    public class QueryParameterEventExtractor implements EventExtractor
031    {
032        private static final Logger log = Logger.getLogger(QueryParameterEventExtractor.class);
033        private final EventParser thriftEventParser;
034    
035        @Inject
036        public QueryParameterEventExtractor(EventParser thriftEventParser)
037        {
038            this.thriftEventParser = thriftEventParser;
039        }
040    
041        @Override
042        public Event extractEvent(String event, ExtractedAnnotation annotation) throws EventParsingException
043        {
044            if (event != null) {
045                log.debug(String.format("Query parameter to process: %s", event));
046                String type = event.substring(0, event.indexOf(","));
047                String eventTypeString = event.substring(event.indexOf(",") + 1);
048    
049                log.debug(String.format("Event type [%s], event string [%s]", type, eventTypeString));
050    
051                return thriftEventParser.parseThriftEvent(type, eventTypeString, annotation);
052            }
053            else {
054                return null;
055            }
056        }
057    }
