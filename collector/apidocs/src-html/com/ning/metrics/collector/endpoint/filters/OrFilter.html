017    package com.ning.metrics.collector.endpoint.filters;
018    
019    import com.google.inject.Inject;
020    
021    import com.ning.metrics.collector.events.parsing.ExtractedAnnotation;
022    import com.ning.metrics.collector.util.Filter;
023    
024    import java.util.List;
025    
026    public class OrFilter implements Filter<ExtractedAnnotation>
027    {
028        private final List<Filter> filterList;
029    
030        @Inject
031        public OrFilter(List<Filter> filterList)
032        {
033            this.filterList = filterList;
034        }
035    
036        @Override
037        public boolean passesFilter(String eventName, ExtractedAnnotation annotation)
038        {
039            for (Filter filter : filterList) {
040                if (filter.passesFilter(eventName, annotation)) {
041                    return true;
042                }
043            }
044    
045            return false;
046        }
047    }
