001    package com.ning.metrics.collector.endpoint;
002    
003    /**
004     * Track stats on a per event basis.
005     */
006    public class EventStats
007    {
008        private final long receivedTimeStamp;
009        private long acceptedTimeStamp;
010    
011        private long extractedTimeStamp;
012    
013        public EventStats()
014        {
015            receivedTimeStamp = System.nanoTime();
016        }
017    
018        /**
019         * Set the timestamp when the event was accepted.
020         * This is set after the event has been processed and is ready to be flushed to disk.
021         */
022        public void recordAccepted()
023        {
024            acceptedTimeStamp = System.nanoTime();
025        }
026    
027        /**
028         * Set the timestamp when the event has been extracted from its original payload.
029         * This can means a bunch of things depending on the API.
030         */
031        public void recordExtracted()
032        {
033            extractedTimeStamp = System.nanoTime();
034        }
035    
036        public double getAcceptedDelayMillis()
037        {
038            return (acceptedTimeStamp - receivedTimeStamp) / 1000000.0;
039        }
040    
041        public double getExtractedDelayMillis()
042        {
043            return (acceptedTimeStamp - extractedTimeStamp) / 1000000.0;
044        }
045    }
