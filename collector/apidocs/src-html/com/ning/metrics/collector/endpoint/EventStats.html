017    package com.ning.metrics.collector.endpoint;
018    
019    /**
020     * Track stats on a per event basis.
021     */
022    public class EventStats
023    {
024        private final long receivedTimeStamp;
025        private long acceptedTimeStamp;
026    
027        private long extractedTimeStamp;
028    
029        public EventStats()
030        {
031            receivedTimeStamp = System.nanoTime();
032        }
033    
034        /**
035         * Set the timestamp when the event was accepted.
036         * This is set after the event has been processed and is ready to be flushed to disk.
037         */
038        public void recordAccepted()
039        {
040            acceptedTimeStamp = System.nanoTime();
041        }
042    
043        /**
044         * Set the timestamp when the event has been extracted from its original payload.
045         * This can means a bunch of things depending on the API.
046         */
047        public void recordExtracted()
048        {
049            extractedTimeStamp = System.nanoTime();
050        }
051    
052        public double getAcceptedDelayMillis()
053        {
054            return (acceptedTimeStamp - receivedTimeStamp) / 1000000.0;
055        }
056    
057        public double getExtractedDelayMillis()
058        {
059            return (acceptedTimeStamp - extractedTimeStamp) / 1000000.0;
060        }
061    }
