017    package com.ning.metrics.collector.endpoint.filters;
018    
019    import com.google.inject.Inject;
020    
021    import com.ning.metrics.serialization.util.Managed;
022    import com.ning.metrics.collector.events.parsing.ExtractedAnnotation;
023    import com.ning.metrics.collector.util.Filter;
024    
025    import java.util.ArrayList;
026    import java.util.List;
027    import java.util.Set;
028    import java.util.concurrent.ConcurrentHashMap;
029    import java.util.concurrent.ConcurrentMap;
030    import java.util.regex.Pattern;
031    
032    public class PatternSetFilter implements Filter<ExtractedAnnotation>
033    {
034        private final FieldExtractor fieldExtractor;
035        private final ConcurrentMap<String, Pattern> patternMap = new ConcurrentHashMap<String, Pattern>();
036    
037        @Inject
038        public PatternSetFilter(FieldExtractor fieldExtractor, Set<Pattern> patterns)
039        {
040            this.fieldExtractor = fieldExtractor;
041    
042            for (Pattern pattern : patterns) {
043                patternMap.put(pattern.toString(), pattern);
044            }
045        }
046    
047        @Override
048        public boolean passesFilter(String name, ExtractedAnnotation annotation)
049        {
050            String input = fieldExtractor.getField(name, annotation);
051    
052            if (input == null) {
053                return false;
054            }
055    
056            for (Pattern pattern : patternMap.values()) {
057                if (pattern.matcher(input).find()) {
058                    return true;
059                }
060            }
061    
062            return false;
063        }
064    
065        @Managed(description = "list of patterns for this filter")
066        public List<String> getPatternSet()
067        {
068            return new ArrayList<String>(patternMap.keySet());
069        }
070    
071        @Managed(description = "add a regular expression to filter set")
072        public void addPattern(String patternString)
073        {
074            patternMap.put(patternString, Pattern.compile(patternString));
075        }
076    
077        @Managed(description = "add a regular expression to filter set")
078        public void removePattern(String patternString)
079        {
080            patternMap.remove(patternString);
081        }
082    }
