017    package com.ning.metrics.collector;
018    
019    import com.google.inject.Guice;
020    import com.google.inject.Injector;
021    import com.google.inject.servlet.ServletModule;
022    import com.ning.metrics.collector.binder.modules.EventCollectorModule;
023    import com.ning.metrics.collector.binder.modules.OpenSourceCollectorModule;
024    import com.ning.metrics.collector.binder.modules.ScribeModule;
025    import com.ning.metrics.collector.endpoint.servers.JettyServer;
026    import com.ning.metrics.collector.endpoint.servers.ScribeServer;
027    import com.ning.metrics.collector.util.F5PoolMemberControl;
028    import com.sun.jersey.api.core.PackagesResourceConfig;
029    import com.sun.jersey.guice.spi.container.servlet.GuiceContainer;
030    import org.apache.log4j.Logger;
031    
032    import java.util.HashMap;
033    import java.util.Map;
034    
035    /**
036     * If you are writing your own Main class, make sure to match the name since
037     * the GuiceServletContextListener implementation needs to access the injector created here.
038     *
039     * @see com.ning.metrics.collector.binder.modules.JettyListener
040     */
041    public class StandaloneCollectorServer
042    {
043        private final static Logger log = Logger.getLogger(StandaloneCollectorServer.class);
044        private static Injector injector = null;
045    
046        public static void main(String... args) throws Exception
047        {
048            final long startTime = System.currentTimeMillis();
049    
050            /* Scan for Jersey endpoints */
051            final Map<String, String> params = new HashMap<String, String>();
052            params.put(PackagesResourceConfig.PROPERTY_PACKAGES, "com.ning.metrics.collector.endpoint");
053    
054            injector = Guice.createInjector(
055                new EventCollectorModule(),      /* Required, wire up the event processor and the writer */
056                new OpenSourceCollectorModule(), /* Open-Source version of certain interfaces */
057    
058                new ServletModule()              /* Optional, provide the Jetty endpoint */
059                {
060                    @Override
061                    protected void configureServlets()
062                    {
063                        // Note! It's "*", NOT "/*"
064                        serve("*").with(GuiceContainer.class, params);
065                    }
066                },
067    
068                new ScribeModule()              /* Optional, provide the Scribe endpoint */
069            );
070    
071            /* Start the Jetty endpoint */
072            JettyServer jetty = injector.getInstance(JettyServer.class);
073    
074            // We need to wait for Jetty to be fully up (it will setup SLF4j). Otherwise, we may encounter a race condition:
075            //SLF4J: The following loggers will not work becasue they were created
076            //SLF4J: during the default configuration phase of the underlying logging system.
077            //SLF4J: See also http://www.slf4j.org/codes.html#substituteLogger
078            //SLF4J: org.apache.thrift.transport.TIOStreamTransport
079            while (!jetty.isInitialized()) {
080                Thread.sleep(100);
081            }
082    
083            /* Start the Scribe endpoint */
084            injector.getInstance(ScribeServer.class);
085    
086            // Create a F5 BIP-IP controller
087            // This is a standalone class, access it via JMX when you need to take a collector in or out of a pool
088            injector.getInstance(F5PoolMemberControl.class);
089    
090            final long secondsToStart = (System.currentTimeMillis() - startTime) / 1000;
091            log.info(String.format("Collector initialized in %d:%02d", secondsToStart / 60, secondsToStart % 60));
092        }
093    
094        /**
095         * Hack to share the injector with the Jersey GuiceFilter
096         *
097         * @see com.ning.metrics.collector.binder.modules.JettyListener
098         */
099        public static Injector getInjector()
100        {
101            return injector;
102        }
103    }
