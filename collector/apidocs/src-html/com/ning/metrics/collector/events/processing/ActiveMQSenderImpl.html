017    package com.ning.metrics.collector.events.processing;
018    
019    import com.google.inject.Inject;
020    import com.ning.metrics.collector.binder.config.CollectorConfig;
021    import com.ning.metrics.serialization.util.Managed;
022    import org.apache.activemq.ActiveMQConnectionFactory;
023    import org.apache.activemq.command.ActiveMQTopic;
024    import org.apache.log4j.Logger;
025    
026    import javax.jms.Connection;
027    import javax.jms.DeliveryMode;
028    import javax.jms.JMSException;
029    import javax.jms.MessageProducer;
030    import javax.jms.Session;
031    import javax.jms.TextMessage;
032    import java.util.Random;
033    import java.util.concurrent.Executors;
034    import java.util.concurrent.ScheduledExecutorService;
035    import java.util.concurrent.ScheduledThreadPoolExecutor;
036    import java.util.concurrent.TimeUnit;
037    import java.util.concurrent.atomic.AtomicBoolean;
038    
039    public class ActiveMQSenderImpl implements ActiveMQSender
040    {
041        private static final Logger logger = Logger.getLogger(ActiveMQSenderImpl.class);
042    
043        // Set by the user via JMX
044        private final AtomicBoolean stopReconnecting = new AtomicBoolean(false);
045        // Set by the sending code, if something goes wrong with the connection
046        private final AtomicBoolean shouldReconnect = new AtomicBoolean(false);
047    
048        private volatile Session session = null;
049        private final ActiveMQConnectionFactory connectionFactory;
050        private volatile Connection connection = null;
051        private final int messagesTTLMilliseconds;
052    
053        private volatile int connectionRetries = 0;
054        private volatile int sessionRetries = 0;
055    
056        @Inject
057        public ActiveMQSenderImpl(
058            CollectorConfig config
059        )
060        {
061            this(config.getActiveMQHost(), config.getActiveMQPort(), config.getMessagesTTLMilliseconds());
062        }
063    
064        public ActiveMQSenderImpl(
065            String host,
066            int port,
067            int messagesTTLMilliseconds
068        )
069        {
070            this.connectionFactory = new ActiveMQConnectionFactory("tcp://" + host + ":" + port);
071            this.messagesTTLMilliseconds = messagesTTLMilliseconds;
072    
073            // Create the connection
074            requestReconnection();
075    
076            /*
077             * Add in a shutdown hook
078             */
079            Runtime.getRuntime().addShutdownHook(new Thread()
080            {
081                @Override
082                public void run()
083                {
084                    try {
085                        shutdown();
086                    }
087                    catch (JMSException e1) {
088                        logger.error("Unable to shutdown the ActiveMQ connection");
089                    }
090                }
091            });
092    
093            /*
094             * Create the reconnection thread
095             */
096            ScheduledExecutorService executor = new ScheduledThreadPoolExecutor(1, Executors.defaultThreadFactory());
097            executor.scheduleWithFixedDelay(new Runnable()
098            {
099                private final Random random = new Random();
100    
101                @Override
102                public void run()
103                {
104                    if (shouldReconnect.get()) {
105                        // Out of paranoia, sleep a bit to avoid a thundering heard of re-connections
106                        try {
107                            Thread.sleep(random.nextInt(2000));
108                        }
109                        catch (InterruptedException e) {
110                            logger.debug("Interrupted sleeping on reconnect", e);
111                            Thread.currentThread().interrupt();
112    
113                            return;
114                        }
115    
116                        connection = null;
117                        session = null;
118    
119                        // This will unset shouldReconnect on success
120                        createConnectionAndSession();
121                    }
122                }
123            }, 0, 1, TimeUnit.SECONDS);
124        }
125    
126        /**
127         * Establish a connection and a session to the specified ActiveMQ endpoint.
128         */
129        private void createConnectionAndSession()
130        {
131            if (!stopReconnecting.get()) {
132                try {
133                    // (Re)create a connection
134                    connectionRetries++;
135                    connection = connectionFactory.createConnection();
136                    connection.start();
137    
138                    // (Re)create a Session
139                    sessionRetries++;
140                    session = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
141    
142                    shouldReconnect.set(false);
143                    logger.info("Connection to ActiveMQ established");
144                }
145                catch (JMSException e) {
146                    logger.warn("Unable to create a connection and/or a session with the ActiveMQ endpoint", e);
147                }
148            }
149        }
150    
151        /**
152         * Re-initialize the connection and session, and try to reconnect to the ActiveMQ endpoint.
153         */
154        private void requestReconnection()
155        {
156            shouldReconnect.set(true);
157        }
158    
159        /**
160         * Stop the connection to the ActiveMQ endpoint
161         *
162         * @throws JMSException if the connection cannot be cleanly shutdown
163         */
164        private void shutdown() throws JMSException
165        {
166            shouldReconnect.set(false);
167            stopReconnecting.set(true);
168            if (connection != null) {
169                connection.stop();
170            }
171        }
172    
173        /**
174         * Low level plumbing to actually send the message.
175         *
176         * @param producer producer of the established session
177         * @param text     message to send
178         * @throws JMSException if something bad happens while sending the message
179         */
180        private void send(MessageProducer producer, String text) throws JMSException
181        {
182            TextMessage message = session.createTextMessage(text);
183    
184            // Tell the producer to send the message
185            producer.send(message);
186            logger.debug("Sent message to ActiveMQ: " + message.hashCode());
187        }
188    
189        @Override
190        public boolean sendMessage(String category, String message)
191        {
192            if (session == null) {
193                // The session has still not been established
194                logger.info(String.format("Connection to ActiveMQ endpoint not established, dropping message: [%s] [%s] and retrying to connect", category, message));
195                requestReconnection();
196                return false;
197            }
198    
199            // Create the destination (Topic or Queue)
200            final ActiveMQTopic topic = new ActiveMQTopic(category);
201    
202            // Create a MessageProducer from the Session to the Topic or Queue
203            MessageProducer producer;
204            try {
205                producer = session.createProducer(topic);
206                producer.setDeliveryMode(DeliveryMode.NON_PERSISTENT);
207                producer.setTimeToLive(messagesTTLMilliseconds);
208                send(producer, message);
209                producer.close();
210    
211                return true;
212            }
213            catch (JMSException e) {
214                // Connection flacky, maybe got dropped? ActiveMQ enpoint died?
215                logger.debug("Something went wrong while sending message to ActiveMQ", e);
216                requestReconnection();
217                return false;
218            }
219        }
220    
221        @Managed(description = "get the number of times we retried to connect to ActiveMQ")
222        public int getConnectionRetries()
223        {
224            return connectionRetries;
225        }
226    
227        @Managed(description = "get the number of times we retried to establish a session to ActiveMQ")
228        public int getSessionRetries()
229        {
230            return sessionRetries;
231        }
232    
233        @Managed(description = "don't keep trying to (re)connect to ActiveMQ")
234        public void giveUpTryingToConnect()
235        {
236            stopReconnecting.set(true);
237        }
238    
239        @Managed(description = "keep trying to (re)connect to ActiveMQ")
240        public void keepTryingToConnect()
241        {
242            stopReconnecting.set(false);
243        }
244    }
