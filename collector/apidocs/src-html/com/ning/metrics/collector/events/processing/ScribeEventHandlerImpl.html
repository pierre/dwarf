017    package com.ning.metrics.collector.events.processing;
018    
019    import com.google.inject.Inject;
020    import com.ning.metrics.serialization.util.Managed;
021    import com.ning.metrics.collector.binder.annotations.ScribeStats;
022    import com.ning.metrics.collector.binder.config.CollectorConfig;
023    import com.ning.metrics.collector.endpoint.EventEndPointStats;
024    import com.ning.metrics.collector.endpoint.EventStats;
025    import com.ning.metrics.serialization.event.Event;
026    import org.apache.log4j.Logger;
027    import scribe.thrift.LogEntry;
028    
029    public class ScribeEventHandlerImpl implements ScribeEventHandler
030    {
031        private static final Logger logger = Logger.getLogger(ScribeEventHandlerImpl.class);
032    
033        private final EventCollector collector;
034        private volatile boolean scribeCollectionEnabled;
035        private final EventEndPointStats stats;
036    
037        @Inject
038        public ScribeEventHandlerImpl(
039            EventCollector collector,
040            CollectorConfig config,
041            @ScribeStats EventEndPointStats endPointStats
042        )
043        {
044            this(collector, config.isScribeCollectionEnabled(), endPointStats);
045        }
046    
047        public ScribeEventHandlerImpl(
048            EventCollector collector,
049            boolean enabled,
050            EventEndPointStats stats)
051        {
052            this.collector = collector;
053            this.scribeCollectionEnabled = enabled;
054            this.stats = stats;
055        }
056    
057        @Override
058        public boolean processEvent(Event event, EventStats eventStats)
059        {
060            assert event != null;
061            boolean success;
062    
063            stats.updateTotalEvents();
064    
065            if (scribeCollectionEnabled) {
066                logger.debug(String.format("Processing event of type [%s], collection enabled", event.getName()));
067    
068                success = collector.collectEvent(event, eventStats);
069                if (success) {
070                    stats.updateSuccesfulEventCounters(event);
071                    logger.debug(String.format("Event accepted: %s", event.getData()));
072                }
073                else {
074                    stats.updateRejectedEvents();
075                    logger.warn(String.format("Event rejected: %s", event.getData()));
076                }
077            }
078            else {
079                stats.updateRejectedEvents();
080                logger.debug(String.format("Rejecting event [%s], collection disabled", event));
081                success = true;
082            }
083            return success;
084        }
085    
086        @Override
087        public void handleFailure(LogEntry l, EventStats eventStats)
088        {
089            logger.warn(String.format("Error parsing request type: %s", l));
090            stats.updateTotalEvents();
091            stats.updateFailedEvents();
092        }
093    
094        @Managed(description = "enable/disable collection of events")
095        public void setScribeCollectionEnabled(boolean value)
096        {
097            scribeCollectionEnabled = value;
098        }
099    
100        @Managed(description = "event collection enabled?")
101        public boolean getScribeCollectionEnabled()
102        {
103            return scribeCollectionEnabled;
104        }
105    }
