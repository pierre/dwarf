017    package com.ning.metrics.collector.events.processing;
018    
019    import com.google.inject.Inject;
020    import com.google.inject.internal.Nullable;
021    import com.ning.metrics.collector.binder.annotations.BufferingEventCollectorEventWriter;
022    import com.ning.metrics.collector.binder.annotations.BufferingEventCollectorExecutor;
023    import com.ning.metrics.serialization.util.Managed;
024    import com.ning.metrics.collector.binder.config.CollectorConfig;
025    import com.ning.metrics.collector.endpoint.EventStats;
026    import com.ning.metrics.serialization.event.Event;
027    import com.ning.metrics.serialization.writer.EventWriter;
028    import com.ning.metrics.collector.util.Stats;
029    import org.apache.log4j.Logger;
030    
031    import java.io.IOException;
032    import java.util.concurrent.RejectedExecutionException;
033    import java.util.concurrent.ScheduledExecutorService;
034    import java.util.concurrent.TimeUnit;
035    import java.util.concurrent.atomic.AtomicInteger;
036    import java.util.concurrent.atomic.AtomicLong;
037    
038    public class BufferingEventCollector implements EventCollector
039    {
040        private static final Logger log = Logger.getLogger(BufferingEventCollector.class);
041    
042        private final EventWriter eventWriter;
043        private final AtomicLong maxQueueSize;
044        private final AtomicInteger refreshDelayInSeconds;
045        private final ScheduledExecutorService executor;
046        private final TaskQueueService taskQueueService;
047        private final ActiveMQController activeMQController;
048    
049        private final Stats acceptanceStats = Stats.timeWindow(30, TimeUnit.MINUTES);
050        private final Stats writerStats = Stats.timeWindow(30, TimeUnit.MINUTES);
051        private final Stats extractionStats = Stats.timeWindow(30, TimeUnit.MINUTES);
052    
053        @Inject
054        public BufferingEventCollector(
055            @BufferingEventCollectorEventWriter EventWriter eventWriter,
056            @BufferingEventCollectorExecutor ScheduledExecutorService executor,
057            TaskQueueService taskQueueService,
058            @Nullable ActiveMQController activeMQController,
059            CollectorConfig config
060        )
061        {
062            this(eventWriter, executor, taskQueueService, activeMQController, config.getMaxQueueSize(), config.getRefreshDelayInSeconds());
063        }
064    
065        public BufferingEventCollector(
066            EventWriter eventWriter,
067            ScheduledExecutorService executor,
068            TaskQueueService taskQueueService,
069            ActiveMQController activeMQController,
070            long maxQueueSize,
071            int refreshDelayInSeconds
072        )
073        {
074            this.eventWriter = eventWriter;
075            this.taskQueueService = taskQueueService;
076            this.maxQueueSize = new AtomicLong(maxQueueSize);
077            this.refreshDelayInSeconds = new AtomicInteger(refreshDelayInSeconds);
078            this.executor = executor;
079            this.activeMQController = activeMQController;
080        }
081    
082        public void shutdown() throws InterruptedException
083        {
084            taskQueueService.shutdown();
085            taskQueueService.awaitTermination(15, TimeUnit.SECONDS);
086            performOperation(new DiskOperation()
087            {
088                @Override
089                public void execute() throws IOException
090                {
091                    eventWriter.forceCommit();
092                }
093            });
094        }
095    
096        @Inject
097        public void startFlusher()
098        {
099            executor.schedule(new Runnable()
100            {
101                @Override
102                public void run()
103                {
104                    try {
105                        performOperation(new DiskOperation()
106                        {
107                            @Override
108                            public void execute() throws IOException
109                            {
110                                eventWriter.commit();
111                            }
112                        });
113                    }
114                    finally {
115                        executor.schedule(this, refreshDelayInSeconds.get(), TimeUnit.SECONDS);
116                    }
117                }
118            }, refreshDelayInSeconds.get(), TimeUnit.SECONDS);
119        }
120    
121        @Override
122        public boolean collectEvent(final Event event, final EventStats eventStats)
123        {
124            if ((activeMQController != null) && (event != null)) {
125                String humanReadableMessage = event.getData().toString();
126                activeMQController.offerEvent(event.getName(), humanReadableMessage);
127            }
128    
129            if (taskQueueService.getQueueSize() < maxQueueSize.get()) {
130                // Note the TimeStamp when we accepted the record
131                eventStats.recordAccepted();
132    
133                // Schedule the write to disk
134                try {
135                    taskQueueService.execute(new Runnable()
136                    {
137                        @Override
138                        public void run()
139                        {
140                            performOperation(new DiskOperation()
141                            {
142                                @Override
143                                public void execute() throws IOException
144                                {
145                                    eventWriter.write(event);
146                                }
147                            });
148                        }
149                    });
150                }
151                catch (RejectedExecutionException e) {
152                    return false;
153                }
154    
155                // Update the statistics
156                updateEndPointsStats(eventStats);
157    
158                return true;
159            }
160    
161            return false;
162        }
163    
164        private void updateEndPointsStats(EventStats eventStats)
165        {
166            extractionStats.record(eventStats.getExtractedDelayMillis());
167            acceptanceStats.record(eventStats.getAcceptedDelayMillis());
168        }
169    
170        /**
171         * Perform a disk operation. On failure, rollback (put the file in the quarantine area).
172         * This needs to be synchronized, since the eventWriter keeps track of the current file being worked on.
173         *
174         * @param operation DiskOperation to perform
175         */
176        synchronized private void performOperation(final DiskOperation operation)
177        {
178            writerStats.profile(new Runnable()
179            {
180                @Override
181                public void run()
182                {
183                    boolean rollback = true;
184    
185                    try {
186                        operation.execute();
187                        rollback = false;
188                    }
189                    catch (IOException e) {
190                        log.warn("Error processing event queue list", e);
191                    }
192    
193                    catch (RuntimeException e) {
194                        log.warn("Runtime exception while processing event queue list", e);
195                    }
196                    finally {
197                        if (rollback) {
198                            try {
199                                log.warn("Exception performing I/O operation, attempting rollback");
200                                eventWriter.rollback();
201                            }
202                            catch (IOException e1) {
203                                //noinspection ThrowFromFinallyBlock
204                                log.warn("Unable to rollback on commit error", e1);
205                            }
206                        }
207                    }
208                }
209            });
210        }
211    
212        @Managed(description = "Set the max number of elements in the in-memory queue; queue size > this => reject events")
213        public void setMaxQueueSize(long maxQueueSize)
214        {
215            this.maxQueueSize.set(maxQueueSize);
216        }
217    
218        @Managed(description = "The max number of elements in the in-memory queue; queue size > this => reject events")
219        public long getMaxQueueSize()
220        {
221            return maxQueueSize.get();
222        }
223    
224        @Managed(description = "Number of events in the in-memory queue")
225        public long getQueueSize()
226        {
227            return taskQueueService.getQueueSize();
228        }
229    
230        @Managed(description = "TP99 of the current capacity (events/second)")
231        public double getEventsSecondTP99()
232        {
233            return 1000.0 / acceptanceStats.getMillisTP99();
234        }
235    
236        @Managed(description = "TP99 of the acceptance time per event (until it's scheduled to be flushed to disk)")
237        public double getEventsMillisTP99()
238        {
239            return acceptanceStats.getMillisTP99();
240        }
241    
242        @Managed(description = "Number of events used to calculate the events TP99")
243        public double getEventsCount()
244        {
245            return acceptanceStats.getCount();
246        }
247    
248        @Managed(description = "TP99 of the write operations")
249        public double getWriteMillisTP99()
250        {
251            return writerStats.getMillisTP99();
252        }
253    
254        @Managed(description = "Number of write operations used to calculate the writes TP99")
255        public double getWriteCount()
256        {
257            return writerStats.getCount();
258        }
259    
260        @Managed(description = "TP99 of the time used to extract events from their original payload")
261        public double getExtractionMillisTP99()
262        {
263            return extractionStats.getMillisTP99();
264        }
265    
266        @Managed(description = "Number of events used to calculate the extraction TP99")
267        public double getExtractionCount()
268        {
269            return extractionStats.getCount();
270        }
