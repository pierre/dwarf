017    package com.ning.metrics.collector.events.processing;
018    
019    import com.google.inject.Inject;
020    import com.ning.metrics.collector.binder.annotations.EventEndpointRequestFilter;
021    import com.ning.metrics.serialization.util.Managed;
022    import com.ning.metrics.collector.binder.config.CollectorConfig;
023    import com.ning.metrics.collector.endpoint.EventEndPointStats;
024    import com.ning.metrics.collector.endpoint.EventStats;
025    import com.ning.metrics.serialization.event.Event;
026    import com.ning.metrics.collector.events.parsing.EventParsingException;
027    import com.ning.metrics.collector.events.parsing.ExtractedAnnotation;
028    import com.ning.metrics.collector.util.Filter;
029    import org.apache.log4j.Logger;
030    
031    import javax.ws.rs.core.CacheControl;
032    import javax.ws.rs.core.Response;
033    
034    public class EventHandlerImpl implements EventHandler
035    {
036        private static final Logger log = Logger.getLogger(EventHandlerImpl.class);
037    
038        private final EventCollector collector;
039        private final Filter requestFilter;
040        private volatile boolean collectionEnabled;
041        private final CacheControl cacheControl;
042    
043        @SuppressWarnings("unused")
044        @Inject
045        public EventHandlerImpl(
046            EventCollector collector,
047            @EventEndpointRequestFilter Filter requestFilter,
048            CollectorConfig config
049        )
050        {
051            this(collector, requestFilter, config.isEventEndpointEnabled());
052        }
053    
054        public EventHandlerImpl(
055            EventCollector collector,
056            Filter requestFilter,
057            boolean enabled
058        )
059        {
060            this.collector = collector;
061            this.requestFilter = requestFilter;
062            this.collectionEnabled = enabled;
063    
064            cacheControl = new CacheControl();
065            cacheControl.setPrivate(true);
066            cacheControl.setNoCache(true);
067            cacheControl.setProxyRevalidate(true);
068        }
069    
070        @SuppressWarnings("unchecked")
071        @Override
072        //TODO no  need for stats
073        public Response processEvent(Event event, ExtractedAnnotation annotation, EventEndPointStats stats, EventStats eventStats)
074        {
075            stats.updateTotalEvents();
076    
077            if (collectionEnabled) {
078                if (event == null) {
079                    String msg = "Received empty event";
080                    log.info(msg);
081                    return handleFailure(Response.Status.BAD_REQUEST, stats, eventStats, new EventParsingException(msg));
082                }
083                else {
084                    if (requestFilter.passesFilter(event.getName(), annotation)) {
085                        stats.updateFilteredEvents();
086                        return Response.status(Response.Status.ACCEPTED).cacheControl(cacheControl).build();
087                    }
088                    else {
089                        log.debug(String.format("Receiving event of type %s", event.getName()));
090    
091                        if (collector.collectEvent(event, eventStats)) {
092                            stats.updateSuccesfulEventCounters(event);
093                            return Response.status(Response.Status.ACCEPTED).cacheControl(cacheControl).build();
094                        }
095                        else {
096                            stats.updateRejectedEvents();
097                            log.warn(String.format("Event rejected: %s", event));
098                            return Response.status(Response.Status.SERVICE_UNAVAILABLE).cacheControl(cacheControl).build();
099                        }
100                    }
101                }
102            }
103            else {
104                stats.updateRejectedEvents();
105    
106                log.info(String.format("Collection disabled, rejecting event: %s", event));
107                return Response.status(Response.Status.SERVICE_UNAVAILABLE).cacheControl(cacheControl).build();
108            }
109        }
110    
111        @Override
112        public Response handleFailure(Response.Status status, EventEndPointStats stats, EventStats eventStats, Exception e)
113        {
114            stats.updateFailedEvents();
115            return Response.status(status)
116                .header("Warning", "199 " + e.toString())
117                .cacheControl(cacheControl)
118                .build();
119        }
120    
121        @Managed(description = "enable/disable collection of events")
122        public void setCollectionEnabled(boolean value)
123        {
124            collectionEnabled = value;
125        }
126    
127        @Managed(description = "event collection enabled?")
128        public boolean getCollectionEnabled()
129        {
130            return collectionEnabled;
131        }
132    }
