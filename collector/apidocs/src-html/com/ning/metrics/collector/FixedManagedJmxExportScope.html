017    package com.ning.metrics.collector;
018    
019    import com.google.inject.Key;
020    import com.google.inject.Provider;
021    import com.google.inject.Scope;
022    import com.google.inject.Scopes;
023    import com.ning.metrics.serialization.util.FixedManagedJmxExport;
024    import org.apache.log4j.Logger;
025    
026    /**
027     * A scope to do automatic exporting of injected objects.
028     * <p/>
029     * Use of this scope also binds the object as a singleton by design.
030     */
031    public class FixedManagedJmxExportScope implements Scope
032    {
033        private final InternalScope innerScope;
034    
035        public FixedManagedJmxExportScope(String objectName)
036        {
037            this(Logger.getLogger(FixedManagedJmxExportScope.class), objectName);
038        }
039    
040        public FixedManagedJmxExportScope(Logger log, String objectName)
041        {
042            this.innerScope = new InternalScope(log, objectName);
043        }
044    
045        /**
046         * Scopes a provider. The returned provider returns objects from this scope.
047         * If an object does not exist in this scope, the provider can use the given
048         * unscoped provider to retrieve one.
049         * <p/>
050         * <p>Scope implementations are strongly encouraged to override
051         * {@link Object#toString} in the returned provider and include the backing
052         * provider's {@code toString()} output.
053         *
054         * @param key      binding key
055         * @param unscoped locates an instance when one doesn't already exist in this
056         *                 scope.
057         * @return a new provider which only delegates to the given unscoped provider
058         *         when an instance of the requested object doesn't already exist in this
059         *         scope
060         */
061        @Override
062        public <T> Provider<T> scope(Key<T> key, Provider<T> unscoped)
063        {
064            return Scopes.SINGLETON.scope(key, innerScope.scope(key, unscoped));
065        }
066    
067        public static class InternalScope implements Scope
068        {
069            private final String objectName;
070            private final Logger log;
071    
072            public InternalScope(Logger log, String objectName)
073            {
074                this.objectName = objectName;
075                this.log = log;
076            }
077    
078            @Override
079            public <T> Provider<T> scope(Key<T> tKey, final Provider<T> tProvider)
080            {
081                return new Provider<T>()
082                {
083                    private volatile boolean exported = false;
084    
085                    @Override
086                    public synchronized T get()
087                    {
088                        final T object = tProvider.get();
089                        log.info(String.format("Exporting object[%s] at [%s]", object.getClass(), objectName));
090                        if (!exported) {
091                            FixedManagedJmxExport.export(objectName, object);
092                            exported = true;
093                        }
094                        return object;
095                    }
096                };
097            }
098        }
099    }
