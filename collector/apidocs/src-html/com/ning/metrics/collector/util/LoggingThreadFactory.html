017    package com.ning.metrics.collector.util;
018    
019    import org.apache.log4j.LogManager;
020    import org.apache.log4j.Logger;
021    
022    import java.util.concurrent.ThreadFactory;
023    
024    public class LoggingThreadFactory implements ThreadFactory
025    {
026        private final static Logger log = LogManager.getLogger(LoggingThreadFactory.class);
027    
028        private final ThreadFactory delegate;
029    
030        public LoggingThreadFactory(ThreadFactory delegate)
031        {
032            this.delegate = delegate;
033        }
034    
035        public LoggingThreadFactory()
036        {
037            this(new RawThreadFactory());
038        }
039    
040        public Thread newThread(final Runnable runnable)
041        {
042            Runnable wrapped = new Runnable()
043            {
044                public void run()
045                {
046                    try {
047                        runnable.run();
048                    }
049                    catch (RuntimeException e) {
050                        log.error(String.format("Thread %s ended abnormally with an exception", Thread.currentThread().getName()), e);
051                        throw e;
052                    }
053                    catch (Error e) {
054                        log.error(String.format("Thread %s ended abnormally with an exception", Thread.currentThread().getName()), e);
055                        throw e;
056                    }
057    
058                    if (log.isDebugEnabled()) {
059                        boolean interrupted = Thread.interrupted();
060                        log.debug(String.format("Thread %s finished executing", Thread.currentThread().getName()));
061                        if (interrupted) {
062                            Thread.currentThread().interrupt();
063                        }
064                    }
065                }
066            };
067    
068            Thread thread = delegate.newThread(wrapped);
069    
070    //              thread.setUncaughtExceptionHandler(new Thread.UncaughtExceptionHandler()
071    //              {
072    //                      public void uncaughtException(Thread thread, Throwable e)
073    //                      {
074    //                              log.error(String.format("Thread %s ended abnormally with an exception", thread.getName()), e);
075    //                      }
076    //              });
077    
078            return thread;
079        }
080    }
