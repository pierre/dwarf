001    package com.ning.metrics.collector.util;
002    
003    import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
004    import org.apache.commons.math.stat.descriptive.UnivariateStatistic;
005    
006    import java.util.ArrayDeque;
007    import java.util.Queue;
008    
009    /**
010     * This class wraps DescriptiveStatistics to provide a time window rather
011     * than a number-of-samples window.  It does so by overriding the methods
012     * that access data to ensure that samples older than the time window are
013     * first discarded.  This class contains an auxiliary queue of millisecond
014     * times to find old samples.  It also overrides the  methods overriden by
015     * SynchronizedDescriptiveStatistics, making those methods synchronous so
016     * that this class is thread-safe.  All of the data access methods call
017     * flushOldSamples() to get rid of samples outside the time window, and then
018     * call super.
019     * <p/>
020     * When elements are added or removed from the queue of times, the window
021     * size of the supertype DescriptiveStatistics is set to the expected number
022     * of elements.  From the implementation of DescriptiveStatistics, it
023     * appears that the performance cost of managing the window size in this
024     * way is low.
025     */
026    public class SynchronizedTimeWindowStatistics extends DescriptiveStatistics
027    {
028    
029        private static final long serialVersionUID = 1L;
030    
031        // Discard samples older than this many ms ago
032        private long timeWindow;
033        // All the access methods are synchronized, so we can use the fastest representation.
034        private Queue<Long> queuedTimes = new ArrayDeque<Long>();
035    
036        /**
037         * @param timeWindow
038         */
039        public SynchronizedTimeWindowStatistics(long timeWindow)
040        {
041            this.timeWindow = timeWindow;
042        }
043    
044        /**
045         * @return sample count
046         */
047        public synchronized int getSampleCount()
048        {
049            return queuedTimes.size();
050        }
051    
052        /**
053         * @return time window
054         */
055        public synchronized long getTimeWindow()
056        {
057            return timeWindow;
058        }
059    
060        /**
061         * @param timeWindow
062         */
063        public synchronized void setTimeWindow(long timeWindow)
064        {
065            this.timeWindow = timeWindow;
066            flushOldSamples();
067        }
068    
069        private void flushOldSamples()
070        {
071            flushOlderThan(System.currentTimeMillis() - timeWindow);
072        }
073    
074        private void flushOlderThan(long time)
075        {
076            Long oldestQueuedTime = 0L;
077            while ((oldestQueuedTime = queuedTimes.peek()) != null && oldestQueuedTime < time) {
078                queuedTimes.remove();
079            }
080            int sampleCount = queuedTimes.size();
081            if (sampleCount == 0) {
082                super.clear();
083            }
084            else {
085                setWindowSize(sampleCount);
086            }
087        }
088    
089        @Override
090        public synchronized void addValue(double value)
091        {
092            final long time = System.currentTimeMillis();
093            flushOlderThan(time - timeWindow);
094            int sampleCount = queuedTimes.size();
095            // Make room for the new sample
096            setWindowSize(sampleCount + 1);
097            queuedTimes.add(time);
098            eDA.addElement(value);//super.addValue(value);
099        }
100    
101        @Override
102        public synchronized void clear()
103        {
104            queuedTimes.clear();
105            super.clear();
106        }
107    
108        // All the remaining methods just ensure the old samples are flushed
109        // and then invoke the superclass method
110    
111        @Override
112        public synchronized double apply(UnivariateStatistic stat)
113        {
114            flushOldSamples();
115            return super.apply(stat);
116        }
117    
118        @Override
119        public synchronized double getElement(int index)
120        {
121            flushOldSamples();
122            return super.getElement(index);
123        }
124    
125        @Override
126        public synchronized long getN()
127        {
128            flushOldSamples();
129            return super.getN();
130        }
131    
132        @Override
133        public synchronized double getStandardDeviation()
134        {
135            flushOldSamples();
136            return super.getStandardDeviation();
137        }
138    
139        @Override
140        public synchronized double[] getValues()
141        {
142            flushOldSamples();
143            return super.getValues();
144        }
145    
146        @Override
147        public synchronized double getPercentile(double p)
148        {
149            flushOldSamples();
150            return super.getPercentile(p);
151        }
152    
153        @Override
154        public synchronized String toString()
155        {
156            flushOldSamples();
157            return super.toString();
158        }
159    }
