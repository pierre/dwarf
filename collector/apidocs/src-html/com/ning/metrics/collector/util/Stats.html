001    package com.ning.metrics.collector.util;
002    
003    import com.ning.metrics.serialization.util.Managed;
004    import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
005    import org.apache.commons.math.stat.descriptive.SynchronizedDescriptiveStatistics;
006    
007    import java.util.concurrent.Callable;
008    import java.util.concurrent.TimeUnit;
009    import java.util.concurrent.atomic.AtomicLong;
010    
011    /**
012     * Provides size and timing statistics on operations.
013     */
014    public class Stats
015    {
016        private final AtomicLong count = new AtomicLong(0);
017        private final AtomicLong size = new AtomicLong(0);
018    
019        private volatile DescriptiveStatistics millisStats;
020        private volatile DescriptiveStatistics sizeStats;
021        private final WindowType windowType;
022        private final String windowString;
023    
024        private int capacity;
025        private long period;
026        private TimeUnit timeUnit;
027    
028        private enum WindowType
029        {
030            SAMPLE,
031            TIME
032        }
033    
034        /**
035         * Computes the number of milliseconds since the give reference time (in ns)
036         *
037         * @param startNanos the reference time in nanoseconds (use System.nanoTime()) to fetch
038         * @return the number of milliseconds
039         */
040        public static double millisElapsedSince(long startNanos)
041        {
042            return (System.nanoTime() - startNanos) / 1e6;
043        }
044    
045        /**
046         * Create a fixed sample size window stats object.
047         *
048         * @param capacity
049         * @return stats
050         */
051        public static Stats sampleWindow(int capacity)
052        {
053            return new Stats(new SynchronizedDescriptiveStatistics(capacity), new SynchronizedDescriptiveStatistics(capacity), WindowType.SAMPLE, capacity, 0, null);
054        }
055    
056        /**
057         * Create a time window stats object.
058         *
059         * @param period
060         * @param unit
061         * @return stats
062         */
063        public static Stats timeWindow(long period, TimeUnit unit)
064        {
065            long millis = TimeUnit.MILLISECONDS.convert(period, unit);
066            return new Stats(new SynchronizedTimeWindowStatistics(millis), new SynchronizedTimeWindowStatistics(millis), WindowType.TIME, 0, period, unit);
067        }
068    
069        private Stats(DescriptiveStatistics millisStats, DescriptiveStatistics sizeStats, WindowType windowType, int capacity, long period, TimeUnit unit)
070        {
071            this.capacity = capacity;
072            this.period = period;
073            this.timeUnit = unit;
074    
075            this.millisStats = millisStats;
076            this.sizeStats = sizeStats;
077            this.windowType = windowType;
078            if (windowType == WindowType.SAMPLE) {
079                windowString = String.format("%d samples", capacity);
080            }
081            else {
082                windowString = String.format("%d second window", TimeUnit.SECONDS.convert(period, unit));
083            }
084        }
085    
086        /**
087         * Record one operation.
088         *
089         * @param millis
090         */
091        public void record(double millis)
092        {
093            count.incrementAndGet();
094            millisStats.addValue(millis);
095        }
096    
097        /**
098         * Record one operation.
099         *
100         * @param millis
101         * @param size   magnitude of operation, i.e., bytes or items processed
102         */
103        public void record(double millis, long size)
104        {
105            count.incrementAndGet();
106            this.size.addAndGet(size);
107            millisStats.addValue(millis);
108            sizeStats.addValue(size);
109        }
110    
111        /**
112         * Profile a {@link Runnable}.
113         *
114         * @param task
115         * @throws Exception
116         */
117        public void profile(Runnable task)
118        {
119            long start = System.nanoTime();
120            try {
121                task.run();
122            }
123            finally {
124                record(millisElapsedSince(start));
125            }
126        }
127    
128        /**
129         * Profile a {@link Callable}.
130         *
131         * @param <T>
132         * @param task
133         * @return result
134         * @throws Exception
135         */
136        public <T> T profile(Callable<T> task) throws Exception
137        {
138            long start = System.nanoTime();
139            try {
140                return task.call();
141            }
142            finally {
143                record(millisElapsedSince(start));
144            }
145        }
146    
147        /**
148         * Profile a {@link Callable}..
149         *
150         * @param <T>
151         * @param task
152         * @param size
153         * @return result
154         * @throws Exception
155         */
156        public <T> T profile(Callable<T> task, long size) throws Exception
157        {
158            long start = System.nanoTime();
159            try {
160                return task.call();
161            }
162            finally {
163                record(millisElapsedSince(start), size);
164            }
165        }
