040    /**
041     * Control the membership to a f5 BIG-IP SLB VIP.
042     * This class is meant to be used via JMX.
043     * <p/>
044     * For environments where JMX is not secure, the password can be passed as a system property.
045     * <p/>
046     * Inspired by http://devcentral.f5.com/wiki/default.aspx/iControl/JavaPoolMemberControl.html
047     */
048    public class F5PoolMemberControl
049    {
050        private static final Logger log = Logger.getLogger(F5PoolMemberControl.class);
051    
052        private final String hostname;
053        private final String username;
054        private final String password;
055        private final String poolName;
056    
057        @Inject
058        public F5PoolMemberControl(CollectorConfig config)
059        {
060            hostname = config.getF5Hostname();
061            username = config.getF5Username();
062            password = config.getF5Password();
063            poolName = config.getF5PoolName();
064        }

127        /**
128         * Given a pool name, retrieve the pool members definitions
129         *
130         * @param poolName pool name to lookup
131         * @param hostname SLB VIP hostname
132         * @param username SLB VIP username
133         * @param password SLB VIP password
134         * @return list of CommonIPPortDefinition for all members
135         * @throws Exception generic exception if something goes wrong
136         */
137        @Managed(description = "Get the list of members in a pool")
138        @SuppressWarnings("unused")
139        public String[] getPoolMembers(String poolName, String hostname, String username, String password) throws Exception
140        {
141            log.info(String.format("Retrieving pool members for %s (hostname: %s, username: %s)", poolName, hostname, username));
142    
143            String[] poolNames = {poolName};
144            CommonIPPortDefinition[][] membersDefinitions;
145            try {
146                membersDefinitions = getInterface(hostname, username, password).getLocalLBPool().get_member(poolNames);
147            }
148            catch (AxisFault e) {
149                log.warn(String.format("Unable to get pool member: %s", e.getFaultString()));
150                throw e;
151            }
152            catch (Exception e) {
153                log.warn(String.format("Unable to get pool member: %s", e.getLocalizedMessage()));
154                throw e;
155            }
156            log.info(String.format("Found pool members for %s (hostname: %s, username: %s): %s", poolName, hostname, username, toString(membersDefinitions)));
157    
158            ArrayList<String> res = new ArrayList<String>();
159            for (CommonIPPortDefinition[] memberDefinitions : membersDefinitions) {
160                for (CommonIPPortDefinition memberDefinition : memberDefinitions) {
161                    res.add(String.format("%s:%d", memberDefinition.getAddress(), memberDefinition.getPort()));
162                }
163            }
164    
165            return res.toArray(new String[res.size()]);
166        }

182        /**
183         * Find the status of a member
184         *
185         * @param memberAddress server IP to lookup
186         * @param poolName      pool name to lookup
187         * @param hostname      SLB VIP hostname
188         * @param username      SLB VIP username
189         * @param password      SLB VIP password
190         * @return list of CommonIPPortDefinition for all members
191         * @throws Exception generic exception if something goes wrong
192         */
193        @Managed(description = "Get the list of members in a pool")
194        @SuppressWarnings("unused")
195        public String[] getPoolMemberStatuses(String memberAddress, String poolName, String hostname, String username, String password) throws Exception
196        {
197            String[] poolNames = {poolName};
198            ArrayList<String> memberStatuses = new ArrayList<String>();
199    
200            log.info(String.format("Retrieving pool member status for %s (pool: %s, hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
201            LocalLBPoolMemberMemberObjectStatus[][] members;
202            try {
203                members = getInterface(hostname, username, password).getLocalLBPoolMember().get_object_status(poolNames);
204            }
205            catch (AxisFault e) {
206                log.warn(String.format("Unable to get pool member: %s", e.getFaultString()));
207                throw e;
208            }
209            catch (Exception e) {
210                log.warn(String.format("Unable to get pool member: %s", e.getLocalizedMessage()));
211                throw e;
212            }
213    
214            for (LocalLBPoolMemberMemberObjectStatus[] statuses : members) {
215                for (LocalLBPoolMemberMemberObjectStatus status : statuses) {
216                    if (status.getMember().getAddress().equals(memberAddress)) {
217                        memberStatuses.add(toString(status.getObject_status()));
218                    }
219                }
220            }
221    
222            log.info(String.format("Found pool member status for %s (pool: %s, hostname: %s, username: %s): %s", memberAddress, poolName, hostname, username, memberStatuses));
223            return memberStatuses.toArray(new String[memberStatuses.size()]);
224        }

245         * @param memberAddress server IP to add
246         * @param memberPort    server port
247         * @param poolName      pool name
248         * @param hostname      SLB VIP hostname
249         * @param username      SLB VIP username
250         * @param password      SLB VIP password
251         * @return the new status of the server
252         * @throws Exception generic exception if something goes wrong

254        @Managed(description = "Add a member to a pool")
255        @SuppressWarnings("unused")
256        public String[] addPoolMember(String memberAddress, int memberPort, String poolName, String hostname, String username, String password) throws Exception
257        {
258            final Interfaces m_interfaces = getInterface(hostname, username, password);
259            final CommonIPPortDefinition definition = new CommonIPPortDefinition(memberAddress, memberPort);
260    
261            String[] poolNames = {poolName};
262            CommonIPPortDefinition[] definitions = {definition};
263            CommonIPPortDefinition[][] membersDefinitions = {definitions};
264            try {
265                m_interfaces.getLocalLBPool().add_member(poolNames, membersDefinitions);
266                log.info(String.format("Added %s to pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
267            }
268            catch (AxisFault e) {
269                log.warn(String.format("Unable to add pool member: %s", e.getFaultString()));
270                throw e;
271            }
272            catch (Exception e) {
273                log.warn(String.format("Unable to add pool member: %s", e.getLocalizedMessage()));
274                throw e;
275            }
276    
277            return getPoolMemberStatuses(poolName, memberAddress, hostname, username, password);
278        }

296        /**
297         * Enable a server in a pool
298         * The server needs to be a member of the pool already.
299         *
300         * @param memberAddress server IP to add
301         * @param memberPort    server port
302         * @param poolName      pool name
303         * @param hostname      SLB VIP hostname
304         * @param username      SLB VIP username
305         * @param password      SLB VIP password
306         * @return the new status of the server
307         * @throws Exception generic exception if something goes wrong
308         */
309        @Managed(description = "Enable a member in a pool")
310        @SuppressWarnings("unused")
311        public String[] enablePoolMember(String memberAddress, int memberPort, String poolName, String hostname, String username, String password) throws Exception
312        {
313            final Interfaces m_interfaces = getInterface(hostname, username, password);
314            final CommonIPPortDefinition definition = new CommonIPPortDefinition(memberAddress, memberPort);
315    
316            log.info(String.format("Enabling %s to pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
317    
318            log.info(String.format("Setting monitor state to ENABLED for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
319            setMonitorState(m_interfaces, definition, poolName, CommonEnabledState.STATE_ENABLED);
320            log.info(String.format("Setting session state to ENABLED for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
321            setSessionState(m_interfaces, definition, poolName, CommonEnabledState.STATE_ENABLED);
322    
323            return getPoolMemberStatuses(poolName, memberAddress, hostname, username, password);
324        }

341        /**
342         * Disable a server from a pool
343         *
344         * @param memberAddress server IP to remove
345         * @param memberPort    server port
346         * @param poolName      pool name
347         * @param hostname      SLB VIP hostname
348         * @param username      SLB VIP username
349         * @param password      SLB VIP password
350         * @return the new status of the server
351         * @throws Exception generic exception if something goes wrong
352         */
353        @Managed(description = "Disable a member from a pool")
354        @SuppressWarnings("unused")
355        public String[] disablePoolMember(String memberAddress, int memberPort, String poolName, String hostname, String username, String password) throws Exception
356        {
357            final Interfaces m_interfaces = getInterface(hostname, username, password);
358            final CommonIPPortDefinition definition = new CommonIPPortDefinition(memberAddress, memberPort);
359    
360            log.info(String.format("Setting session state to DISABLED for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
361            setSessionState(m_interfaces, definition, poolName, CommonEnabledState.STATE_DISABLED);
362    
363            log.info(String.format("Waiting for current connections to drop to zero for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
364            CommonIPPortDefinition[] memberDefinitions = {definition};
365            CommonIPPortDefinition[][] membersDefinitions = {memberDefinitions};
366    
367            long cur_connections = 1;
368            final String[] poolNames = {poolName};
369            while (cur_connections > 0) {
370                LocalLBPoolMemberMemberStatistics[] memberStatistics;
371                try {
372                    memberStatistics = m_interfaces.getLocalLBPoolMember().get_statistics(poolNames, membersDefinitions);
373                }
374                catch (AxisFault e) {
375                    log.warn(String.format("Unable to get statistics: %s", e.getFaultString()));
376                    throw e;
377                }
378                catch (Exception e) {
379                    log.warn(String.format("Unable to get statistics: %s", e.getLocalizedMessage()));
380                    throw e;
381                }
382    
383                LocalLBPoolMemberMemberStatistics memberStats = memberStatistics[0];
384    
385                LocalLBPoolMemberMemberStatisticEntry[] statisticEntries = memberStats.getStatistics();
386                LocalLBPoolMemberMemberStatisticEntry statisticEntry = statisticEntries[0];
387    
388                CommonStatistic[] statistics = statisticEntry.getStatistics();
389    
390                for (CommonStatistic stats : statistics) {
391                    CommonStatisticType type = stats.getType();
392                    CommonULong64 value64 = stats.getValue();
393    
394                    if (type == CommonStatisticType.STATISTIC_SERVER_SIDE_CURRENT_CONNECTIONS) {
395                        cur_connections = value64.getLow();
396                        log.info(String.format("Current connections: %s (%s in pool %s, hostname: %s, username: %s)", cur_connections, memberAddress, poolName, hostname, username));
397                    }
398                }
399                Thread.currentThread();
400                Thread.sleep(1000);
401            }
402    
403            log.info(String.format("Setting monitor state to DISABLED for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
404            setMonitorState(m_interfaces, definition, poolName, CommonEnabledState.STATE_DISABLED);
405    
406            return getPoolMemberStatuses(poolName, memberAddress, hostname, username, password);
407        }

409        private static void setSessionState(Interfaces m_interfaces, CommonIPPortDefinition definition, String poolName, CommonEnabledState state) throws Exception
410        {
411            final String[] poolNames = {poolName};
412    
413            LocalLBPoolMemberMemberSessionState memberSessionState = new LocalLBPoolMemberMemberSessionState(definition, state);
414    
415            LocalLBPoolMemberMemberSessionState[] memberSessionStates = {memberSessionState};
416            LocalLBPoolMemberMemberSessionState[][] membersSessionStates = {memberSessionStates};
417    
418            log.info(String.format("Updating SLB VIP session state: %s", toString(memberSessionState)));
419            try {
420                m_interfaces.getLocalLBPoolMember().set_session_enabled_state(poolNames, membersSessionStates);
421                log.info("Update successful");
422            }
423            catch (AxisFault e) {
424                log.warn(String.format("Unable to set session enabled state: %s", e.getFaultString()));
425                throw e;
426            }
427            catch (Exception e) {
428                log.warn(String.format("Unable to set session enabled state: %s", e.getLocalizedMessage()));
429                throw e;
430            }
431        }

457        private String toString(CommonIPPortDefinition[][] membersDefinitions)
458        {
459            String res = "";
460            for (CommonIPPortDefinition[] memberDefinitions : membersDefinitions) {
461                for (CommonIPPortDefinition memberDefinition : memberDefinitions) {
462                    res = String.format("%s%s,", res, toString(memberDefinition));
463                }
464            }
465            return StringUtils.chomp(res, ",");
466        }
