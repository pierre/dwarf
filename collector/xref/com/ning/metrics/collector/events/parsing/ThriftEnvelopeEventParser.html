32  import java.util.ArrayList;
33  import java.util.List;

35  public class ThriftEnvelopeEventParser implements EventParser
36  {
37      private static final String TOKEN_SEPARATOR = ",";
38      private static final BooleanConverter booleanConverter = new BooleanConverter();
39      private static final DoubleConverter doubleConverter = new DoubleConverter();
40      private final NumberConverter numberConverter;
41      private final ByteConverter byteConverter;
42      private final ShortConverter shortConverter;
43      private final IntegerConverter integerConverter;
44  
45      @Inject
46      public ThriftEnvelopeEventParser(NumberConverter numberConverter)
47      {
48          this.numberConverter = numberConverter;
49          this.byteConverter = new ByteConverter(numberConverter);
50          this.shortConverter = new ShortConverter(numberConverter);
51          this.integerConverter = new IntegerConverter(numberConverter);
52      }
53  
54      @Override
55      public Event parseThriftEvent(final String eventTypeName, String input, final ExtractedAnnotation extractedAnnotation) throws EventParsingException
56      {
57          try {
58              Tokenizer tokenizer = new UrlDecodingTokenizer(new SplitTokenizer(input, TOKEN_SEPARATOR));
59              List<ThriftField> payload = new ArrayList<ThriftField>();
60              short id = 1;
61  
62              while (tokenizer.hasNext()) {
63                  Token token = tokenizer.next();
64  
65                  if (!token.isEmpty()) {
66                      ThriftField field;
67  
68                      switch (token.getType()) {
69                          case 'b':
70                              field = ThriftField.createThriftField(booleanConverter.convert(token.getValue()), id);
71                              break;
72                          case '1':
73                              field = ThriftField.createThriftField(byteConverter.convert(token.getValue()), id);
74                              break;
75                          case '2':
76                              field = ThriftField.createThriftField(shortConverter.convert(token.getValue()), id);
77                              break;
78                          case '4':
79                              field = ThriftField.createThriftField(integerConverter.convert(token.getValue()), id);
80                              break;
81                          case '8':
82                              field = ThriftField.createThriftField(numberConverter.convert(token.getValue()), id);
83                              break;
84                          case 'd':
85                              field = ThriftField.createThriftField(doubleConverter.convert(token.getValue()), id);
86                              break;
87                          case 's':
88                              field = ThriftField.createThriftField(token.getValue(), id);
89                              break;
90                          case 'x':
91                              field = getAnnotatedValue(id, token, extractedAnnotation);
92                              break;
93                          default:
94                              throw new EventParsingException(String.format("unknown type %c", token.getType()));
95                      }
96  
97                      payload.add(field);
98                  }
99  
100                 id++;
101             }
102 
103             ThriftEnvelope thriftEnvelope = new ThriftEnvelope(eventTypeName, payload);
104 
105             return new ThriftEnvelopeEvent(extractedAnnotation.getDateTime(), thriftEnvelope, extractedAnnotation.getBucketGranularity());
106         }
107         catch (RuntimeException e) {
108             throw new EventParsingException("runtime exception parsing event", e);
109         }
110     }
111 
112     private ThriftField getAnnotatedValue(short id, Token token, ExtractedAnnotation annotation)
113     {
114         String function = token.getValue().toLowerCase();
115         if ("date".equals(function)) {
116             return ThriftField.createThriftField(annotation.getDateTime().getMillis(), id);
117         }
118         else if ("host".equals(function)) {
119             return ThriftField.createThriftField(annotation.getReferrerHost(), id);
120         }
121         else if ("path".equals(function)) {
122             return ThriftField.createThriftField(annotation.getReferrerPath(), id);
123         }
124         else if ("ua".equals(function)) {
125             return ThriftField.createThriftField(annotation.getUserAgent(), id);
126         }
127         else if ("ip".equals(function)) {
128             return ThriftField.createThriftField(Ip.ipToInt(annotation.getIpAddress()), id);
129         }
130 
131         throw new IllegalArgumentException(String.format("invalid annotation function: %s", function));
132     }
