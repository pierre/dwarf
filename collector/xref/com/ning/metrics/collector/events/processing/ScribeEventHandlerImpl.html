17  package com.ning.metrics.collector.events.processing;
18  
19  import com.google.inject.Inject;
20  import com.ning.metrics.serialization.util.Managed;
21  import com.ning.metrics.collector.binder.annotations.ScribeStats;
22  import com.ning.metrics.collector.binder.config.CollectorConfig;
23  import com.ning.metrics.collector.endpoint.EventEndPointStats;
24  import com.ning.metrics.collector.endpoint.EventStats;
25  import com.ning.metrics.serialization.event.Event;
26  import org.apache.log4j.Logger;
27  import scribe.thrift.LogEntry;
28  
29  public class ScribeEventHandlerImpl implements ScribeEventHandler
30  {
31      private static final Logger logger = Logger.getLogger(ScribeEventHandlerImpl.class);
32  
33      private final EventCollector collector;
34      private volatile boolean scribeCollectionEnabled;
35      private final EventEndPointStats stats;
36  
37      @Inject
38      public ScribeEventHandlerImpl(
39          EventCollector collector,
40          CollectorConfig config,
41          @ScribeStats EventEndPointStats endPointStats
42      )
43      {
44          this(collector, config.isScribeCollectionEnabled(), endPointStats);
45      }
46  
47      public ScribeEventHandlerImpl(
48          EventCollector collector,
49          boolean enabled,
50          EventEndPointStats stats)
51      {
52          this.collector = collector;
53          this.scribeCollectionEnabled = enabled;
54          this.stats = stats;
55      }
56  
57      @Override
58      public boolean processEvent(Event event, EventStats eventStats)
59      {
60          assert event != null;
61          boolean success;
62  
63          stats.updateTotalEvents();
64  
65          if (scribeCollectionEnabled) {
66              logger.debug(String.format("Processing event of type [%s], collection enabled", event.getName()));
67  
68              success = collector.collectEvent(event, eventStats);
69              if (success) {
70                  stats.updateSuccesfulEventCounters(event);
71                  logger.debug(String.format("Event accepted: %s", event.getData()));
72              }
73              else {
74                  stats.updateRejectedEvents();
75                  logger.warn(String.format("Event rejected: %s", event.getData()));
76              }
77          }
78          else {
79              stats.updateRejectedEvents();
80              logger.debug(String.format("Rejecting event [%s], collection disabled", event));
81              success = true;
82          }
83          return success;
84      }
85  
86      @Override
87      public void handleFailure(LogEntry l, EventStats eventStats)
88      {
89          logger.warn(String.format("Error parsing request type: %s", l));
90          stats.updateTotalEvents();
91          stats.updateFailedEvents();
92      }
93  
94      @Managed(description = "enable/disable collection of events")
95      public void setScribeCollectionEnabled(boolean value)
96      {
97          scribeCollectionEnabled = value;
98      }
99  
100     @Managed(description = "event collection enabled?")
101     public boolean getScribeCollectionEnabled()
102     {
103         return scribeCollectionEnabled;
104     }
105 }
