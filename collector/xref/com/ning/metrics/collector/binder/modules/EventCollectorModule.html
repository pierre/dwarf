79  /**
80   * This module contains the minimal plumbing to instantiate the EventCollector.
81   */
82  public class EventCollectorModule implements Module
83  {
84      private static final Logger log = Logger.getLogger(EventCollectorModule.class);
85  
86      @Override
87      public void configure(Binder binder)
88      {
89          CollectorConfig config = new ConfigurationObjectFactory(System.getProperties()).build(CollectorConfig.class);
90          binder.bind(CollectorConfig.class).toInstance(config);
91  
92          if (config.isActiveMQEnabled()) {
93              binder.bind(ActiveMQSender.class)
94                  .to(ActiveMQSenderImpl.class)
95                  .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ActiveMQEventCollectorSender"));
96  
97              binder.bind(ActiveMQController.class)
98                  .to(ActiveMQControllerImpl.class)
99                  .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ActiveMQEventCollector"));
100         }
101         else {
102             binder.bind(ActiveMQSender.class)
103                 .toProvider(Providers.<ActiveMQSender>of(null));
104 
105             binder.bind(ActiveMQController.class)
106                 .toProvider(Providers.<ActiveMQController>of(null));
107         }
108 
109         binder.bind(FileSystem.class)
110             .toProvider(FileSystemProvider.class)
111             .asEagerSingleton();
112 
113         /*
114          * EventWriters
115          */
116         binder.bind(EventWriter.class)
117             .annotatedWith(HdfsEventWriter.class)
118             .to(HadoopFileEventWriter.class)
119             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=HadoopFileEventWriter"));
120 
121         binder.bind(DiskSpoolEventWriter.class)
122             .toProvider(DiskSpoolEventWriterProvider.class)
123             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=DiskSpoolEventWriter"));
124 
125         binder.bind(EventWriter.class)
126             .annotatedWith(BufferingEventCollectorEventWriter.class)
127             .toProvider(ThresholdEventWriterProvider.class)
128             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ThresholdEventWriter"));
129 
130         /*
131          * TaskQueue
132          */
133         binder.bind(TaskQueueService.class)
134             .to(TaskQueueServiceImpl.class);
135         /*
136             * ScheduledExecutorServices
137             */
138         binder.bind(ScheduledExecutorService.class)
139             .annotatedWith(DiskSpoolFlushExecutor.class)
140             .toInstance(new ScheduledThreadPoolExecutor(2, new NamedThreadFactory("spool to HDFS promoter")));
141 
142         binder.bind(ScheduledExecutorService.class)
143             .annotatedWith(BufferingEventCollectorExecutor.class)
144             .toInstance(new ScheduledThreadPoolExecutor(2, new NamedThreadFactory("tmp to spool promoter")));
145 
146         /*
147          * Filters
148          */
149         String filterListDelimiter = config.getFilters();
150 
151         binder.bind(Filter.class)
152             .annotatedWith(EventEndpointRequestFilter.class)
153             .to(OrFilter.class);
154 
155         binder.bind(Filter.class)
156             .annotatedWith(Names.named("host"))
157             .toProvider(new EventFilterProvider(FieldExtractors.HOST, config.getFiltersHost(), filterListDelimiter))
158             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=Host"));
159 
160         binder.bind(Filter.class)
161             .annotatedWith(Names.named("ip"))
162             .toProvider(new EventFilterProvider(FieldExtractors.IP, config.getFiltersIp(), filterListDelimiter))
163             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=IP"));
164 
165         binder.bind(Filter.class)
166             .annotatedWith(Names.named("user-agent"))
167             .toProvider(new EventFilterProvider(FieldExtractors.USERAGENT, config.getFiltersUserAgent(), filterListDelimiter))
168             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=UserAgent"));
169 
170         binder.bind(Filter.class)
171             .annotatedWith(Names.named("path"))
172             .toProvider(new EventFilterProvider(FieldExtractors.PATH, config.getFiltersPath(), filterListDelimiter))
173             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=Path"));
174 
175         binder.bind(Filter.class)
176             .annotatedWith(Names.named("event-type"))
177             .toProvider(new EventFilterProvider(FieldExtractors.EVENT_TYPE, config.getFiltersEventType(), filterListDelimiter))
178             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector.filters:name=EventType"));
179 
180         binder.bind(new TypeLiteral<List<Filter>>()
181         {
182         })
183             .toProvider(new ArrayListProvider<Filter>()
184                 .add(Names.named("host"), Filter.class)
185                 .add(Names.named("ip"), Filter.class)
186                 .add(Names.named("user-agent"), Filter.class)
187                 .add(Names.named("path"), Filter.class)
188                 .add(Names.named("event-type"), Filter.class)
189             )
190             .asEagerSingleton();
191 
192         /*
193          * Request handlers
194          */
195         binder.bind(EventRequestHandler.class)
196             .annotatedWith(ExternalEventRequestHandler.class)
197             .toProvider(new EventRequestHandlerProvider(
198                 QueryParameterEventExtractor.class,
199                 Names.named("base10"),
200                 ExternalEventEndPointStats.class
201             ))
202             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=ExternalEventHandler"));
203 
204         binder.bind(ThriftEnvelopeEventParser.class)
205             .annotatedWith(Names.named("base64"))
206             .toProvider(new ThriftEnvelopeEventParserProvider(Base64NumberConverter.class))
207             .in(Scopes.SINGLETON);
208 
209         binder.bind(QueryParameterEventExtractor.class)
210             .annotatedWith(Names.named("base64"))
211             .toProvider(new ExternalEventExtractorProvider(Names.named("base64")))
212             .in(Scopes.SINGLETON);
213 
214         binder.bind(ThriftEnvelopeEventParser.class)
215             .annotatedWith(Names.named("base10"))
216             .toProvider(new ThriftEnvelopeEventParserProvider(DecimalNumberConverter.class))
217             .in(Scopes.SINGLETON);
218 
219         binder.bind(QueryParameterEventExtractor.class)
220             .annotatedWith(Names.named("base10"))
221             .toProvider(new ExternalEventExtractorProvider(Names.named("base10")))
222             .in(Scopes.SINGLETON);
223 
224         binder.bind(EventRequestHandler.class)
225             .annotatedWith(Base64ExternalEventRequestHandler.class)
226             .toProvider(new EventRequestHandlerProvider(
227                 QueryParameterEventExtractor.class,
228                 Names.named("base64"),
229                 ExternalEventEndPointStats.class
230             ))
231             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=Base64ExternalEventHandler"));
232 
233         binder.bind(EventRequestHandler.class)
234             .annotatedWith(InternalEventRequestHandler.class)
235             .toProvider(new EventRequestHandlerProvider(
236                 BodyEventExtractor.class,
237                 InternalEventEndPointStats.class
238             ))
239             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=InternalEventHandler"));
240 
241         binder.bind(EventEndPointStats.class)
242             .annotatedWith(ExternalEventEndPointStats.class)
243             .toProvider(EventEndPointStatsProvider.class)
244             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector:name=ExternalEventEndPointStats"));
245 
246         binder.bind(EventEndPointStats.class)
247             .annotatedWith(InternalEventEndPointStats.class)
248             .toProvider(EventEndPointStatsProvider.class)
249             .in(new FixedManagedJmxExportScope("com.ning.metrics.collector:name=InternalEventEndPointStats"));
250 
251         binder.bind(EventHandler.class)
252             .to(EventHandlerImpl.class)
253             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=EventHandler"));
254 
255         /*
256         * Final wiring
257         */
258         binder.bind(BufferingEventCollector.class)
259             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=BufferingEventCollector"));
260 
261         binder.bind(EventCollector.class)
262             .to(BufferingEventCollector.class);
263 
264         binder.bind(F5PoolMemberControl.class)
265             .in(new FixedManagedJmxExportScope(log, "com.ning.metrics.collector:name=F5poolMemberControl"));
266     }
