27  import javax.ws.rs.Consumes;
28  import javax.ws.rs.POST;
29  import javax.ws.rs.Path;
30  import javax.ws.rs.Produces;
31  import javax.ws.rs.QueryParam;
32  import javax.ws.rs.core.Context;
33  import javax.ws.rs.core.HttpHeaders;
34  import javax.ws.rs.core.MediaType;
35  import javax.ws.rs.core.Response;
36  import java.io.ByteArrayInputStream;
37  
38  /**
39   * Internal event API
40   */
41  @Path("/rest/1.0/event")
42  public class ThriftResource
43  {
44      private final EventRequestHandler requestHandler;
45      private final EventExtractorUtil extractorUtil;
46  
47      @Inject
48      public ThriftResource(
49          @InternalEventRequestHandler EventRequestHandler requestHandler,
50          EventExtractorUtil extractorUtil
51      )
52      {
53          this.requestHandler = requestHandler;
54          this.extractorUtil = extractorUtil;
55      }
56  
57      @POST
58      @Consumes("ning/thrift")
59      @Produces(MediaType.TEXT_PLAIN)
60      @Path("/rest/1.0/event")
61      public Response post(
62          @QueryParam("name") String eventName,
63          @QueryParam("date") String eventDateTimeString,
64          @QueryParam(Granularity.GRANULARITY_QUERY_PARAM) String eventGranularity,
65          byte[] content,
66          @Context HttpHeaders request
67      )
68      {
69          EventStats eventStats = new EventStats();
70          DateTime eventDateTime = new DateTime(eventDateTimeString);
71          return requestHandler.handleEventRequest(eventName, new ParsedRequest(request, new ByteArrayInputStream(content), eventDateTime, eventGranularity, extractorUtil), eventStats);
72      }
73  }
