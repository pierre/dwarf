17  package com.ning.metrics.collector.endpoint.resources;
18  
19  import com.google.inject.Inject;
20  import com.ning.metrics.collector.binder.annotations.Base64ExternalEventRequestHandler;
21  import com.ning.metrics.collector.endpoint.EventStats;
22  import com.ning.metrics.collector.events.parsing.EventExtractorUtil;
23  import com.ning.metrics.collector.events.parsing.ParsedRequest;
24  import com.ning.metrics.serialization.event.Granularity;
25  import org.joda.time.DateTime;
26  
27  import javax.ws.rs.GET;
28  import javax.ws.rs.Path;
29  import javax.ws.rs.Produces;
30  import javax.ws.rs.QueryParam;
31  import javax.ws.rs.core.Context;
32  import javax.ws.rs.core.HttpHeaders;
33  import javax.ws.rs.core.MediaType;
34  import javax.ws.rs.core.Response;
35  
36  /**
37   * Version 2 of the collector's external API encodes numbers as in a custom base64 encoding.
38   */
39  @Path("/2")
40  public class Base64CollectorResource
41  {
42      private final EventRequestHandler requestHandler;
43      private final EventExtractorUtil extractorUtil;
44  
45      @Inject
46      public Base64CollectorResource(
47          @Base64ExternalEventRequestHandler EventRequestHandler requestHandler,
48          EventExtractorUtil extractorUtil
49      )
50      {
51          this.requestHandler = requestHandler;
52          this.extractorUtil = extractorUtil;
53      }
54  
55      @GET
56      @Produces(MediaType.TEXT_PLAIN)
57      public Response get(
58          @QueryParam("v") String event,
59          @QueryParam("date") String eventDateTimeString,
60          @QueryParam(Granularity.GRANULARITY_QUERY_PARAM) String eventGranularity,
61          @Context HttpHeaders httpHeaders
62      )
63      {
64          EventStats eventStats = new EventStats();
65          DateTime eventDateTime = new DateTime(eventDateTimeString);
66          return requestHandler.handleEventRequest(event, new ParsedRequest(httpHeaders, eventDateTime, eventGranularity, extractorUtil), eventStats);
67      }
68  }
