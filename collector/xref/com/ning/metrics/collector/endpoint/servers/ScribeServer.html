17  package com.ning.metrics.collector.endpoint.servers;
18  
19  import com.google.inject.Inject;
20  import com.ning.metrics.collector.binder.config.CollectorConfig;
21  import com.ning.metrics.collector.util.NamedThreadFactory;
22  import org.apache.log4j.Logger;
23  import org.apache.thrift.server.TNonblockingServer;
24  import org.apache.thrift.transport.TNonblockingServerSocket;
25  import org.apache.thrift.transport.TNonblockingServerTransport;
26  import org.apache.thrift.transport.TTransportException;
27  import scribe.thrift.scribe.Iface;
28  import scribe.thrift.scribe.Processor;
29  
30  import java.util.concurrent.ScheduledExecutorService;
31  import java.util.concurrent.ScheduledThreadPoolExecutor;
32  
33  /**
34   * Thrift service. Contacted usually by Scribe client.
35   */
36  public class ScribeServer
37  {
38      private final Iface eventRequestHandler;
39      private final int port;
40      private TNonblockingServer server = null;
41  
42      private static final Logger log = Logger.getLogger(ScribeServer.class);
43  
44      @SuppressWarnings("unused")
45      @Inject
46      public ScribeServer(
47          Iface eventRequestHandler,
48          CollectorConfig config
49      ) throws TTransportException
50      {
51          this(eventRequestHandler, config.getScribePort());
52      }
53  
54      public ScribeServer(
55          Iface eventRequestHandler,
56          int port
57      ) throws TTransportException
58      {
59          this.eventRequestHandler = eventRequestHandler;
60          this.port = port;
61  
62          ScheduledExecutorService executor = new ScheduledThreadPoolExecutor(1, new NamedThreadFactory("CollectorTHsHaServer"));
63          executor.execute(new Runnable()
64          {
65              @Override
66              public void run()
67              {
68                  try {
69                      start();
70                  }
71                  catch (TTransportException e) {
72                      log.warn("Unable to start the Scribe server", e);
73                      Thread.currentThread().interrupt();
74                  }
75              }
76          });
77      }
78  
79      /**
80       * Start the terminal Scribe server
81       *
82       * @throws TTransportException if the TNonblockingServerSocket cannot be instantiated
83       */
84      private void start() throws TTransportException
85      {
86          TNonblockingServerTransport socket = new TNonblockingServerSocket(port);
87          Processor processor = new Processor(eventRequestHandler);
88  
89          server = new CollectorTHsHaServer(processor, socket);
90          log.info(String.format("Starting terminal Scribe server on port %d", port));
91          server.serve();
92  
93          /*
94           * Add in a shutdown hook
95           */
96          Runtime.getRuntime().addShutdownHook(new Thread()
97          {
98              @Override
99              public void run()
100             {
101                 shutdown();
102             }
103         });
104     }
105 
106     /**
107      * Stop the terminal Scribe server
108      */
109     private void shutdown()
110     {
111         if (server != null) {
112             server.stop();
113         }
114     }
115 }
