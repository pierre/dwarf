134     @Managed(description = "Get the list of members in a pool")
135     public String[] getPoolMembers(String poolName, String hostname, String username, String password) throws Exception
136     {
137         log.info(String.format("Retrieving pool members for %s (hostname: %s, username: %s)", poolName, hostname, username));
138 
139         String[] poolNames = {poolName};
140         CommonIPPortDefinition[][] membersDefinitions;
141         try {
142             membersDefinitions = getInterface(hostname, username, password).getLocalLBPool().get_member(poolNames);
143         }
144         catch (AxisFault e) {
145             log.warn(String.format("Unable to get pool member: %s", e.getFaultString()));
146             throw e;
147         }
148         catch (Exception e) {
149             log.warn(String.format("Unable to get pool member: %s", e.getLocalizedMessage()));
150             throw e;
151         }
152         log.info(String.format("Found pool members for %s (hostname: %s, username: %s): %s", poolName, hostname, username, toString(membersDefinitions)));
153 
154         ArrayList<String> res = new ArrayList<String>();
155         for (CommonIPPortDefinition[] memberDefinitions : membersDefinitions) {
156             for (CommonIPPortDefinition memberDefinition : memberDefinitions) {
157                 res.add(String.format("%s:%d", memberDefinition.getAddress(), memberDefinition.getPort()));
158             }
159         }
160 
161         return res.toArray(new String[res.size()]);

177     /**
178      * Find the status of a member
179      *
180      * @param memberAddress server IP to lookup
181      * @param poolName      pool name to lookup
182      * @param hostname      SLB VIP hostname
183      * @param username      SLB VIP username
184      * @param password      SLB VIP password
185      * @return list of CommonIPPortDefinition for all members
186      * @throws Exception generic exception if something goes wrong
187      */
188     @Managed(description = "Get the list of members in a pool")
189     public String[] getPoolMemberStatuses(String memberAddress, String poolName, String hostname, String username, String password) throws Exception
190     {
191         String[] poolNames = {poolName};
192         ArrayList<String> memberStatuses = new ArrayList<String>();
193 
194         log.info(String.format("Retrieving pool member status for %s (pool: %s, hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
195         LocalLBPoolMemberMemberObjectStatus[][] members;
196         try {
197             members = getInterface(hostname, username, password).getLocalLBPoolMember().get_object_status(poolNames);
198         }
199         catch (AxisFault e) {
200             log.warn(String.format("Unable to get pool member: %s", e.getFaultString()));
201             throw e;
202         }
203         catch (Exception e) {
204             log.warn(String.format("Unable to get pool member: %s", e.getLocalizedMessage()));
205             throw e;
206         }
207 
208         for (LocalLBPoolMemberMemberObjectStatus[] statuses : members) {
209             for (LocalLBPoolMemberMemberObjectStatus status : statuses) {
210                 if (status.getMember().getAddress().equals(memberAddress)) {
211                     memberStatuses.add(toString(status.getObject_status()));
212                 }
213             }
214         }
215 
216         log.info(String.format("Found pool member status for %s (pool: %s, hostname: %s, username: %s): %s", memberAddress, poolName, hostname, username, memberStatuses));
217         return memberStatuses.toArray(new String[memberStatuses.size()]);
218     }

330     /**
331      * Disable a server from a pool
332      *
333      * @param memberAddress server IP to remove
334      * @param memberPort    server port
335      * @param poolName      pool name
336      * @param hostname      SLB VIP hostname
337      * @param username      SLB VIP username
338      * @param password      SLB VIP password
339      * @return the new status of the server
340      * @throws Exception generic exception if something goes wrong
341      */
342     @Managed(description = "Disable a member from a pool")
343     public String[] disablePoolMember(String memberAddress, int memberPort, String poolName, String hostname, String username, String password) throws Exception
344     {
345         final Interfaces m_interfaces = getInterface(hostname, username, password);
346         final CommonIPPortDefinition definition = new CommonIPPortDefinition(memberAddress, memberPort);
347 
348         log.info(String.format("Setting session state to DISABLED for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
349         setSessionState(m_interfaces, definition, poolName, CommonEnabledState.STATE_DISABLED);
350 
351         log.info(String.format("Waiting for current connections to drop to zero for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
352         CommonIPPortDefinition[] memberDefinitions = {definition};
353         CommonIPPortDefinition[][] membersDefinitions = {memberDefinitions};
354 
355         long cur_connections = 1;
356         final String[] poolNames = {poolName};
357         while (cur_connections > 0) {
358             LocalLBPoolMemberMemberStatistics[] memberStatistics;
359             try {
360                 memberStatistics = m_interfaces.getLocalLBPoolMember().get_statistics(poolNames, membersDefinitions);
361             }
362             catch (AxisFault e) {
363                 log.warn(String.format("Unable to get statistics: %s", e.getFaultString()));
364                 throw e;
365             }
366             catch (Exception e) {
367                 log.warn(String.format("Unable to get statistics: %s", e.getLocalizedMessage()));
368                 throw e;
369             }
370 
371             LocalLBPoolMemberMemberStatistics memberStats = memberStatistics[0];
372 
373             LocalLBPoolMemberMemberStatisticEntry[] statisticEntries = memberStats.getStatistics();
374             LocalLBPoolMemberMemberStatisticEntry statisticEntry = statisticEntries[0];
375 
376             CommonStatistic[] statistics = statisticEntry.getStatistics();
377 
378             for (CommonStatistic stats : statistics) {
379                 CommonStatisticType type = stats.getType();
380                 CommonULong64 value64 = stats.getValue();
381 
382                 if (type == CommonStatisticType.STATISTIC_SERVER_SIDE_CURRENT_CONNECTIONS) {
383                     cur_connections = value64.getLow();
384                     log.info(String.format("Current connections: %s (%s in pool %s, hostname: %s, username: %s)", cur_connections, memberAddress, poolName, hostname, username));
385                 }
386             }
387             Thread.currentThread();
388             Thread.sleep(1000);
389         }
390 
391         log.info(String.format("Setting monitor state to DISABLED for %s in pool %s (hostname: %s, username: %s)", memberAddress, poolName, hostname, username));
392         setMonitorState(m_interfaces, definition, poolName, CommonEnabledState.STATE_DISABLED);
393 
394         return getPoolMemberStatuses(poolName, memberAddress, hostname, username, password);
395     }

445     private String toString(CommonIPPortDefinition[][] membersDefinitions)
446     {
447         String res = "";
448         for (CommonIPPortDefinition[] memberDefinitions : membersDefinitions) {
449             for (CommonIPPortDefinition memberDefinition : memberDefinitions) {
450                 res = String.format("%s%s,", res, toString(memberDefinition));
451             }
452         }
453         return StringUtils.chomp(res, ",");
454     }
