57      private static class ScribeClient implements Runnable
58      {
59          /**
60           * When an object implementing interface <code>Runnable</code> is used
61           * to create a thread, starting the thread causes the object's
62           * <code>run</code> method to be called in that separately executing
63           * thread.
64           * <p/>
65           * The general contract of the method <code>run</code> is that it may
66           * take any action whatsoever.
67           *
68           * @see Thread#run()
69           */
70          @Override
71          public void run()
72          {
73              ResultCode rescode = null;
74              for (int i = 0; i < NUMBER_OF_MESSAGES_PER_SCRIBE_CLIENT; i++) {
75                  try {
76                      synchronized (lockObject) {
77                          // Not Thread-safe. Sigh. Really Facebook?
78                          rescode = client.Log(messages);
79                      }
80                  }
81                  catch (TException e) {
82                      log.warn(e.getLocalizedMessage());
83                  }
84  
85                  switch (rescode) {
86                      case OK:
87                          log.debug(String.format("%d messages sent successfully.", messages.size()));
88                          break;
89                      case TRY_LATER:
90                          log.error("Falling over");
91                          System.exit(1);
92                          break;
93                      default:
94                          log.warn(String.format("Unknown error: %d", rescode.getValue()));
95                  }
96              }
97          }
98      }

145     private static long scheduleScribeAgents() throws InterruptedException
146     {
147         ExecutorService e = Executors.newFixedThreadPool(THREADPOOL_SIZE, new NamedThreadFactory("Performance tests (Scribe client)"));
148 
149         long startTime = System.currentTimeMillis();
150         for (int i = 0; i < NUMBER_OF_SCRIBE_CLIENTS; i++) {
151             e.execute(new ScribeClient());
152             log.debug(String.format("Thread %d/%d submitted", i + 1, NUMBER_OF_SCRIBE_CLIENTS));
153         }
154 
155         e.shutdown();
156         e.awaitTermination(10, TimeUnit.MINUTES);
157         return startTime;
158     }

160     private static TFramedTransport createScribeClient() throws TTransportException
161     {
162         final TSocket sock = new TSocket("127.0.0.1", 7911);
163         final TFramedTransport transport = new TFramedTransport(sock);
164         TBinaryProtocol protocol = new TBinaryProtocol(transport, false, false);
165         client = new scribe.Client(protocol, protocol);
166         transport.open();
167         return transport;

170     private static String createThriftPayload() throws TException

172         ArrayList<ThriftField> data = new ArrayList<ThriftField>();

179     @SuppressWarnings("unused")
180     private static String createSmilePayload() throws IOException
181     {
182         final ByteArrayOutputStream stream = new ByteArrayOutputStream();
183         SmileFactory f = new SmileFactory();
184         JsonGenerator g = f.createJsonGenerator(stream);
185 
186         g.writeStartObject();
187         g.writeNumberField(SmileEnvelopeEvent.SMILE_EVENT_DATETIME_TOKEN_NAME, new DateTime().getMillis());
188         g.writeStringField("FuuField", "fuu");
189         g.writeBooleanField("TrueField", true);
190         g.writeNumberField("Pi", 3.1459);
191         g.writeNumberField("Long", 10001000000L);
192         g.writeEndObject();
193         g.close(); // important: will force flushing of output, close underlying output stream
194 
195         return stream.toString();
196     }
