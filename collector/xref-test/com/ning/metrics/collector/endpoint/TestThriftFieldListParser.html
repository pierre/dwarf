17  package com.ning.metrics.collector.endpoint;
18  
19  import com.ning.metrics.serialization.thrift.ThriftField;
20  import com.ning.metrics.serialization.thrift.ThriftFieldListParser;
21  import com.ning.metrics.serialization.thrift.ThriftFieldListSerializer;
22  import org.testng.Assert;
23  import org.testng.annotations.Test;
24  
25  import java.io.ByteArrayInputStream;
26  import java.util.ArrayList;
27  
28  public class TestThriftFieldListParser
29  {
30      private final ThriftFieldListParser parser = new ThriftFieldListParser();
31  
32      @Test(groups = "fast")
33      public void testParse() throws Exception
34      {
35          ArrayList<ThriftField> input = new ArrayList<ThriftField>();
36          input.add(ThriftField.createThriftField("string", (short) 0));
37          input.add(ThriftField.createThriftField(true, (short) 1));
38          input.add(ThriftField.createThriftField((byte) 1, (short) 2));
39          input.add(ThriftField.createThriftField((short) 2, (short) 2));
40          input.add(ThriftField.createThriftField(4, (short) 3));
41          input.add(ThriftField.createThriftField(8L, (short) 4));
42          input.add(ThriftField.createThriftField(3.14, (short) 5));
43  
44          ThriftFieldListSerializer serializer = new ThriftFieldListSerializer();
45          byte[] payload = serializer.createPayload(input);
46  
47          ArrayList<ThriftField> result = parser.parse(payload.length, new ByteArrayInputStream(payload));
48  
49          Assert.assertFalse(result == input);
50          Assert.assertEquals(result, input);
51      }
52  }
