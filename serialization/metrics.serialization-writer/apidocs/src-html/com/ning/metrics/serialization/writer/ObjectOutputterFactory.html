017    package com.ning.metrics.serialization.writer;
018    
019    import java.io.FileOutputStream;
020    import java.io.IOException;
021    
022    public class ObjectOutputterFactory
023    {
024        public static ObjectOutputter createObjectOutputter(FileOutputStream out, SyncType type, int batchSize) throws IOException
025        {
026            switch (type) {
027                case NONE:
028                    return new DefaultObjectOutputter(out);
029                case FLUSH:
030                    return new FlushingObjectOutputter(out, batchSize);
031                case SYNC:
032                    return new SyncingObjectOutputter(out, batchSize);
033            }
034    
035            throw new IllegalArgumentException("unable to construct ObjectOutputter given type");
036        }
037    }
