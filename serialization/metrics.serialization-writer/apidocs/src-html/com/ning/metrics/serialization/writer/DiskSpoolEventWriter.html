018    
019    import com.ning.metrics.serialization.event.Event;
020    import org.apache.log4j.Logger;
021    import org.joda.time.Period;

024    import java.io.BufferedInputStream;
025    import java.io.File;
026    import java.io.FileInputStream;
027    import java.io.FileOutputStream;
028    import java.io.IOException;
029    import java.io.ObjectInputStream;

032    import java.util.List;

035    import java.util.concurrent.atomic.AtomicBoolean;

038    /**
039     * Disk-backed persistent queue. The DiskSpoolEventWriter writes events to disk and pass them to an EventHandler on
040     * a periodic basis.
041     * <p/>
042     * This writer writes events to disk in a temporary spool area directly upon receive. Events are stored in flat files.
043     * <p/>
044     * One can control the type of writes performed by specifying one of SyncType values. For instance, if data integrity is
045     * important, specify SyncType.SYNC to trigger a sync() of the disk after each write. Note that this will seriously impact
046     * performance.
047     * <p/>
048     * Commit and forced commit have the same behavior and will promote the current file to the final spool area. Note that
049     * the DiskSpoolEventWriter will never promote files automatically. To control this behavior programmatically, use ThresholdEventWriter.
050     * There are also JMX knobs available.
051     * <p/>
052     * Periodically, events in the final spool area will be flushed to the specified EventHandler. On failure, files are moved
053     * to a quarantine area. Quarantined files are never retried, except on startup.
054     * <p/>
055     * The rollback operation moves the current open file to the quarantine area.
056     *
057     * @see com.ning.metrics.serialization.writer.SyncType
058     */
059    public class DiskSpoolEventWriter implements EventWriter
060    {
061        private static final Logger log = Logger.getLogger(DiskSpoolEventWriter.class);
062    
063        private final AtomicLong fileId = new AtomicLong(System.currentTimeMillis() * 1000000);
064        private final AtomicBoolean flushEnabled;
065        private final AtomicLong flushIntervalInSeconds;
066        private final EventHandler eventHandler;
067        private final int rateWindowSizeMinutes;
068        private final SyncType syncType;
069        private final int syncBatchSize;
070        private final File spoolDirectory;
071        private final ScheduledExecutorService executor;
072        private final File tmpSpoolDirectory;
073        private final File quarantineDirectory;
074        private final File lockDirectory;
075        private final AtomicBoolean currentlyFlushing = new AtomicBoolean(false);
076        private final AtomicLong eventSerializationFailures = new AtomicLong(0);
077        private final EventRate writeRate;
078    
079        private volatile ObjectOutputter currentOutputter;
080        private volatile File currentOutputFile;
081    
082        public DiskSpoolEventWriter(
083            EventHandler eventHandler,
084            String spoolPath,
085            boolean flushEnabled,
086            long flushIntervalInSeconds,
087            ScheduledExecutorService executor,
088            SyncType syncType,
089            int syncBatchSize,
090            int rateWindowSizeMinutes
091        )
092        {
093            this.eventHandler = eventHandler;
094            this.rateWindowSizeMinutes = rateWindowSizeMinutes;
095            this.syncType = syncType;
096            this.syncBatchSize = syncBatchSize;
097            this.spoolDirectory = new File(spoolPath);
098            this.executor = executor;
099            this.tmpSpoolDirectory = new File(spoolDirectory, "_tmp");
100            this.quarantineDirectory = new File(spoolDirectory, "_quarantine");
101            this.lockDirectory = new File(spoolDirectory, "_lock");
102            this.flushEnabled = new AtomicBoolean(flushEnabled);
103            this.flushIntervalInSeconds = new AtomicLong(flushIntervalInSeconds);
104    
105            writeRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
106    
107            createSpoolDir(spoolDirectory);
108            createSpoolDir(tmpSpoolDirectory);
109            createSpoolDir(quarantineDirectory);
110            createSpoolDir(lockDirectory);
111            scheduleFlush();
112            recoverFiles();
113        }
114    
115        private void createSpoolDir(File dir)
116        {
117            if (!dir.exists() && !dir.mkdirs()) {
118                log.error(String.format("unable to create spool directory %s", dir));
119            }
120        }
121    
122        private void recoverFiles()
123        {
124            //only call on startup
125            for (File file : tmpSpoolDirectory.listFiles()) {
126                renameFile(file, spoolDirectory);
127            }
128        }
129    
130        public void shutdown() throws InterruptedException
131        {
132            executor.shutdown();
133            executor.awaitTermination(15, TimeUnit.SECONDS);
134        }
135    
136        private void scheduleFlush()
137        {
138            executor.schedule(new Runnable()
139            {
140                @Override
141                public void run()
142                {
143                    try {
144                        flush();
145                    }
146                    catch (Exception e) {
147                        log.error(String.format("Failed commit by %s", eventHandler.toString()), e);
148                    }
149                    finally {
150                        long sleepSeconds = getSpooledFileList().isEmpty() || !flushEnabled.get() ? flushIntervalInSeconds.get() : 0;
151                        log.debug(String.format("Sleeping %d seconds before next flush by %s", sleepSeconds, eventHandler.toString()));
152                        executor.schedule(this, sleepSeconds, TimeUnit.SECONDS);
153                    }
154                }
155            }, flushIntervalInSeconds.get(), TimeUnit.SECONDS);
156        }
157    
158        //protected for overriding during unit tests
159    
160        protected List<File> getSpooledFileList()
161        {
162            List<File> spooledFileList = new ArrayList<File>();
163    
164            for (File file : spoolDirectory.listFiles()) {
165                if (file.isFile()) {
166                    spooledFileList.add(file);
167                }
168            }
169    
170            return spooledFileList;
171        }
172    
173        @Override
174        public synchronized void write(Event event) throws IOException
175        {
176            if (currentOutputter == null) {
177                currentOutputFile = new File(tmpSpoolDirectory, String.format("%d.bin", fileId.incrementAndGet()));
178    
179                currentOutputter = ObjectOutputterFactory.createObjectOutputter(new FileOutputStream(currentOutputFile), syncType, syncBatchSize);
180            }
181    
182            try {
183                currentOutputter.writeObject(event);
184                writeRate.increment();
185            }
186            catch (RuntimeException e) {
187                eventSerializationFailures.incrementAndGet();
188                //noinspection AccessToStaticFieldLockedOnInstance
189                throw new IOException("unable to serialize event", e);
190            }
191            catch (IOException e) {
192                eventSerializationFailures.incrementAndGet();
193                //noinspection AccessToStaticFieldLockedOnInstance
194                throw new IOException("unable to serialize event", e);
195            }
196        }
197    
198        @Override
199        public synchronized void commit() throws IOException
200        {
201            forceCommit();
202        }
203    
204        @Override
205        public synchronized void forceCommit() throws IOException
206        {
207            if (currentOutputFile != null) {
208                currentOutputter.close();
209    
210                renameFile(currentOutputFile, spoolDirectory);
211    
212                currentOutputFile = null;
213                currentOutputter = null;
214            }
215        }
216    
217        @Override
218        public synchronized void rollback() throws IOException
219        {
220            if (currentOutputFile != null) {
221                currentOutputter.close();
222    
223                renameFile(currentOutputFile, quarantineDirectory);
224    
225                currentOutputFile = null;
226                currentOutputter = null;
227            }
228        }
229    
230        @Managed(description = "Commit events (forward them to final handler)")
231        public void flush()
232        {
233            if (!currentlyFlushing.compareAndSet(false, true)) {
234                return;
235            }
236    
237            for (File file : getSpooledFileList()) {
238                if (flushEnabled.get()) {
239                    final File lockedFile = renameFile(file, lockDirectory);
240                    try {
241                        // Move files aside, to avoid sending dups (the handler can take longer than the flushing period)
242                        ObjectInputStream objectInputStream = new ObjectInputStream(new BufferedInputStream(new FileInputStream(lockedFile)));
243    
244                        // Blocking call on the stream
245                        eventHandler.handle(objectInputStream, new CallbackHandler()
246                        {
247                            // This handler quarantines individual failed events
248    
249                            File quarantineFile = null;
250    
251                            // Called if the file was read just fine but there's an error reading a single event.
252                            @Override
253                            public synchronized void onError(Throwable t, Event event)
254                            {
255                                log.warn(String.format("Error trying to flush event [%s]", event), t);
256    
257                                if (event != null) {
258                                    // write the failed event to the quarantine file
259                                    try {
260                                        // if no events have failed yet, open up a quarantine file
261                                        if (quarantineFile == null) {
262                                            quarantineFile = new File(quarantineDirectory, lockedFile.getName());
263                                        }
264    
265                                        // open a new stream to write to the file.
266                                        // TODO if we had an onComplete method we wouldn't need to keep opening and closing streams.
267                                        ObjectOutputStream quarantineStream = new ObjectOutputStream(new FileOutputStream(quarantineFile));
268                                        event.writeExternal(quarantineStream);
269                                        quarantineStream.flush();
270                                        quarantineStream.close();
271                                    }
272                                    catch (IOException e) {
273                                        log.warn(String.format("Unable to write event to quarantine file: %s", event), e);
274                                    }
275                                }
276                            }
277    
278                            @Override
279                            public void onSuccess(Event event)
280                            {
281                                // no-op
282                            }
283                        });
284                    }
285                    catch (ClassNotFoundException e) {
286                        log.warn(String.format("Unable to deserialize objects in file %s and write to serialization (quarantining to %s)", file, quarantineDirectory), e);
287                        quarantineFile(lockedFile);
288                    }
289                    catch (IOException e) {
290                        log.warn(String.format("Error transferring events from local disk spool to serialization. Quarantining local file %s to directory %s", file, quarantineDirectory), e);
291                        quarantineFile(lockedFile);
292                    }
293                    catch (RuntimeException e) {
294                        log.warn(String.format("Unknown error transferring events from local disk spool to serialization. Quarantining local file %s to directory %s", file, quarantineDirectory), e);
295                        quarantineFile(lockedFile);
296                    }
297    
298                    // if lockedFile hasn't been moved yet, delete it
299                    if (lockedFile.exists() && !lockedFile.delete()) {
300                        log.warn(String.format("Unable to cleanup file %s", lockedFile));
301                    }
302                }
303            }
304    
305            currentlyFlushing.set(false);
306        }
307    
308        private void quarantineFile(File file)
309        {
310            renameFile(file, quarantineDirectory);
311    
312            // TODO we never even try to roll back.
313        }
314    
315        @Managed(description = "enable/disable flushing to hdfs")
316        public void setFlushEnabled(boolean enabled)
317        {
318            log.info(String.format("setting flush enabled to %b", enabled));
319            flushEnabled.set(enabled);
320        }
321    
322        @Managed(description = "check if hdfs flushing is enabled")
323        public boolean getFlushEnabled()
324        {
325            return flushEnabled.get();
326        }
327    
328        @Managed(description = "set the commit interval for next scheduled commit to hdfs in seconds")
329        public void setFlushIntervalInSeconds(long seconds)
330        {
331            log.info(String.format("setting persistent flushing to %d seconds", seconds));
332            flushIntervalInSeconds.set(seconds);
333        }
334    
335        @Managed(description = "get the current commit interval to hdfs in seconds")
336        public long getFlushIntervalInSeconds()
337        {
338            return flushIntervalInSeconds.get();
339        }
340    
341        @Managed(description = "size in kilobytes of disk spool queue not yet written to hdfs")
342        public long getDiskSpoolSize()
343        {
344            long size = 0;
345    
346            for (File file : getSpooledFileList()) {
347                size += file.length();
348            }
349    
350            return size / 1024;
351        }
352    
353        @Managed(description = "size in kilobytes of quarantined data that could not be written to hdfs")
354        // TODO: periodically retry?
355        public long getQuarantineSize()
356        {
357            long size = 0;
358    
359            for (File file : quarantineDirectory.listFiles()) {
360                size += file.length();
361            }
362    
363            return size / 1024;
364        }
365    
366        @Managed(description = "attempt to process quarantined files")
367        public synchronized void processQuarantinedFiles()
368        {
369            for (File file : quarantineDirectory.listFiles()) {
370                if (file.isFile()) {
371                    File dest = new File(spoolDirectory, file.getName());
372    
373                    if (!file.renameTo(dest)) {
374                        log.info(String.format("error moving quarantined file %s to %s", file, dest));
375                    }
376                }
377            }
378        }
379    
380        @Managed(description = "rate at which write() calls are succeeding to local disk")
381        public long getWriteRate()
382        {
383            return writeRate.getRate() / rateWindowSizeMinutes;
384        }
385    
386        @Managed(description = "count of events that could not be serialized from memory to disk")
387        public long getEventSeralizationFailureCount()
388        {
389            return eventSerializationFailures.get();
390        }
391    
392        private File renameFile(File srcFile, File destDir)
393        {
394            File destinationOutputFile = new File(destDir, srcFile.getName());
395    
396            if (!srcFile.renameTo(destinationOutputFile)) {
397                String msg = String.format("unable to rename spool file %s to %s", srcFile, destinationOutputFile);
398                log.error(msg);
399            }
400    
401            return destinationOutputFile;
402        }
