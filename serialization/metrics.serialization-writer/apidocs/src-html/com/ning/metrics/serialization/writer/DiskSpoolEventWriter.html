017    package com.ning.metrics.serialization.writer;
018    
019    import com.ning.metrics.serialization.event.Event;
020    import com.ning.metrics.serialization.util.Managed;
021    import org.apache.log4j.Logger;
022    import org.joda.time.Period;
023    
024    import java.io.BufferedInputStream;
025    import java.io.File;
026    import java.io.FileInputStream;
027    import java.io.FileOutputStream;
028    import java.io.IOException;
029    import java.io.ObjectInputStream;
030    import java.util.ArrayList;
031    import java.util.List;
032    import java.util.concurrent.ScheduledExecutorService;
033    import java.util.concurrent.TimeUnit;
034    import java.util.concurrent.atomic.AtomicBoolean;
035    import java.util.concurrent.atomic.AtomicLong;
036    
037    /**
038     * Disk-backed persistent queue. The DiskSpoolEventWriter writes events to disk and pass them to an EventHandler on
039     * a periodic basis.
040     * <p/>
041     * This writer writes events to disk in a temporary spool area directly upon receive. Events are stored in flat files.
042     * <p/>
043     * One can control the type of writes performed by specifying one of SyncType values. For instance, if data integrity is
044     * important, specify SyncType.SYNC to trigger a sync() of the disk after each write. Note that this will seriously impact
045     * performance.
046     * <p/>
047     * Commit and forced commit have the same behavior and will promote the current file to the final spool area. Note that
048     * the DiskSpoolEventWriter will never promote files automatically. To control this behavior programmatically, use ThresholdEventWriter.
049     * There are also JMX knobs available.
050     * <p/>
051     * Periodically, events in the final spool area will be flushed to the specified EventHandler. On failure, files are moved
052     * to a quarantine area. Quarantined files are never retried, except on startup.
053     * <p/>
054     * The rollback operation moves the current open file to the quarantine area.
055     *
056     * @see com.ning.metrics.serialization.writer.SyncType
057     */
058    public class DiskSpoolEventWriter implements EventWriter
059    {
060        private static final Logger log = Logger.getLogger(DiskSpoolEventWriter.class);
061    
062        private final AtomicLong fileId = new AtomicLong(System.currentTimeMillis() * 1000000);
063        private final AtomicBoolean flushEnabled;
064        private final AtomicLong flushIntervalInSeconds;
065        private final EventHandler eventHandler;
066        private final int rateWindowSizeMinutes;
067        private final SyncType syncType;
068        private final int syncBatchSize;
069        private final File spoolDirectory;
070        private final ScheduledExecutorService executor;
071        private final File tmpSpoolDirectory;
072        private final File quarantineDirectory;
073        private final AtomicBoolean currentlyFlushing = new AtomicBoolean(false);
074        private final AtomicLong eventSerializationFailures = new AtomicLong(0);
075        private final EventRate writeRate;
076    
077        private volatile ObjectOutputter currentOutputter;
078        private volatile File currentOutputFile;
079    
080        public DiskSpoolEventWriter(
081            EventHandler eventHandler,
082            String spoolPath,
083            boolean flushEnabled,
084            long flushIntervalInSeconds,
085            ScheduledExecutorService executor,
086            SyncType syncType,
087            int syncBatchSize,
088            int rateWindowSizeMinutes
089        )
090        {
091            this.eventHandler = eventHandler;
092            this.rateWindowSizeMinutes = rateWindowSizeMinutes;
093            this.syncType = syncType;
094            this.syncBatchSize = syncBatchSize;
095            this.spoolDirectory = new File(spoolPath);
096            this.executor = executor;
097            this.tmpSpoolDirectory = new File(spoolDirectory, "_tmp");
098            this.quarantineDirectory = new File(spoolDirectory, "_quarantine");
099            this.flushEnabled = new AtomicBoolean(flushEnabled);
100            this.flushIntervalInSeconds = new AtomicLong(flushIntervalInSeconds);
101    
102            writeRate = new EventRate(Period.minutes(rateWindowSizeMinutes));
103    
104            createSpoolDir(spoolDirectory);
105            createSpoolDir(tmpSpoolDirectory);
106            createSpoolDir(quarantineDirectory);
107            scheduleFlush();
108            recoverFiles();
109        }
110    
111        private void createSpoolDir(File dir)
112        {
113            if (!dir.exists() && !dir.mkdirs()) {
114                log.error(String.format("unable to create spool directory %s", dir));
115            }
116        }
117    
118        private void recoverFiles()
119        {
120            //only call on startup
121            for (File file : tmpSpoolDirectory.listFiles()) {
122                renameFile(file, spoolDirectory);
123            }
124        }
125    
126        private void scheduleFlush()
127        {
128            executor.schedule(new Runnable()
129            {
130                @Override
131                public void run()
132                {
133                    try {
134                        flush();
135                    }
136                    catch (Exception e) {
137                        log.error(String.format("Failed commit by %s", eventHandler.toString()), e);
138                    }
139                    finally {
140                        long sleepSeconds = getSpooledFileList().isEmpty() || !flushEnabled.get() ? flushIntervalInSeconds.get() : 0;
141                        log.debug(String.format("Sleeping %d seconds before next flush by %s", sleepSeconds, eventHandler.toString()));
142                        executor.schedule(this, sleepSeconds, TimeUnit.SECONDS);
143                    }
144                }
145            }, flushIntervalInSeconds.get(), TimeUnit.SECONDS);
146        }
147    
148        //protected for overriding during unit tests
149        protected List<File> getSpooledFileList()
150        {
151            List<File> spooledFileList = new ArrayList<File>();
152    
153            for (File file : spoolDirectory.listFiles()) {
154                if (file.isFile()) {
155                    spooledFileList.add(file);
156                }
157            }
158    
159            return spooledFileList;
160        }
161    
162        @Override
163        public synchronized void write(Event event) throws IOException
164        {
165            if (currentOutputter == null) {
166                currentOutputFile = new File(tmpSpoolDirectory, String.format("%d.thrift", fileId.incrementAndGet()));
167    
168                currentOutputter = ObjectOutputterFactory.createObjectOutputter(new FileOutputStream(currentOutputFile), syncType, syncBatchSize);
169            }
170    
171            try {
172                currentOutputter.writeObject(event);
173                writeRate.increment();
174            }
175            catch (RuntimeException e) {
176                eventSerializationFailures.incrementAndGet();
177                //noinspection AccessToStaticFieldLockedOnInstance
178                throw new IOException("unable to serialize event", e);
179            }
180            catch (IOException e) {
181                eventSerializationFailures.incrementAndGet();
182                //noinspection AccessToStaticFieldLockedOnInstance
183                throw new IOException("unable to serialize event", e);
184            }
185        }
186    
187        @Override
188        public synchronized void commit() throws IOException
189        {
190            forceCommit();
191        }
192    
193        @Override
194        public synchronized void forceCommit() throws IOException
195        {
196            if (currentOutputFile != null) {
197                currentOutputter.close();
198    
199                renameFile(currentOutputFile, spoolDirectory);
200    
201                currentOutputFile = null;
202                currentOutputter = null;
203            }
204        }
205    
206        @Override
207        public synchronized void rollback() throws IOException
208        {
209            if (currentOutputFile != null) {
210                currentOutputter.close();
211    
212                renameFile(currentOutputFile, quarantineDirectory);
213    
214                currentOutputFile = null;
215                currentOutputter = null;
216            }
217        }
218    
219        @Managed(description = "Commit events (forward them to final handler)")
220        public void flush() throws IOException
221        {
222            if (!currentlyFlushing.compareAndSet(false, true)) {
223                return;
224            }
225    
226            try {
227                for (File file : getSpooledFileList()) {
228                    if (flushEnabled.get()) {
229                        boolean fileSucceeded = false;
230    
231                        try {
232                            ObjectInputStream objectInputStream = new ObjectInputStream(new BufferedInputStream(new FileInputStream(file)));
233                            eventHandler.handle(objectInputStream);
234    
235                            if (!file.delete()) {
236                                log.warn(String.format("Unable to cleanup file %s", file));
237                            }
238    
239                            fileSucceeded = true;
240                        }
241                        catch (ClassNotFoundException e) {
242                            log.warn(String.format("Unable to deserialize objects in file %s and write to serialization (quarantining to %s)", file, quarantineDirectory), e);
243                        }
244                        catch (IOException e) {
245                            log.warn(String.format("Error transferring events from local disk spool to serialization. Quarantining local file %s to directory %s", file, quarantineDirectory), e);
246                        }
247                        catch (RuntimeException e) {
248                            log.warn(String.format("Unknown error transferring events from local disk spool to serialization. Quarantining local file %s to directory %s", file, quarantineDirectory), e);
249                        }
250                        finally {
251                            if (!fileSucceeded) {
252                                renameFile(file, quarantineDirectory);
253                                eventHandler.rollback();
254                            }
255                        }
256                    }
257                }
258            }
259            finally {
260                currentlyFlushing.set(false);
261            }
262        }
263    
264        @Managed(description = "enable/disable flushing to hdfs")
265        public void setFlushEnabled(boolean enabled)
266        {
267            log.info(String.format("setting flush enabled to %b", enabled));
268            flushEnabled.set(enabled);
269        }
270    
271        @Managed(description = "check if hdfs flushing is enabled")
272        public boolean getFlushEnabled()
273        {
274            return flushEnabled.get();
275        }
276    
277        @Managed(description = "set the commit interval for next scheduled commit to hdfs in seconds")
278        public void setFlushIntervalInSeconds(long seconds)
279        {
280            log.info(String.format("setting persistent flushing to %d seconds", seconds));
281            flushIntervalInSeconds.set(seconds);
282        }
283    
284        @Managed(description = "get the current commit interval to hdfs in seconds")
285        public long getFlushIntervalInSeconds()
286        {
287            return flushIntervalInSeconds.get();
288        }
289    
290        @Managed(description = "size in kilobytes of disk spool queue not yet written to hdfs")
291        public long getDiskSpoolSize()
292        {
293            long size = 0;
294    
295            for (File file : getSpooledFileList()) {
296                size += file.length();
297            }
298    
299            return size / 1024;
300        }
301    
302        @Managed(description = "size in kilobytes of quarantined data that could not be written to hdfs")
303        // TODO: periodically retry?
304        public long getQuarantineSize()
305        {
306            long size = 0;
307    
308            for (File file : quarantineDirectory.listFiles()) {
309                size += file.length();
310            }
311    
312            return size / 1024;
313        }
314    
315        @Managed(description = "attempt to process quarantined files")
316        public synchronized void processQuarantinedFiles()
317        {
318            for (File file : quarantineDirectory.listFiles()) {
319                if (file.isFile()) {
320                    File dest = new File(spoolDirectory, file.getName());
321    
322                    if (!file.renameTo(dest)) {
323                        log.info(String.format("error moving quarantined file %s to %s", file, dest));
324                    }
325                }
326            }
327        }
328    
329        @Managed(description = "rate at which write() calls are succeeding to local disk")
330        public long getWriteRate()
331        {
332            return writeRate.getRate() / rateWindowSizeMinutes;
333        }
334    
335        @Managed(description = "count of events that could not be serialized from memory to disk")
336        public long getEventSeralizationFailureCount()
337        {
338            return eventSerializationFailures.get();
339        }
340    
341        private void renameFile(File srcFile, File destDir)
342        {
343            File destinationOutputFile = new File(destDir, srcFile.getName());
344    
345            if (!srcFile.renameTo(destinationOutputFile)) {
346                String msg = String.format("unable to rename spool file %s to %s", srcFile, destinationOutputFile);
347                log.error(msg);
348            }
349        }
