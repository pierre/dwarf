017    package com.ning.metrics.serialization.writer;
018    
019    import org.joda.time.DateTime;
020    import org.joda.time.MutableDateTime;
021    import org.joda.time.ReadableDateTime;
022    import org.joda.time.ReadablePeriod;
023    
024    import java.util.AbstractMap;
025    import java.util.ArrayList;
026    import java.util.List;
027    import java.util.Map;
028    import java.util.concurrent.BlockingDeque;
029    import java.util.concurrent.LinkedBlockingDeque;
030    import java.util.concurrent.atomic.AtomicInteger;
031    
032    public class EventRate
033    {
034        private static final int MAX_EVENT_QUEUE_SIZE = 1000;
035    
036        private final BlockingDeque<Map.Entry<DateTime, AtomicInteger>> queue = new LinkedBlockingDeque<Map.Entry<DateTime, AtomicInteger>>();
037        private int rate = 0;
038        private final ReadablePeriod rateWindow;
039        private final BlockingDeque<DateTime> dateTimeQueue = new LinkedBlockingDeque<DateTime>();
040    
041        public EventRate(ReadablePeriod rateWindow)
042        {
043            this.rateWindow = rateWindow;
044        }
045    
046        public void increment()
047        {
048            dateTimeQueue.add(truncateToSecond(getNow()));
049    
050            if (dateTimeQueue.size() > MAX_EVENT_QUEUE_SIZE) {
051                updateRate();
052            }
053        }
054    
055        public long getRate()
056        {
057            updateRate();
058    
059            return rate;
060        }
061    
062        private synchronized void updateRate()
063        {
064            List<DateTime> eventList = new ArrayList<DateTime>();
065            dateTimeQueue.drainTo(eventList);
066    
067            for (DateTime dateTime : eventList) {
068                rate++;
069                Map.Entry<DateTime, AtomicInteger> lastPair = queue.peekLast();
070    
071                if (lastPair != null && lastPair.getKey().equals(dateTime)) {
072                    lastPair.getValue().incrementAndGet();
073                }
074                else {
075                    queue.add(new AbstractMap.SimpleImmutableEntry<DateTime, AtomicInteger>(dateTime, new AtomicInteger(1)));
076                }
077    
078            }
079    
080            DateTime now = getNow();
081    
082            while (!queue.isEmpty() && !queue.peekFirst().getKey().plus(rateWindow).isAfter(now)) {
083                Map.Entry<DateTime, AtomicInteger> firstPair = queue.pollFirst();
084                rate -= firstPair.getValue().get();
085            }
086        }
087    
088        public DateTime truncateToSecond(ReadableDateTime time)
089        {
090            MutableDateTime result = new MutableDateTime(time);
091            result.setMillisOfSecond(0);
092            return new DateTime(result);
093        }
094    
095        //hook for unit testing
096        protected DateTime getNow()
097        {
098            return new DateTime();
099        }
100    }
