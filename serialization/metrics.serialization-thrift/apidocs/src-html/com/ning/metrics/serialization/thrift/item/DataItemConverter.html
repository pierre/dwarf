017    package com.ning.metrics.serialization.thrift.item;
018    
019    public class DataItemConverter
020    {
021        public static DataItem convert(DataItem dataItem, DataItemTypes type)
022        {
023            switch (type) {
024                case BOOLEAN:
025                    return new BooleanDataItem(dataItem.getBoolean());
026                case BYTE:
027                    return new ByteDataItem(dataItem.getByte());
028                case SHORT:
029                    return new ShortDataItem(dataItem.getShort());
030                case INTEGER:
031                    return new IntegerDataItem(dataItem.getInteger());
032                case LONG:
033                    return new LongDataItem(dataItem.getLong());
034                case DOUBLE:
035                    return new DoubleDataItem(dataItem.getDouble());
036                case STRING:
037                    return new StringDataItem(dataItem.getString());
038            }
039    
040            throw new IllegalArgumentException(String.format("for some reason, %s didn't match a case in the switch statement", type));
041        }
042    }
