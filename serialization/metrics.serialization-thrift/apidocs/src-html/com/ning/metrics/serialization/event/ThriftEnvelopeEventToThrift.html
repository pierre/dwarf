017    package com.ning.metrics.serialization.event;
018    
019    import com.ning.metrics.serialization.thrift.ThriftEnvelope;
020    import com.ning.metrics.serialization.thrift.ThriftField;
021    
022    import java.io.Serializable;
023    import java.lang.reflect.Field;
024    
025    public class ThriftEnvelopeEventToThrift
026    {
027        public static <T extends Serializable> T extractThrift(Class<T> clazz, ThriftEnvelopeEvent envelopeEvent)
028        {
029            ThriftEnvelope envelope = (ThriftEnvelope) envelopeEvent.getData();
030    
031            try {
032                T thriftObject = clazz.newInstance();
033    
034                Field[] fields = clazz.getFields();
035                for (ThriftField tField : envelope.getPayload()) {
036                    if (tField.getId() >= fields.length) {
037                        continue; // ignore field
038                    }
039    
040                    // Thrift fields start at 1, not 0
041                    Field field = fields[tField.getId() - 1];
042                    Class<?> type = field.getType();
043                    if (type.isAssignableFrom(Boolean.class)) {
044                        field.setBoolean(thriftObject, tField.getDataItem().getBoolean());
045                    }
046                    else if (type.isAssignableFrom(boolean.class)) {
047                        field.setBoolean(thriftObject, tField.getDataItem().getBoolean());
048                    }
049                    else if (type.isAssignableFrom(Byte.class)) {
050                        field.setByte(thriftObject, tField.getDataItem().getByte());
051                    }
052                    else if (type.isAssignableFrom(byte.class)) {
053                        field.setByte(thriftObject, tField.getDataItem().getByte());
054                    }
055                    else if (type.isAssignableFrom(Short.class)) {
056                        field.setShort(thriftObject, tField.getDataItem().getShort());
057                    }
058                    else if (type.isAssignableFrom(short.class)) {
059                        field.setShort(thriftObject, tField.getDataItem().getShort());
060                    }
061                    else if (type.isAssignableFrom(Integer.class)) {
062                        field.setInt(thriftObject, tField.getDataItem().getInteger());
063                    }
064                    else if (type.isAssignableFrom(int.class)) {
065                        field.setInt(thriftObject, tField.getDataItem().getInteger());
066                    }
067                    else if (type.isAssignableFrom(Long.class)) {
068                        field.setLong(thriftObject, tField.getDataItem().getLong());
069                    }
070                    else if (type.isAssignableFrom(long.class)) {
071                        field.setLong(thriftObject, tField.getDataItem().getLong());
072                    }
073                    else if (type.isAssignableFrom(Double.class)) {
074                        field.setDouble(thriftObject, tField.getDataItem().getDouble());
075                    }
076                    else if (type.isAssignableFrom(double.class)) {
077                        field.setDouble(thriftObject, tField.getDataItem().getDouble());
078                    }
079                    else if (type.isAssignableFrom(String.class)) {
080                        field.set(thriftObject, tField.getDataItem().getString());
081                    }
082                }
083    
084                return thriftObject;
085            }
086            catch (InstantiationException e) {
087                return null;
088            }
089            catch (IllegalAccessException e) {
090                return null;
091            }
092        }
093    }
