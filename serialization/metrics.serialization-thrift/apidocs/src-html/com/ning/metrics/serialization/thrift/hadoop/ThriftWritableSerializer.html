001    package com.ning.metrics.serialization.thrift.hadoop;
002    
003    import java.io.DataOutputStream;
004    import java.io.IOException;
005    import java.io.OutputStream;
006    
007    public class ThriftWritableSerializer
008    {
009        private DataOutputStream dataOut;
010    
011        public void open(OutputStream out)
012        {
013            if (out instanceof DataOutputStream) {
014                dataOut = (DataOutputStream) out;
015            }
016            else {
017                dataOut = new DataOutputStream(out);
018            }
019        }
020    
021        public void serialize(ThriftWritable w) throws IOException
022        {
023            w.write(dataOut);
024        }
025    
026        public void close() throws IOException
027        {
028            dataOut.close();
029        }
030    }
