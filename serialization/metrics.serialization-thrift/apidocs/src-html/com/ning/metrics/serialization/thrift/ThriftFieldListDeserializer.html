017    package com.ning.metrics.serialization.thrift;
018    
019    import com.ning.metrics.serialization.thrift.item.DataItem;
020    import com.ning.metrics.serialization.thrift.item.DataItemDeserializer;
021    import org.apache.thrift.TException;
022    import org.apache.thrift.protocol.TBinaryProtocol;
023    import org.apache.thrift.protocol.TField;
024    import org.apache.thrift.protocol.TProtocol;
025    import org.apache.thrift.protocol.TType;
026    import org.apache.thrift.transport.TIOStreamTransport;
027    
028    import java.io.ByteArrayInputStream;
029    import java.util.ArrayList;
030    import java.util.List;
031    
032    public class ThriftFieldListDeserializer
033    {
034        private final DataItemDeserializer dataItemDeserializer = new DataItemDeserializer();
035    
036        public List<ThriftField> readPayload(byte[] payload) throws TException
037        {
038            List<ThriftField> thriftFieldList = new ArrayList<ThriftField>();
039    
040            ByteArrayInputStream payloadInputStream = new ByteArrayInputStream(payload);
041            TProtocol payloadProtocol = new TBinaryProtocol(new TIOStreamTransport(payloadInputStream));
042            payloadProtocol.readStructBegin();
043    
044            TField payloadField = payloadProtocol.readFieldBegin();
045    
046            while (payloadField.type != TType.STOP) {
047                DataItem dataItem = dataItemDeserializer.fromThrift(payloadProtocol, payloadField);
048                thriftFieldList.add(new ThriftFieldImpl(dataItem, payloadField));
049                payloadProtocol.readFieldEnd();
050                payloadField = payloadProtocol.readFieldBegin();
051            }
052    
053            payloadProtocol.readStructEnd();
054            payloadProtocol.getTransport().close();
055    
056            return thriftFieldList;
057        }
058    }
