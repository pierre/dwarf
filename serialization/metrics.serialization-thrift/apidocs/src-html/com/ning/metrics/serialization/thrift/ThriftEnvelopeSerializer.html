017    package com.ning.metrics.serialization.thrift;
018    
019    import org.apache.thrift.TException;
020    import org.apache.thrift.protocol.TBinaryProtocol;
021    import org.apache.thrift.protocol.TField;
022    import org.apache.thrift.protocol.TProtocol;
023    import org.apache.thrift.protocol.TStruct;
024    import org.apache.thrift.protocol.TType;
025    import org.apache.thrift.transport.TIOStreamTransport;
026    
027    import java.io.IOException;
028    import java.io.OutputStream;
029    
030    public class ThriftEnvelopeSerializer
031    {
032        private static final TField TYPE_NAME_FIELD = new TField("typeName", TType.STRING, ThriftEnvelopeSerialization.TYPE_ID);
033        private static final TField PAYLOAD_FIELD = new TField("payload", TType.STRING, ThriftEnvelopeSerialization.PAYLOAD_ID);
034        private static final TField NAME_FIELD = new TField("name", TType.STRING, ThriftEnvelopeSerialization.NAME_ID);
035    
036        private TProtocol protocol;
037        private final ThriftFieldListSerializer payloadSerializer = new ThriftFieldListSerializer();
038    
039        public void open(OutputStream out) throws IOException
040        {
041            protocol = new TBinaryProtocol(new TIOStreamTransport(out));
042        }
043    
044        public void serialize(ThriftEnvelope thriftEnvelope) throws IOException
045        {
046            try {
047                protocol.writeStructBegin(new TStruct("EventEnvelope"));
048    
049                protocol.writeFieldBegin(TYPE_NAME_FIELD);
050                protocol.writeString(thriftEnvelope.getTypeName());
051                protocol.writeFieldEnd();
052    
053                if (!thriftEnvelope.getTypeName().equals(thriftEnvelope.getName())) {
054                    protocol.writeFieldBegin(NAME_FIELD);
055                    protocol.writeString(thriftEnvelope.getName());
056                    protocol.writeFieldEnd();
057                }
058    
059    
060                byte[] payload = payloadSerializer.createPayload(thriftEnvelope.getPayload());
061    
062                protocol.writeFieldBegin(PAYLOAD_FIELD);
063                protocol.writeBinary(payload);
064                protocol.writeFieldEnd();
065    
066                protocol.writeFieldStop();
067                protocol.writeStructEnd();
068            }
069            catch (TException e) {
070                throw new IOException("error serializing Thrift" + thriftEnvelope, e);
071            }
072        }
073    
074        public void close() throws IOException
075        {
076            protocol.getTransport().close();
077        }
078    }
