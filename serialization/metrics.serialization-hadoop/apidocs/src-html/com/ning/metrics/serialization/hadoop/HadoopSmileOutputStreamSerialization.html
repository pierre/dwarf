017    package com.ning.metrics.serialization.hadoop;
018    
019    import com.ning.metrics.serialization.smile.SmileOutputStream;
020    import org.apache.hadoop.io.serializer.Deserializer;
021    import org.apache.hadoop.io.serializer.Serialization;
022    import org.apache.hadoop.io.serializer.Serializer;
023    
024    import java.io.ByteArrayOutputStream;
025    import java.io.IOException;
026    import java.io.InputStream;
027    import java.io.OutputStream;
028    import java.nio.charset.Charset;
029    import java.util.Arrays;
030    
031    /**
032     * Hadoop serializer for Smile.
033     * <p/>
034     * It's really similar to a SmileOutputStream (i.e. raw Smile), but we have to serialize the event name
035     * as well.
036     */
037    public class HadoopSmileOutputStreamSerialization implements Serialization<SmileOutputStream>
038    {
039        private static final Charset CHARSET = Charset.forName("ISO-8859-1");
040    
041        @Override
042        public boolean accept(Class<?> c)
043        {
044            return SmileOutputStream.class.isAssignableFrom(c);
045        }
046    
047        private static class HadoopSmileOutputStreamDeserializer implements Deserializer<SmileOutputStream>
048        {
049            private InputStream in;
050    
051            /**
052             * <p>Prepare the deserializer for reading.</p>
053             */
054            @Override
055            public void open(InputStream in) throws IOException
056            {
057                this.in = in;
058            }
059    
060            /**
061             * <p>
062             * Deserialize the next object from the underlying input stream.
063             * If the object <code>t</code> is non-null then this deserializer
064             * <i>may</i> set its internal state to the next object read from the input
065             * stream. Otherwise, if the object <code>t</code> is null a new
066             * deserialized object will be created.
067             * </p>
068             *
069             * @return the deserialized object
070             */
071            @Override
072            public SmileOutputStream deserialize(SmileOutputStream stream) throws IOException
073            {
074                ByteArrayOutputStream outStream = new ByteArrayOutputStream(1024);
075                byte[] buffer = new byte[1024];
076    
077                int count;
078                while ((count = in.read(buffer)) > 0) {
079                    outStream.write(buffer, 0, count);
080                }
081    
082                byte[] bytes = outStream.toByteArray();
083                if (bytes.length < 4) {
084                    return null;
085                }
086    
087                int eventNameSize = byteArrayToInt(bytes);
088                byte[] eventNameBytes = new byte[eventNameSize];
089                System.arraycopy(bytes, 4, eventNameBytes, 0, eventNameSize);
090                String eventName = new String(eventNameBytes, CHARSET);
091    
092                byte[] payload = new byte[bytes.length - 4 - eventNameSize];
093                System.arraycopy(bytes, 4 + eventNameSize, payload, 0, payload.length);
094    
095                // TODO Should the payload be a power of 2?
096                SmileOutputStream finalStream = new SmileOutputStream(eventName, payload.length);
097                finalStream.write(payload);
098    
099                if (stream != null) {
100                    stream.write(bytes);
101                }
102    
103                return finalStream;
104            }
105    
106            /**
107             * <p>Close the underlying input stream and clear up any resources.</p>
108             */
109            @Override
110            public void close() throws IOException
111            {
112                in.close();
113            }
114        }
115    
116        private static class HadoopSmileOutputStreamSerializer implements Serializer<SmileOutputStream>
117        {
118            private OutputStream out;
119    
120            /**
121             * <p>Prepare the serializer for writing.</p>
122             */
123            @Override
124            public void open(OutputStream out) throws IOException
125            {
126                this.out = out;
127            }
128    
129            /**
130             * <p>Serialize <code>t</code> to the underlying output stream.</p>
131             */
132            @Override
133            public void serialize(SmileOutputStream outStream) throws IOException
134            {
135                byte[] payload = outStream.toByteArray();
136                byte[] eventName = outStream.getTypeName().getBytes(CHARSET);
137                int eventNameSize = eventName.length;
138    
139                byte[] eventNameSizeBytes = {
140                    (byte) (eventNameSize >>> 24),
141                    (byte) (eventNameSize >>> 16),
142                    (byte) (eventNameSize >>> 8),
143                    (byte) eventNameSize};
144    
145                byte[] eventMetadata = concat(eventNameSizeBytes, eventName);
146    
147                out.write(concat(eventMetadata, payload));
148            }
149    
150            /**
151             * <p>Close the underlying output stream and clear up any resources.</p>
152             */
153            @Override
154            public void close() throws IOException
155            {
156                out.close();
157            }
158        }
159    
160        @Override
161        public Deserializer<SmileOutputStream> getDeserializer(Class<SmileOutputStream> c)
162        {
163            return new HadoopSmileOutputStreamDeserializer();
164        }
165    
166        @Override
167        public Serializer<SmileOutputStream> getSerializer(Class<SmileOutputStream> c)
168        {
169            return new HadoopSmileOutputStreamSerializer();
170        }
171    
172        public static byte[] concat(byte[] first, byte[] second)
173        {
174            byte[] result = Arrays.copyOf(first, first.length + second.length);
175            System.arraycopy(second, 0, result, first.length, second.length);
176            return result;
177        }
178    
179        public static int byteArrayToInt(byte[] b)
180        {
181            return (b[0] << 24)
182                + ((b[1] & 0xFF) << 16)
183                + ((b[2] & 0xFF) << 8)
184                + (b[3] & 0xFF);
185        }
186    }
