017    package com.ning.metrics.serialization.hadoop;
018    
019    import com.ning.metrics.serialization.thrift.ThriftEnvelope;
020    import com.ning.metrics.serialization.thrift.ThriftEnvelopeDeserializer;
021    import com.ning.metrics.serialization.thrift.ThriftEnvelopeSerializer;
022    import org.apache.hadoop.io.serializer.Deserializer;
023    import org.apache.hadoop.io.serializer.Serialization;
024    import org.apache.hadoop.io.serializer.Serializer;
025    
026    import java.io.IOException;
027    import java.io.InputStream;
028    import java.io.OutputStream;
029    
030    public class HadoopThriftEnvelopeSerialization implements Serialization<ThriftEnvelope>
031    {
032        @Override
033        public boolean accept(Class<?> c)
034        {
035            return ThriftEnvelope.class.isAssignableFrom(c);
036        }
037    
038        private static class HadoopThriftEnvelopeDeserializer implements Deserializer<ThriftEnvelope>
039        {
040            private final ThriftEnvelopeDeserializer delegate;
041    
042            private HadoopThriftEnvelopeDeserializer()
043            {
044                delegate = new ThriftEnvelopeDeserializer();
045            }
046    
047            /**
048             * <p>Prepare the deserializer for reading.</p>
049             */
050            @Override
051            public void open(InputStream in) throws IOException
052            {
053                delegate.open(in);
054            }
055    
056            /**
057             * <p>
058             * Deserialize the next object from the underlying input stream.
059             * If the object <code>t</code> is non-null then this deserializer
060             * <i>may</i> set its internal state to the next object read from the input
061             * stream. Otherwise, if the object <code>t</code> is null a new
062             * deserialized object will be created.
063             * </p>
064             *
065             * @return the deserialized object
066             */
067            @Override
068            public ThriftEnvelope deserialize(ThriftEnvelope thriftEnvelope) throws IOException
069            {
070                return delegate.deserialize(thriftEnvelope);
071            }
072    
073            /**
074             * <p>Close the underlying input stream and clear up any resources.</p>
075             */
076            @Override
077            public void close() throws IOException
078            {
079                delegate.close();
080            }
081        }
082    
083        private static class HadoopThriftEnvelopeSerializer implements Serializer<ThriftEnvelope>
084        {
085            private final ThriftEnvelopeSerializer delegate;
086    
087            private HadoopThriftEnvelopeSerializer()
088            {
089                delegate = new ThriftEnvelopeSerializer();
090            }
091    
092            /**
093             * <p>Prepare the serializer for writing.</p>
094             */
095            @Override
096            public void open(OutputStream out) throws IOException
097            {
098                delegate.open(out);
099            }
100    
101            /**
102             * <p>Serialize <code>t</code> to the underlying output stream.</p>
103             */
104            @Override
105            public void serialize(ThriftEnvelope thriftEnvelope) throws IOException
106            {
107                delegate.serialize(thriftEnvelope);
108            }
109    
110            /**
111             * <p>Close the underlying output stream and clear up any resources.</p>
112             */
113            @Override
114            public void close() throws IOException
115            {
116                delegate.close();
117            }
118        }
119    
120        @Override
121        public Deserializer<ThriftEnvelope> getDeserializer(Class<ThriftEnvelope> c)
122        {
123            return new HadoopThriftEnvelopeDeserializer();
124        }
125    
126        @Override
127        public Serializer<ThriftEnvelope> getSerializer(Class<ThriftEnvelope> c)
128        {
129            return new HadoopThriftEnvelopeSerializer();
130        }
131    }
