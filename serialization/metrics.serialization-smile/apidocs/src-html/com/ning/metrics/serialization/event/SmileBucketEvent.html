017    package com.ning.metrics.serialization.event;
018    
019    import com.ning.metrics.serialization.smile.SmileBucket;
020    import com.ning.metrics.serialization.smile.SmileBucketDeserializer;
021    import com.ning.metrics.serialization.smile.SmileBucketSerializer;
022    import com.ning.metrics.serialization.smile.SmileOutputStream;
023    import org.joda.time.DateTime;
024    
025    import java.io.ByteArrayInputStream;
026    import java.io.ByteArrayOutputStream;
027    import java.io.IOException;
028    import java.io.ObjectInput;
029    import java.io.ObjectOutput;
030    import java.nio.charset.Charset;
031    
032    /**
033     * A meta-event wrapping a collector of Smile events of the same type and same granularity.
034     * This is used for serializing, to leverage compression (back-references) that Smile offers.
035     */
036    public class SmileBucketEvent implements Event
037    {
038        private String eventName;
039        private Granularity granularity;
040        private String suffixOutputPath = ""; // Not necessarily set, and avoid triggering an NPE in serialization methods
041        private SmileBucket bucket;
042    
043        private SmileOutputStream eventStream = null;
044        private static final Charset CHARSET = Charset.forName("UTF-8");
045    
046        public SmileBucketEvent(String eventName, Granularity granularity, SmileBucket bucket)
047        {
048            this(eventName, granularity, "", bucket);
049        }
050    
051        @Deprecated
052        public SmileBucketEvent()
053        {
054        }
055    
056        public SmileBucketEvent(String eventName, Granularity granularity, String baseOutputDir, SmileBucket bucket)
057        {
058            this.eventName = eventName;
059            this.granularity = granularity;
060            this.suffixOutputPath = baseOutputDir;
061            this.bucket = bucket;
062        }
063    
064        public SmileBucket getBucket()
065        {
066            return bucket;
067        }
068    
069        @Override
070        public DateTime getEventDateTime()
071        {
072            return null;
073        }
074    
075        @Override
076        public String getName()
077        {
078            return eventName;
079        }
080    
081        @Override
082        public Granularity getGranularity()
083        {
084            return granularity;
085        }
086    
087        @Override
088        public String getVersion()
089        {
090            return null;
091        }
092    
093        @Override
094        public String getOutputDir(String prefix)
095        {
096            if (suffixOutputPath.length() == 0) {
097                // Add a safeguard here - if it's not set, the caller must be doing something wrong
098                throw new RuntimeException("suffixOutputPath not set, events not properly grouped. See SmileEnvelopeEventsToSmileBucketEvents class.");
099            }
100            else {
101                return String.format("%s%s", prefix, suffixOutputPath);
102            }
103        }
104    
105        /**
106         * Create a Hadoop-friendly serializable object representing this bucket event.
107         *
108         * @return Object representing the data, compressed Smile version of the events
109         */
110        @Override
111        public Object getData()
112        {
113            if (eventStream == null) {
114                // Start with 16kB buffer
115                eventStream = new SmileOutputStream(eventName, 16384);
116    
117                try {
118                    SmileBucketSerializer.serialize(bucket, eventStream);
119                }
120                catch (IOException e) {
121                    return null;
122                }
123            }
124    
125            return eventStream;
126        }
127    
128        /**
129         * Serialize an event to a byte array.
130         * This method is optional, methods relying on this call should handle gracefully null.
131         *
132         * Only serializes the data, not the event metadata.
133         *
134         * @return byte array representation of an event, can return null
135         */
136        @Override
137        public byte[] getSerializedEvent()
138        {
139            return ((ByteArrayOutputStream) getData()).toByteArray();
140        }
141    
142        /*
143         * Use writeExternal when writing to a file (before sending to the collector) to preserve metadata
144         * so that we can reconstruct the event. When POSTing, use getSerializedEvent() to generate the POST body
145         * because we don't want to duplicate the metadata in both the URI and the post body.
146         */
147        @Override
148        public void writeExternal(ObjectOutput objectOutput) throws IOException
149        {
150            int eventNameLength = eventName.length();
151            objectOutput.writeInt(eventNameLength);
152            objectOutput.write(eventName.getBytes(CHARSET));
153    
154            int granularityLength = granularity.toString().length();
155            objectOutput.writeInt(granularityLength);
156            objectOutput.write(granularity.toString().getBytes(CHARSET));
157    
158            objectOutput.writeInt(suffixOutputPath.length());
159            objectOutput.write(suffixOutputPath.getBytes(CHARSET));
160    
161            byte[] data = getSerializedEvent();
162            int dataLen = data.length;
163    
164            objectOutput.writeInt(dataLen);
165            objectOutput.write(data);
166        }
167    
168        @Override
169        public void readExternal(ObjectInput objectInput) throws IOException, ClassNotFoundException
170        {
171            int eventNameLength = objectInput.readInt();
172            byte[] eventNameData = new byte[eventNameLength];
173            objectInput.readFully(eventNameData);
174            eventName = new String(eventNameData, CHARSET);
175    
176            int granularityLength = objectInput.readInt();
177            byte[] granularityData = new byte[granularityLength];
178            objectInput.readFully(granularityData);
179            try {
180                granularity = Granularity.valueOf(new String(granularityData, CHARSET));
181            }
182            catch (IllegalArgumentException e) {
183                granularity = Granularity.HOURLY;
184            }
185    
186            int suffixOutputPathLength = objectInput.readInt();
187            byte[] suffixOuputPathData = new byte[suffixOutputPathLength];
188            objectInput.readFully(suffixOuputPathData);
189            suffixOutputPath = new String(suffixOuputPathData, CHARSET);
190    
191            int dataLen = objectInput.readInt();
192            byte[] data = new byte[dataLen];
193            objectInput.readFully(data);
194    
195            bucket = SmileBucketDeserializer.deserialize(new ByteArrayInputStream(data));
196        }
197    
198        public int getNumberOfEvent()
199        {
200            return bucket.size();
201        }
202    }
