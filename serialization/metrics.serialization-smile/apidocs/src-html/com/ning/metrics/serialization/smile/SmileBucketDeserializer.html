017    package com.ning.metrics.serialization.smile;
018    
019    import org.codehaus.jackson.JsonFactory;
020    import org.codehaus.jackson.JsonNode;
021    import org.codehaus.jackson.JsonParser;
022    import org.codehaus.jackson.map.ObjectMapper;
023    import org.codehaus.jackson.node.ArrayNode;
024    import org.codehaus.jackson.smile.SmileFactory;
025    import org.codehaus.jackson.smile.SmileGenerator;
026    import org.codehaus.jackson.smile.SmileParser;
027    
028    import java.io.IOException;
029    import java.io.InputStream;
030    import java.io.PushbackInputStream;
031    
032    public class SmileBucketDeserializer
033    {
034        protected final static SmileFactory smileFactory = new SmileFactory();
035        protected final static JsonFactory jsonFactory = new JsonFactory();
036    
037        static {
038            // yes, full 'compression' by checking for repeating names, short string values:
039            smileFactory.configure(SmileGenerator.Feature.CHECK_SHARED_NAMES, true);
040            smileFactory.configure(SmileGenerator.Feature.CHECK_SHARED_STRING_VALUES, true);
041            // and for now let's not mandate header for input
042            smileFactory.configure(SmileParser.Feature.REQUIRE_HEADER, false);
043        }
044    
045        private static final ObjectMapper smileObjectMapper = new ObjectMapper(smileFactory);
046        private static final ObjectMapper jsonObjectMapper = new ObjectMapper(jsonFactory);
047        private static final byte SMILE_MARKER = ':';
048    
049        /**
050         * Given a stream of Json or Smile, create a SmileBucket representation (vector of JsonNodes).
051         *
052         * @param in data stream (Json or Smile)
053         * @return SmileBucket representation
054         * @throws IOException generic serialization error
055         */
056        public static SmileBucket deserialize(InputStream in) throws IOException
057        {
058            PushbackInputStream pbIn = new PushbackInputStream(in);
059    
060            byte firstByte = (byte) pbIn.read();
061    
062            // EOF?
063            if (firstByte == -1) {
064                return null;
065            }
066    
067            pbIn.unread(firstByte);
068    
069            if (firstByte == SMILE_MARKER) {
070                return deserialize(pbIn, smileObjectMapper);
071            }
072            else {
073                return deserialize(pbIn, jsonObjectMapper);
074            }
075        }
076    
077        /**
078         * Given a stream of Json or Smile, create a SmileBucket representation (vector of JsonNodes).
079         *
080         * @param in           data stream (Json or Smile)
081         * @param objectMapper objectMapper with the correct factory (Json or Smile)
082         * @return SmileBucket representation
083         * @throws IOException generic serialization error
084         */
085        public static SmileBucket deserialize(InputStream in, ObjectMapper objectMapper) throws IOException
086        {
087            SmileBucket bucket = new SmileBucket();
088    
089            JsonParser jp = objectMapper.getJsonFactory().createJsonParser(in);
090            JsonNode root = objectMapper.readValue(jp, JsonNode.class);
091    
092            if (root instanceof ArrayNode) {
093                ArrayNode nodes = (ArrayNode) root;
094                for (JsonNode node : nodes) {
095                    bucket.add(node);
096                }
097            }
098            else if (root != null) { // Don't add null values in the bucket
099                bucket.add(root);
100                while (jp.nextToken() != null) {
101                    bucket.add(objectMapper.readValue(jp, JsonNode.class));
102                }
103            }
104    
105            jp.close();
106    
107            return bucket;
108        }
109    }
