017    package com.ning.metrics.serialization.smile;
018    
019    import com.ning.metrics.serialization.event.SmileBucketEvent;
020    import com.ning.metrics.serialization.event.SmileEnvelopeEvent;
021    import org.codehaus.jackson.JsonNode;
022    
023    import java.util.ArrayList;
024    import java.util.Collection;
025    import java.util.HashMap;
026    import java.util.List;
027    
028    public class SmileEnvelopeEventsToSmileBucketEvents
029    {
030        /**
031         * Given a list of SmileEnvelopeEvents, create the equivalent SmileBucketEvents.
032         * This utility function groups the events by name and output path.
033         * <p/>
034         * We return a list here as there may be multiple event types in the list.
035         * TODO This assumes that all events of same type share the same granularity. This is generally
036         * true in practice but isn't explicitly enforced generally (but here).
037         *
038         * @param events Json events to extracts
039         * @return a list of SmileBucketEvent, fully populated and ready to be shipped to the collector
040         */
041        public static Collection<SmileBucketEvent> extractEvents(List<SmileEnvelopeEvent> events)
042        {
043            // MyEvent => {
044            //          /2010/01/02/10/00 => SmileBucketEvent
045            //          /2010/01/02/10/01 => SmileBucketEvent
046            // }
047            HashMap<String, HashMap<String, SmileBucketEvent>> finalEvents = new HashMap<String, HashMap<String, SmileBucketEvent>>();
048    
049            String eventName;
050            String outputDir;
051            HashMap<String, SmileBucketEvent> eventsByPath;
052            SmileBucketEvent event;
053            for (SmileEnvelopeEvent envelope : events) {
054                eventName = envelope.getName();
055    
056                // New event type?
057                if (finalEvents.get(eventName) == null) {
058                    finalEvents.put(eventName, new HashMap<String, SmileBucketEvent>());
059                }
060                eventsByPath = finalEvents.get(eventName);
061    
062                // New path?
063                // TODO This assumes same granularity for same event types
064                outputDir = envelope.getOutputDir("");
065                if (eventsByPath.get(outputDir) == null) {
066                    eventsByPath.put(outputDir, new SmileBucketEvent(eventName, envelope.getGranularity(), envelope.getOutputDir(""), new SmileBucket()));
067                }
068                event = eventsByPath.get(outputDir);
069    
070                event.getBucket().add((JsonNode) envelope.getData());
071            }
072    
073            Collection<SmileBucketEvent> results = new ArrayList<SmileBucketEvent>();
074            for (HashMap<String, SmileBucketEvent> finalEventsByPath : finalEvents.values()) {
075                results.addAll(finalEventsByPath.values());
076            }
077    
078            return results;
079        }
080    }
