017    package com.ning.metrics.serialization.smile;
018    
019    import com.ning.metrics.serialization.event.SmileBucketEvent;
020    import com.ning.metrics.serialization.event.SmileEnvelopeEvent;
021    import org.codehaus.jackson.JsonNode;
022    
023    import java.io.IOException;
024    import java.io.InputStream;
025    import java.util.ArrayList;
026    import java.util.Collection;
027    
028    public class JsonStreamToSmileBucketEvent
029    {
030        /**
031         * Given a stream of Json (smile or plain Json) events of specified type and granularity,
032         * construct SmileBucketEvent wrappers. We return a collection here because events are grouped by
033         * output path.
034         *
035         * @param eventName events type (all events in the stream are supposed to be of this type)
036         * @param in        Json (smile or plain) stream of events
037         * @return Event wrappers around these events
038         * @throws IOException generic serialization exception
039         */
040        public static Collection<SmileBucketEvent> extractEvent(String eventName, InputStream in) throws IOException
041        {
042            SmileBucket bucket = SmileBucketDeserializer.deserialize(in);
043    
044            if (bucket == null) {
045                return null;
046            }
047    
048            ArrayList<SmileEnvelopeEvent> events = new ArrayList<SmileEnvelopeEvent>();
049            for (JsonNode node : bucket) {
050                events.add(new SmileEnvelopeEvent(eventName, node));
051            }
052    
053            return SmileEnvelopeEventsToSmileBucketEvents.extractEvents(events);
054        }
055    }
