017    package com.ning.metrics.serialization.event;
018    
019    import org.codehaus.jackson.*;
020    import org.codehaus.jackson.map.ObjectMapper;
021    import org.codehaus.jackson.smile.SmileFactory;
022    import org.codehaus.jackson.smile.SmileGenerator;
023    import org.codehaus.jackson.smile.SmileParser;
024    import org.joda.time.DateTime;
025    
026    import java.io.ByteArrayOutputStream;
027    import java.io.IOException;
028    import java.io.ObjectInput;
029    import java.io.ObjectOutput;
030    import java.nio.charset.Charset;
031    import java.util.HashMap;
032    
033    /**
034     * Event representation of a single Smile event. This class is useful to send Json trees
035     * to the collector via the eventtracker library.
036     * Under the cover though, we use SmileBucketEvent on the wire (when sending to the collector) and in
037     * Hadoop sequencefiles (to leverage Smile back-references).
038     *
039     * @see com.ning.metrics.serialization.event.SmileBucketEvent
040     */
041    public class SmileEnvelopeEvent implements Event
042    {
043        // UTF-8 won't work!
044        public static final Charset CHARSET = Charset.forName("ISO-8859-1");
045    
046        public static final Charset NAME_CHARSET = Charset.forName("UTF-8");
047    
048        protected final static SmileFactory smileFactory = new SmileFactory();
049        protected final static JsonFactory jsonFactory = new JsonFactory();
050    
051        static {
052            // yes, full 'compression' by checking for repeating names, short string values:
053            smileFactory.configure(SmileGenerator.Feature.CHECK_SHARED_NAMES, true);
054            smileFactory.configure(SmileGenerator.Feature.CHECK_SHARED_STRING_VALUES, true);
055            // and for now let's not mandate header for input
056            smileFactory.configure(SmileParser.Feature.REQUIRE_HEADER, false);
057        }
058    
059        private static final ObjectMapper smileObjectMapper = new ObjectMapper(smileFactory);
060        private static final ObjectMapper jsonObjectMapper = new ObjectMapper(jsonFactory);
061    
062        public static final String SMILE_EVENT_DATETIME_TOKEN_NAME = "eventDate";
063        public static final String SMILE_EVENT_GRANULARITY_TOKEN_NAME = "eventGranularity";
064    
065        protected DateTime eventDateTime = null;
066        protected String eventName;
067        protected Granularity granularity = null;
068        protected JsonNode root;
069    
070        private boolean isPlainJson = false;
071    
072        @Deprecated
073        public SmileEnvelopeEvent()
074        {
075        }
076    
077        /**
078         * Given a map ("json-like"), create an event with hourly granularity
079         *
080         * @param eventName     name of the event
081         * @param eventDateTime event timestamp
082         * @param map           event data
083         * @throws IOException generic serialization exception
084         */
085        public SmileEnvelopeEvent(String eventName, DateTime eventDateTime, HashMap<String, Object> map) throws IOException
086        {
087            this.eventName = eventName;
088            this.eventDateTime = eventDateTime;
089            this.granularity = Granularity.HOURLY;
090    
091            ByteArrayOutputStream stream = new ByteArrayOutputStream();
092            JsonGenerator g = smileFactory.createJsonGenerator(stream);
093    
094            g.writeStartObject();
095            g.writeNumberField(SmileEnvelopeEvent.SMILE_EVENT_DATETIME_TOKEN_NAME, eventDateTime.getMillis());
096            g.writeStringField(SmileEnvelopeEvent.SMILE_EVENT_GRANULARITY_TOKEN_NAME, granularity.toString());
097            for (String key : map.keySet()) {
098                g.writeObjectField(key, map.get(key)); // will hopefully do the right thing (e.g. take care of numerics)
099            }
100            g.writeEndObject();
101            g.close(); // important: will force flushing of output, close underlying output stream
102    
103            setPayloadFromByteArray(stream.toByteArray());
104        }
105    
106        public SmileEnvelopeEvent(String eventName, JsonNode node)
107        {
108            this(eventName, null, node);
109        }
110    
111        public SmileEnvelopeEvent(String eventName, Granularity granularity, JsonNode node)
112        {
113            this.eventName = eventName;
114            this.root = node;
115            this.granularity = granularity;
116    
117            setEventPropertiesFromNode(node);
118        }
119    
120        public SmileEnvelopeEvent(String eventName, byte[] inputBytes, DateTime eventDateTime, Granularity granularity) throws IOException
121        {
122            this.eventName = eventName;
123            this.eventDateTime = eventDateTime;
124            this.granularity = granularity;
125            setPayloadFromByteArray(inputBytes);
126        }
127    
128        @Override
129        public DateTime getEventDateTime()
130        {
131            return eventDateTime;
132        }
133    
134        @Override
135        public String getName()
136        {
137            return eventName;
138        }
139    
140        @Override
141        public Granularity getGranularity()
142        {
143            return granularity;
144        }
145    
146        @Override
147        public String getVersion()
148        {
149            // TODO Not sure how to version these schemata. Need more thinking here.
150            return "1";
151        }
152    
153        @Override
154        public String getOutputDir(String prefix)
155        {
156            GranularityPathMapper pathMapper = new GranularityPathMapper(String.format("%s/%s", prefix, eventName), granularity);
157    
158            return pathMapper.getPathForDateTime(getEventDateTime());
159        }
160    
161        /**
162         * @return a JsonNode representation of a SMILE event (json)
163         */
164        @Override
165        public Object getData()
166        {
167            return root;
168        }
169    
170        public boolean isPlainJson()
171        {
172            return isPlainJson;
173        }
174    
175        public void setPlainJson(boolean plainJson)
176        {
177            isPlainJson = plainJson;
178        }
179    
180        public ObjectMapper getObjectMapper()
181        {
182            if (isPlainJson()) {
183                return jsonObjectMapper;
184            }
185            else {
186                return smileObjectMapper;
187            }
188        }
189    
190        @Override
191        public byte[] getSerializedEvent()
192        {
193            ByteArrayOutputStream outStream = new ByteArrayOutputStream();
194    
195            try {
196                JsonGenerator gen = getObjectMapper().getJsonFactory().createJsonGenerator(outStream, JsonEncoding.UTF8);
197                getObjectMapper().writeTree(gen, root);
198                gen.close();
199            }
200            catch (IOException e) {
201                return null;
202            }
203    
204            return outStream.toByteArray();
205        }
206    
207        /**
208         * The object implements the writeExternal method to save its contents
209         * by calling the methods of DataOutput for its primitive values or
210         * calling the writeObject method of ObjectOutput for objects, strings,
211         * and arrays.
212         *
213         * @param out the stream to write the object to
214         * @throws java.io.IOException Includes any I/O exceptions that may occur
215         * @serialData Overriding methods should use this tag to describe
216         * the data layout of this Externalizable object.
217         * List the sequence of element types and, if possible,
218         * relate the element to a public/protected field and/or
219         * method of this Externalizable class.
220         */
221        @Override
222        public void writeExternal(ObjectOutput out) throws IOException
223        {
224            // Name of the event
225            byte[] eventNameBytes = eventName.getBytes(NAME_CHARSET);
226            out.writeInt(eventNameBytes.length);
227            out.write(eventNameBytes);
228    
229            byte[] payloadBytes = getSerializedEvent();
230    
231            // Size of Smile payload. Needed for deserialization, see below
232            out.writeInt(payloadBytes.length);
233    
234            out.write(payloadBytes);
235        }
236    
237        /**
238         * The object implements the readExternal method to restore its
239         * contents by calling the methods of DataInput for primitive
240         * types and readObject for objects, strings and arrays.  The
241         * readExternal method must read the values in the same sequence
242         * and with the same types as were written by writeExternal.
243         *
244         * @param in the stream to read data from in order to restore the object
245         * @throws java.io.IOException    if I/O errors occur
246         */
247        @Override
248        public void readExternal(ObjectInput in) throws IOException
249        {
250            // Name of the event first
251            int smileEventNameBytesSize = in.readInt();
252            byte[] eventNameBytes = new byte[smileEventNameBytesSize];
253            in.readFully(eventNameBytes);
254            eventName = new String(eventNameBytes, NAME_CHARSET);
255    
256            // Then payload
257            int smilePayloadSize = in.readInt();
258            byte[] smilePayload = new byte[smilePayloadSize];
259            in.readFully(smilePayload);
260    
261            setPayloadFromByteArray(smilePayload);
262    
263            setEventPropertiesFromNode(root);
264        }
265    
266        private void setEventPropertiesFromNode(JsonNode node)
267        {
268            eventDateTime = getEventDateTimeFromJson(node);
269    
270            if (granularity == null) {
271                granularity = getGranularityFromJson(node);
272            }
273        }
274    
275        public static DateTime getEventDateTimeFromJson(JsonNode node)
276        {
277            JsonNode eventDateTimeNode = node.path(SMILE_EVENT_DATETIME_TOKEN_NAME);
278    
279            DateTime nodeDateTime = new DateTime();
280            if (!eventDateTimeNode.isMissingNode()) {
281                nodeDateTime = new DateTime(eventDateTimeNode.getLongValue());
282            }
283    
284            return nodeDateTime;
285        }
286    
287        public static Granularity getGranularityFromJson(JsonNode node)
288        {
289            JsonNode granularityNode = node.path(SMILE_EVENT_GRANULARITY_TOKEN_NAME);
290    
291            Granularity nodeGranularity = Granularity.HOURLY;
292            if (!granularityNode.isMissingNode()) {
293                try {
294                    nodeGranularity = Granularity.valueOf(granularityNode.getValueAsText());
295                }
296                catch (IllegalArgumentException e) {
297                    nodeGranularity = null;
298                }
299            }
300    
301            return nodeGranularity;
302        }
303    
304        private void setPayloadFromByteArray(byte[] smilePayload) throws IOException
305        {
306            JsonParser jp = getObjectMapper().getJsonFactory().createJsonParser(smilePayload);
307            root = getObjectMapper().readTree(jp);
308            jp.close();
309        }
310    
311        @Override
312        public String toString()
313        {
314            return root.toString();
315        }
316    }
