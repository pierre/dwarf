001    package com.ning.metrics.serialization.event;
002    
003    import org.codehaus.jackson.JsonParser;
004    import org.codehaus.jackson.JsonToken;
005    import org.codehaus.jackson.smile.SmileFactory;
006    import org.codehaus.jackson.smile.SmileGenerator;
007    import org.codehaus.jackson.smile.SmileParser;
008    import org.joda.time.DateTime;
009    
010    import java.io.IOException;
011    import java.io.ObjectInput;
012    import java.io.ObjectOutput;
013    import java.nio.charset.Charset;
014    
015    public class SmileEnvelopeEvent implements Event
016    {
017        /**
018         * We will use bogus charset to do truncation from char[] to byte[]; latin-1
019         * works since it is 8-bit subset of Unicode.
020         */
021        protected final static Charset CHARSET_LATIN1 = Charset.forName("ISO-8859-1");
022        
023        protected final static SmileFactory factory = new SmileFactory();
024        static {
025            // yes, full 'compression' by checking for repeating names, short string values:
026            factory.configure(SmileGenerator.Feature.CHECK_SHARED_NAMES, true);
027            factory.configure(SmileGenerator.Feature.CHECK_SHARED_STRING_VALUES, true);
028            // and for now let's not mandate header for input
029            factory.configure(SmileParser.Feature.REQUIRE_HEADER, false);
030        }
031    
032        public static final String SMILE_EVENT_DATETIME_TOKEN_NAME = "eventDate";
033        public static final String SMILE_EVENT_GRANULARITY_TOKEN_NAME = "eventGranularity";
034    
035        protected DateTime eventDateTime = null;
036        protected String eventName;
037        protected Granularity granularity = null;
038        protected byte[] payload;
039    
040        @Deprecated
041        public SmileEnvelopeEvent()
042        {
043        }
044    
045        /**
046         * Create an event from a binary JSON format
047         *
048         * @param eventName name the event schema
049         * @param message   String representation of a SMILE-serialized event
050         * @throws IOException if the payload is invalid
051         */
052        public SmileEnvelopeEvent(String eventName, String message) throws IOException
053        {
054            this.eventName = eventName;
055            parseEvent(bytesFromString(message));
056        }
057    
058        public SmileEnvelopeEvent(String eventName, byte[] payload) throws IOException
059        {
060            this.eventName = eventName;
061            this.eventDateTime = null;
062            parseEvent(payload);
063        }
064    
065        /**
066         * Constructor that will not (have to) parse payload, but is instead directly
067         * given metadata that accompanies (and should be included in) serialized payload
068         */
069        public SmileEnvelopeEvent(String eventName, byte[] payload,
070                DateTime eventDateTime, Granularity granularity)
071        {
072            this.eventName = eventName;
073            this.payload = payload;
074            this.eventDateTime = eventDateTime;
075            this.granularity = granularity;
076        }
077        
078        private void parseEvent(byte[] payload) throws IOException
079        {
080            if (payload == null) {
081                throw new IllegalArgumentException("Missing payload for parseEvent()");
082            }
083            
084            this.payload = payload;
085            granularity = null;
086            eventDateTime = null;
087    
088            JsonParser parser = factory.createJsonParser(payload);
089    
090            // Go through the stream once
091            JsonToken t = parser.nextToken();
092            while (t != null && (eventDateTime == null || granularity == null)) {
093                if (t == JsonToken.FIELD_NAME) {
094                    String name = parser.getCurrentName();
095                    t = parser.nextValue();
096                    if (SMILE_EVENT_DATETIME_TOKEN_NAME.equals(name)) {
097                        if (!t.isNumeric()) {
098                            throw new IOException(String.format("Invalid JSON: property '%s' has non-numeric value type [%s]", name, t.asString()));
099                        }
100                        try {
101                            eventDateTime = new DateTime(parser.getLongValue());
102                        }
103                        catch (NumberFormatException e) {
104                            throw new IOException(String.format("Invalid JSON: [%s] is not a timestamp", t.asString()));
105                        }
106                    } else if (SMILE_EVENT_GRANULARITY_TOKEN_NAME.equals(name)) {
107                        String text = parser.getText();
108                        try {
109                            granularity = Granularity.valueOf(parser.getText());
110                        }
111                        catch (IllegalArgumentException e) {
112                            throw new IOException(String.format("Invalid JSON: property '%s', value '%s' is not a valid granularity",
113                                    name, text));
114                        }
115                    }
116                } else {
117                    t = parser.nextToken();
118                }
119            }
120            if (granularity == null) {
121                granularity = Granularity.HOURLY;
122            }
123            parser.close();
124        }
125    
126        @Override
127        public DateTime getEventDateTime()
128        {
129            return eventDateTime;
130        }
131    
132        @Override
133        public String getName()
134        {
135            return eventName;
136        }
137    
138        @Override
139        public Granularity getGranularity()
140        {
141            return granularity;
142        }
143    
144        @Override
145        public String getVersion()
146        {
147            // TODO Not sure how to version these schemata. Need more thinking here.
148            return "1";
149        }
150    
151        @Override
152        public String getOutputDir(String prefix)
153        {
154            GranularityPathMapper pathMapper = new GranularityPathMapper(String.format("%s/%s", prefix, eventName), granularity);
155    
156            return pathMapper.getPathForDateTime(getEventDateTime());
157        }
158    
159        /**
160         * @return byte array (byte[]) that contains Smile event
161         */
162        @Override
163        public Object getData()
164        {
165            return payload; // This is a byte[] representation of a serialized SMILE event
166        }
167    
168        /**
169         * The object implements the writeExternal method to save its contents
170         * by calling the methods of DataOutput for its primitive values or
171         * calling the writeObject method of ObjectOutput for objects, strings,
172         * and arrays.
173         *
174         * @param out the stream to write the object to
175         * @throws java.io.IOException Includes any I/O exceptions that may occur
176         * @serialData Overriding methods should use this tag to describe
177         * the data layout of this Externalizable object.
178         * List the sequence of element types and, if possible,
179         * relate the element to a public/protected field and/or
180         * method of this Externalizable class.
181         */
182        @Override
183        public void writeExternal(ObjectOutput out) throws IOException
184        {
185            out.write(payload);
186        }
187    
188        /**
189         * The object implements the readExternal method to restore its
190         * contents by calling the methods of DataInput for primitive
191         * types and readObject for objects, strings and arrays.  The
192         * readExternal method must read the values in the same sequence
193         * and with the same types as were written by writeExternal.
194         *
195         * @param in the stream to read data from in order to restore the object
196         * @throws java.io.IOException    if I/O errors occur
197         * @throws ClassNotFoundException If the class for an object being
198         *                                restored cannot be found.
199         */
200        @Override
201        public void readExternal(ObjectInput in) throws IOException
202        {
203            int numBytes = in.readInt();
204            byte[] bytes = new byte[numBytes];
205    
206            in.readFully(bytes);
207            parseEvent(payload);
208            // One has to set the name manually
209        }
210    
211        public void setEventName(String eventName)
212        {
213            this.eventName = eventName;
214        }
215    
216        @Override
217        public String toString()
218        {
219            return "SmileEnvelopeEvent{" +
220                "eventDateTime=" + eventDateTime +
221                ", eventName='" + eventName + '\'' +
222                ", granularity=" + granularity +
223                ", payloadLength=" + payload.length +
224                '}';
225        }
226    
227        /*
228        /////////////////////////////////////////////////////////////////////// 
229        // Helper methods for sub-classes
230        /////////////////////////////////////////////////////////////////////// 
231         */
232    
233        protected static byte[] bytesFromString(String str)
234        { 
235            return str.getBytes(CHARSET_LATIN1); 
236        }
237    
238        protected String stringFromBytes(byte[] raw)
239        {
240            return new String(raw, CHARSET_LATIN1);
241        }
242    }
