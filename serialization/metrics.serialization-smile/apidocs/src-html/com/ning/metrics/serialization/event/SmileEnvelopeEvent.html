017    package com.ning.metrics.serialization.event;
018    
019    import org.codehaus.jackson.JsonEncoding;
020    import org.codehaus.jackson.JsonGenerator;
021    import org.codehaus.jackson.JsonNode;
022    import org.codehaus.jackson.JsonParser;
023    import org.codehaus.jackson.map.ObjectMapper;
024    import org.codehaus.jackson.smile.SmileFactory;
025    import org.codehaus.jackson.smile.SmileGenerator;
026    import org.codehaus.jackson.smile.SmileParser;
027    import org.joda.time.DateTime;
028    
029    import java.io.ByteArrayOutputStream;
030    import java.io.IOException;
031    import java.io.ObjectInput;
032    import java.io.ObjectOutput;
033    import java.nio.charset.Charset;
034    
035    /**
036     * Event representation of a single Smile event. This class is useful to send Json trees
037     * to the collector via the eventtracker library.
038     * Under the cover though, we use SmileBucketEvent on the wire (when sending to the collector) and in
039     * Hadoop sequencefiles (to leverage Smile back-references).
040     *
041     * @see com.ning.metrics.serialization.event.SmileBucketEvent
042     */
043    public class SmileEnvelopeEvent implements Event
044    {
045        // UTF-8 won't work!
046        public static final Charset CHARSET = Charset.forName("ISO-8859-1");
047    
048        public static final Charset NAME_CHARSET = Charset.forName("UTF-8");
049    
050        protected final static SmileFactory factory = new SmileFactory();
051    
052        static {
053            // yes, full 'compression' by checking for repeating names, short string values:
054            factory.configure(SmileGenerator.Feature.CHECK_SHARED_NAMES, true);
055            factory.configure(SmileGenerator.Feature.CHECK_SHARED_STRING_VALUES, true);
056            // and for now let's not mandate header for input
057            factory.configure(SmileParser.Feature.REQUIRE_HEADER, false);
058        }
059    
060        private static final ObjectMapper objectMapper = new ObjectMapper(factory);
061    
062        public static final String SMILE_EVENT_DATETIME_TOKEN_NAME = "eventDate";
063        public static final String SMILE_EVENT_GRANULARITY_TOKEN_NAME = "eventGranularity";
064    
065        protected DateTime eventDateTime = null;
066        protected String eventName;
067        protected Granularity granularity = null;
068        protected JsonNode root;
069    
070        @Deprecated
071        public SmileEnvelopeEvent()
072        {
073        }
074    
075        public SmileEnvelopeEvent(String eventName, JsonNode node)
076        {
077            this(eventName, null, node);
078        }
079    
080        public SmileEnvelopeEvent(String eventName, Granularity granularity, JsonNode node)
081        {
082            this.eventName = eventName;
083            this.root = node;
084            this.granularity = granularity;
085    
086            setEventPropertiesFromNode(node);
087        }
088    
089        public SmileEnvelopeEvent(String eventName, byte[] inputBytes, DateTime eventDateTime, Granularity granularity) throws IOException
090        {
091            this.eventName = eventName;
092            this.eventDateTime = eventDateTime;
093            this.granularity = granularity;
094            setPayloadFromByteArray(inputBytes);
095        }
096    
097        @Override
098        public DateTime getEventDateTime()
099        {
100            return eventDateTime;
101        }
102    
103        @Override
104        public String getName()
105        {
106            return eventName;
107        }
108    
109        @Override
110        public Granularity getGranularity()
111        {
112            return granularity;
113        }
114    
115        @Override
116        public String getVersion()
117        {
118            // TODO Not sure how to version these schemata. Need more thinking here.
119            return "1";
120        }
121    
122        @Override
123        public String getOutputDir(String prefix)
124        {
125            GranularityPathMapper pathMapper = new GranularityPathMapper(String.format("%s/%s", prefix, eventName), granularity);
126    
127            return pathMapper.getPathForDateTime(getEventDateTime());
128        }
129    
130        /**
131         * @return a JsonNode representation of a SMILE event (json)
132         */
133        @Override
134        public Object getData()
135        {
136            return root;
137        }
138    
139        @Override
140        public byte[] getSerializedEvent()
141        {
142            ByteArrayOutputStream outStream = new ByteArrayOutputStream();
143    
144            try {
145                JsonGenerator gen = objectMapper.getJsonFactory().createJsonGenerator(outStream, JsonEncoding.UTF8);
146                objectMapper.writeTree(gen, root);
147                gen.close();
148            }
149            catch (IOException e) {
150                return null;
151            }
152    
153            return outStream.toByteArray();
154        }
155    
156        /**
157         * The object implements the writeExternal method to save its contents
158         * by calling the methods of DataOutput for its primitive values or
159         * calling the writeObject method of ObjectOutput for objects, strings,
160         * and arrays.
161         *
162         * @param out the stream to write the object to
163         * @throws java.io.IOException Includes any I/O exceptions that may occur
164         * @serialData Overriding methods should use this tag to describe
165         * the data layout of this Externalizable object.
166         * List the sequence of element types and, if possible,
167         * relate the element to a public/protected field and/or
168         * method of this Externalizable class.
169         */
170        @Override
171        public void writeExternal(ObjectOutput out) throws IOException
172        {
173            // Name of the event
174            byte[] eventNameBytes = eventName.getBytes(NAME_CHARSET);
175            out.writeInt(eventNameBytes.length);
176            out.write(eventNameBytes);
177    
178            byte[] payloadBytes = getSerializedEvent();
179    
180            // Size of Smile payload. Needed for deserialization, see below
181            out.writeInt(payloadBytes.length);
182    
183            out.write(payloadBytes);
184        }
185    
186        /**
187         * The object implements the readExternal method to restore its
188         * contents by calling the methods of DataInput for primitive
189         * types and readObject for objects, strings and arrays.  The
190         * readExternal method must read the values in the same sequence
191         * and with the same types as were written by writeExternal.
192         *
193         * @param in the stream to read data from in order to restore the object
194         * @throws java.io.IOException    if I/O errors occur
195         * @throws ClassNotFoundException If the class for an object being
196         *                                restored cannot be found.
197         */
198        @Override
199        public void readExternal(ObjectInput in) throws IOException
200        {
201            // Name of the event first
202            int smileEventNameBytesSize = in.readInt();
203            byte[] eventNameBytes = new byte[smileEventNameBytesSize];
204            in.readFully(eventNameBytes);
205            eventName = new String(eventNameBytes, NAME_CHARSET);
206    
207            // Then payload
208            int smilePayloadSize = in.readInt();
209            byte[] smilePayload = new byte[smilePayloadSize];
210            in.readFully(smilePayload);
211    
212            setPayloadFromByteArray(smilePayload);
213    
214            setEventPropertiesFromNode(root);
215        }
216    
217        private void setEventPropertiesFromNode(JsonNode node)
218        {
219            eventDateTime = getEventDateTimeFromJson(node);
220    
221            if (granularity == null) {
222                granularity = getGranularityFromJson(node);
223            }
224        }
225    
226        public static DateTime getEventDateTimeFromJson(JsonNode node)
227        {
228            JsonNode eventDateTimeNode = node.path(SMILE_EVENT_DATETIME_TOKEN_NAME);
229    
230            DateTime nodeDateTime = new DateTime();
231            if (!eventDateTimeNode.isMissingNode()) {
232                nodeDateTime = new DateTime(eventDateTimeNode.getLongValue());
233            }
234    
235            return nodeDateTime;
236        }
237    
238        public static Granularity getGranularityFromJson(JsonNode node)
239        {
240            JsonNode granularityNode = node.path(SMILE_EVENT_GRANULARITY_TOKEN_NAME);
241    
242            Granularity nodeGranularity = Granularity.HOURLY;
243            if (!granularityNode.isMissingNode()) {
244                try {
245                    nodeGranularity = Granularity.valueOf(granularityNode.getValueAsText());
246                }
247                catch (IllegalArgumentException e) {
248                    nodeGranularity = null;
249                }
250            }
251    
252            return nodeGranularity;
253        }
254    
255        private void setPayloadFromByteArray(byte[] smilePayload) throws IOException
256        {
257            JsonParser jp = objectMapper.getJsonFactory().createJsonParser(smilePayload);
258            root = objectMapper.readTree(jp);
259            jp.close();
260        }
261    
262        @Override
263        public String toString()
264        {
265            return root.toString();
266        }
267    }
