017    package com.ning.metrics.serialization.schema;
018    
019    import java.util.ArrayList;
020    import java.util.Collections;
021    import java.util.Comparator;
022    import java.util.HashMap;
023    
024    /**
025     * A Schema is a collection of SchemaFields.
026     *
027     * @see SchemaField
028     */
029    public class Schema
030    {
031        private static final long serialVersionUID = 1L;
032    
033        private final String name;
034        private HashMap<Short, SchemaField> schemaFields = new HashMap<Short, SchemaField>();
035    
036        public Schema(String name)
037        {
038            this.name = name;
039        }
040    
041        public String getName()
042        {
043            return name;
044        }
045    
046        public void addSchemaField(SchemaField schemaField)
047        {
048            schemaFields.put(schemaField.getId(), schemaField);
049        }
050    
051        public SchemaField getFieldByPosition(short id)
052        {
053            return schemaFields.get(id);
054        }
055    
056        public SchemaField getFieldByName(String name)
057        {
058            for (SchemaField field : schemaFields.values()) {
059                if (field.getName().equals(name)) {
060                    return field;
061                }
062            }
063    
064            return null;
065        }
066    
067        /**
068         * Get the schema as a collection of fields.
069         * We guarantee the ordering by field id.
070         *
071         * @return the sorted collection of fields
072         */
073        public ArrayList<SchemaField> getSchema()
074        {
075            ArrayList<SchemaField> items = new ArrayList<SchemaField>(schemaFields.values());
076    
077            Collections.sort(items, new Comparator<SchemaField>()
078            {
079                @Override
080                public int compare(SchemaField left, SchemaField right)
081                {
082                    return left.getId() - right.getId();
083                }
084            });
085    
086            return items;
087        }
088    }
