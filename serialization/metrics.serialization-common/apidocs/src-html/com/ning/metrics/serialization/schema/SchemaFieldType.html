019    import java.util.Arrays;

021    /**
022     * Hack - import the org.apache.thrift.protocol.TType class constants here
023     * to avoid a dependency on org.apache.thrift:thrift.
024     *
025     * These should hopefully not change often.
026     */
027    final class TType
028    {
029        public static final byte STOP = 0;
030        public static final byte VOID = 1;
031        public static final byte BOOL = 2;
032        public static final byte BYTE = 3;
033        public static final byte DOUBLE = 4;
034        public static final byte I16 = 6;
035        public static final byte I32 = 8;
036        public static final byte I64 = 10;
037        public static final byte STRING = 11;
038        public static final byte STRUCT = 12;
039        public static final byte MAP = 13;
040        public static final byte SET = 14;
041        public static final byte LIST = 15;
042        public static final byte ENUM = 16;
043    }
044    
045    /**
046     * The SchemaFieldType ties the high level data representation
047     * (e.g. date, IP) to the underlying protocol types (e.g. i64, ...).
048     */
049    public enum SchemaFieldType
050    {
051        BOOLEAN(TType.BOOL, Boolean.class),
052        BYTE(TType.BYTE, Byte.class),
053        SHORT(TType.I16, Short.class),
054        INTEGER(TType.I32, Integer.class),
055        LONG(TType.I64, Long.class),
056        DOUBLE(TType.DOUBLE, Double.class),
057        STRING(TType.STRING, String.class),
058        DATE(TType.I64, Long.class),
059        IP(TType.I32, Integer.class);
060    
061        private final byte thriftType;
062        private final Class<?> smileType;
063    
064        SchemaFieldType(byte thriftType, Class<?> smileType)
065        {
066            this.thriftType = thriftType;
067            this.smileType = smileType;
068        }
069    
070        public byte getThriftType()
071        {
072            return thriftType;
073        }
074    
075        public Class<?> getSmileType()
076        {
077            return smileType;
078        }
079    
080        public SchemaField createSchemaField(String name, short id)
081        {
082            switch (this) {
083                case DATE:
084                    return new DateSchemaField(name, id);
085                case IP:
086                    return new IpSchemaField(name, id);
087                default:
088                    return new IdentitySchemaField(name, this, id);
089            }
090        }
091    
092        /**
093         * Used primarily by Goodwill to create schemata.
094         *
095         * @param name name of the field
096         * @param type plain text name of the field
097         * @param id   position in the schema (should be unique by schema)
098         * @return newly created SchemaField
099         */
100        @SuppressWarnings("unused")
101        public static SchemaField createSchemaField(String name, String type, short id)
102        {
103            try {
104                SchemaFieldType schemaFieldType = valueOf(type.toUpperCase());
105                return schemaFieldType.createSchemaField(name, id);
106            }
107            catch (IllegalArgumentException e) {
108                throw new IllegalArgumentException(String.format(
109                    "Expected type to be one of %s but got %s",
110                    Arrays.toString(values()).toLowerCase(),
111                    type
112                ));
113            }
114        }
115    }
