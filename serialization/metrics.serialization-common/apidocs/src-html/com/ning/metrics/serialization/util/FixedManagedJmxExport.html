001    package com.ning.metrics.serialization.util;
002    
003    import org.apache.log4j.Logger;
004    
005    import javax.management.Descriptor;
006    import javax.management.IntrospectionException;
007    import javax.management.MBeanServer;
008    import javax.management.ObjectName;
009    import javax.management.modelmbean.DescriptorSupport;
010    import javax.management.modelmbean.ModelMBeanAttributeInfo;
011    import javax.management.modelmbean.ModelMBeanConstructorInfo;
012    import javax.management.modelmbean.ModelMBeanInfo;
013    import javax.management.modelmbean.ModelMBeanInfoSupport;
014    import javax.management.modelmbean.ModelMBeanNotificationInfo;
015    import javax.management.modelmbean.ModelMBeanOperationInfo;
016    import javax.management.modelmbean.RequiredModelMBean;
017    import java.lang.management.ManagementFactory;
018    import java.lang.reflect.Method;
019    import java.util.ArrayList;
020    import java.util.HashMap;
021    import java.util.List;
022    import java.util.Map;
023    import java.util.regex.Matcher;
024    import java.util.regex.Pattern;
025    
026    /**
027     * Expose all @Managed classes to JMX
028     */
029    public class FixedManagedJmxExport
030    {
031        private static final Logger log = Logger.getLogger(FixedManagedJmxExport.class);
032        private static final Pattern getterOrSetterPattern = Pattern.compile("(get|set|is)(.)(.*)");
033        private static final MBeanServer server = ManagementFactory.getPlatformMBeanServer();
034    
035        @SuppressWarnings("unused")
036        public static void export(String name, Object monitoredObject)
037        {
038            try {
039                ObjectName objectName = new ObjectName(name);
040    
041                RequiredModelMBean mbean = getMBean(monitoredObject);
042                mbean.setManagedResource(monitoredObject, "objectReference");
043    
044                // register the model MBean in the MBean server
045                server.registerMBean(mbean, objectName);
046            }
047            catch (Exception e) {
048                throw new RuntimeException(e);
049            }
050        }
051    
052        private static RequiredModelMBean getMBean(Object monitoredObject)
053        {
054            try {
055                ModelMBeanInfo info = buildInfo(monitoredObject.getClass());
056    
057                return new RequiredModelMBean(new ModelMBeanInfoSupport(info));
058            }
059            catch (Exception e) {
060                throw new RuntimeException(e);
061            }
062        }
063    
064        private static boolean includeMethod(Method method)
065        {
066            log.debug(String.format("Include method[%s]? %s", method.getName(), method.getAnnotation(Managed.class) != null));
067            return method.getAnnotation(Managed.class) != null;
068        }
069    
070        private static String getDescription(Method method)
071        {
072            Managed annotation = method.getAnnotation(Managed.class);
073            if (annotation == null) {
074                throw new RuntimeException("FixedManagedJmxExportScope is trying to export a method without a managed annotation.  Method name: " + method.getName());
075            }
076    
077            log.debug(String.format("Method[%s] given description[%s].", method.getName(), annotation.description()));
078            return annotation.description();
079        }
080    
081        private static ModelMBeanInfo buildInfo(Class clazz)
082            throws IntrospectionException
083        {
084            List<OperationDescriptor> operations = new ArrayList<OperationDescriptor>();
085            Map<String, AttributeDescriptor> attributeDescriptors = new HashMap<String, AttributeDescriptor>();
086    
087            String className = clazz.getName();
088    
089            while (clazz != null) {
090                for (Method method : clazz.getMethods()) {
091                    if (!includeMethod(method)) {
092                        continue;
093                    }
094    
095                    String name = method.getName();
096                    String description = getDescription(method);
097    
098                    Matcher matcher = getterOrSetterPattern.matcher(name);
099                    OperationDescriptor operation = null;
100    
101                    if (matcher.matches()) {
102                        String type = matcher.group(1);
103                        String first = matcher.group(2);
104                        String rest = matcher.group(3);
105    
106                        String attributeName = first + (rest != null ? rest : "");
107    
108                        AttributeDescriptor descriptor = attributeDescriptors.get(attributeName);
109                        if (descriptor == null) {
110                            descriptor = new AttributeDescriptor(attributeName, description);
111                        }
112    
113                        if ((type.equals("get") || type.equals("is")) && method.getParameterTypes().length == 0) {
114                            descriptor.setGetter(method);
115                            operation = new OperationDescriptor(method, "getter");
116                            attributeDescriptors.put(attributeName, descriptor);
117                        }
118                        else if (type.equals("set") && method.getParameterTypes().length == 1) {
119                            descriptor.setSetter(method);
120                            operation = new OperationDescriptor(method, "setter");
121                            attributeDescriptors.put(attributeName, descriptor);
122                        }
123                    }
124    
125                    if (operation == null) {
126                        operation = new OperationDescriptor(method, "operation");
127                    }
128    
129                    operations.add(operation);
130                }
131    
132                clazz = clazz.getSuperclass();
133            }
134    
135    
136            List<ModelMBeanAttributeInfo> attributeInfos = new ArrayList<ModelMBeanAttributeInfo>();
137    
138            for (AttributeDescriptor attribute : attributeDescriptors.values()) {
139                Descriptor descriptor = new DescriptorSupport();
140                descriptor.setField("name", attribute.getName());
141                descriptor.setField("descriptorType", "attribute");
142                if (attribute.getGetter() != null) {
143                    descriptor.setField("getMethod", attribute.getGetter().getName());
144                }
145                if (attribute.getSetter() != null) {
146                    descriptor.setField("setMethod", attribute.getSetter().getName());
147                }
148    
149                attributeInfos.add(new ModelMBeanAttributeInfo(attribute.getName(),
150                    attribute.getDescription(),
151                    attribute.getGetter(),
152                    attribute.getSetter(),
153                    descriptor));
154            }
155    
156            List<ModelMBeanOperationInfo> operationInfos = new ArrayList<ModelMBeanOperationInfo>();
157    
158            for (OperationDescriptor operation : operations) {
159                Descriptor descriptor = new DescriptorSupport();
160                descriptor.setField("name", operation.getMethod().getName());
161                descriptor.setField("class", className);
162                descriptor.setField("descriptorType", "operation");
163                descriptor.setField("role", operation.getRole());
164    
165                operationInfos.add(new ModelMBeanOperationInfo("",
166                    operation.getMethod(),
167                    descriptor));
168    
169                // TODO: discover parameter names
170                // see http://jroller.com/eu/entry/using_asm_to_read_mathod
171            }
172    
173            return new ModelMBeanInfoSupport(className, null,
174                attributeInfos.toArray(new ModelMBeanAttributeInfo[attributeInfos.size()]),
175                new ModelMBeanConstructorInfo[0],
176                operationInfos.toArray(new ModelMBeanOperationInfo[attributeInfos.size()]),
177                new ModelMBeanNotificationInfo[0]);
178        }
179    
180        private static class OperationDescriptor
181        {
182            private final Method method;
183            private final String role;
184    
185            public OperationDescriptor(Method method, String role)
186            {
187                this.method = method;
188                this.role = role;
189            }
190    
191            public Method getMethod()
192            {
193                return method;
194            }
195    
196            public String getRole()
197            {
198                return role;
199            }
200        }
201    
202        private static class AttributeDescriptor
203        {
204            private Method getter = null;
205            private Method setter = null;
206            private final String description;
207            private final String name;
208    
209            private AttributeDescriptor(String name, String description)
210            {
211                this.name = name;
212                this.description = description;
213            }
214    
215            public String getName()
216            {
217                return name;
218            }
219    
220            public Method getGetter()
221            {
222                return getter;
223            }
224    
225            public void setGetter(Method getter)
226            {
227                this.getter = getter;
228            }
229    
230            public Method getSetter()
231            {
232                return setter;
233            }
234    
235            public void setSetter(Method setter)
236            {
237                this.setter = setter;
238            }
239    
240            public String getDescription()
241            {
242                return description;
243            }
244        }
245    }
