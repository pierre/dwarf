017    package com.ning.metrics.serialization.util;
018    
019    import org.apache.log4j.Logger;
020    
021    import javax.management.Descriptor;
022    import javax.management.IntrospectionException;
023    import javax.management.MBeanServer;
024    import javax.management.ObjectName;
025    import javax.management.modelmbean.DescriptorSupport;
026    import javax.management.modelmbean.ModelMBeanAttributeInfo;
027    import javax.management.modelmbean.ModelMBeanConstructorInfo;
028    import javax.management.modelmbean.ModelMBeanInfo;
029    import javax.management.modelmbean.ModelMBeanInfoSupport;
030    import javax.management.modelmbean.ModelMBeanNotificationInfo;
031    import javax.management.modelmbean.ModelMBeanOperationInfo;
032    import javax.management.modelmbean.RequiredModelMBean;
033    import java.lang.management.ManagementFactory;
034    import java.lang.reflect.Method;
035    import java.util.ArrayList;
036    import java.util.HashMap;
037    import java.util.List;
038    import java.util.Map;
039    import java.util.regex.Matcher;
040    import java.util.regex.Pattern;
041    
042    /**
043     * Expose all @Managed classes to JMX
044     */
045    public class FixedManagedJmxExport
046    {
047        private static final Logger log = Logger.getLogger(FixedManagedJmxExport.class);
048        private static final Pattern getterOrSetterPattern = Pattern.compile("(get|set|is)(.)(.*)");
049        private static final MBeanServer server = ManagementFactory.getPlatformMBeanServer();
050    
051        @SuppressWarnings("unused")
052        public static void export(String name, Object monitoredObject)
053        {
054            try {
055                ObjectName objectName = new ObjectName(name);
056    
057                RequiredModelMBean mbean = getMBean(monitoredObject);
058                mbean.setManagedResource(monitoredObject, "objectReference");
059    
060                // register the model MBean in the MBean server
061                server.registerMBean(mbean, objectName);
062            }
063            catch (Exception e) {
064                throw new RuntimeException(e);
065            }
066        }
067    
068        private static RequiredModelMBean getMBean(Object monitoredObject)
069        {
070            try {
071                ModelMBeanInfo info = buildInfo(monitoredObject.getClass());
072    
073                return new RequiredModelMBean(new ModelMBeanInfoSupport(info));
074            }
075            catch (Exception e) {
076                throw new RuntimeException(e);
077            }
078        }
079    
080        private static boolean includeMethod(Method method)
081        {
082            log.debug(String.format("Include method[%s]? %s", method.getName(), method.getAnnotation(Managed.class) != null));
083            return method.getAnnotation(Managed.class) != null;
084        }
085    
086        private static String getDescription(Method method)
087        {
088            Managed annotation = method.getAnnotation(Managed.class);
089            if (annotation == null) {
090                throw new RuntimeException("FixedManagedJmxExportScope is trying to export a method without a managed annotation.  Method name: " + method.getName());
091            }
092    
093            log.debug(String.format("Method[%s] given description[%s].", method.getName(), annotation.description()));
094            return annotation.description();
095        }
096    
097        private static ModelMBeanInfo buildInfo(Class<?> clazz)
098            throws IntrospectionException
099        {
100            List<OperationDescriptor> operations = new ArrayList<OperationDescriptor>();
101            Map<String, AttributeDescriptor> attributeDescriptors = new HashMap<String, AttributeDescriptor>();
102    
103            String className = clazz.getName();
104    
105            while (clazz != null) {
106                for (Method method : clazz.getMethods()) {
107                    if (!includeMethod(method)) {
108                        continue;
109                    }
110    
111                    String name = method.getName();
112                    String description = getDescription(method);
113    
114                    Matcher matcher = getterOrSetterPattern.matcher(name);
115                    OperationDescriptor operation = null;
116    
117                    if (matcher.matches()) {
118                        String type = matcher.group(1);
119                        String first = matcher.group(2);
120                        String rest = matcher.group(3);
121    
122                        String attributeName = first + (rest != null ? rest : "");
123    
124                        AttributeDescriptor descriptor = attributeDescriptors.get(attributeName);
125                        if (descriptor == null) {
126                            descriptor = new AttributeDescriptor(attributeName, description);
127                        }
128    
129                        if ((type.equals("get") || type.equals("is")) && method.getParameterTypes().length == 0) {
130                            descriptor.setGetter(method);
131                            operation = new OperationDescriptor(method, "getter");
132                            attributeDescriptors.put(attributeName, descriptor);
133                        }
134                        else if (type.equals("set") && method.getParameterTypes().length == 1) {
135                            descriptor.setSetter(method);
136                            operation = new OperationDescriptor(method, "setter");
137                            attributeDescriptors.put(attributeName, descriptor);
138                        }
139                    }
140    
141                    if (operation == null) {
142                        operation = new OperationDescriptor(method, "operation");
143                    }
144    
145                    operations.add(operation);
146                }
147    
148                clazz = clazz.getSuperclass();
149            }
150    
151    
152            List<ModelMBeanAttributeInfo> attributeInfos = new ArrayList<ModelMBeanAttributeInfo>();
153    
154            for (AttributeDescriptor attribute : attributeDescriptors.values()) {
155                Descriptor descriptor = new DescriptorSupport();
156                descriptor.setField("name", attribute.getName());
157                descriptor.setField("descriptorType", "attribute");
158                if (attribute.getGetter() != null) {
159                    descriptor.setField("getMethod", attribute.getGetter().getName());
160                }
161                if (attribute.getSetter() != null) {
162                    descriptor.setField("setMethod", attribute.getSetter().getName());
163                }
164    
165                attributeInfos.add(new ModelMBeanAttributeInfo(attribute.getName(),
166                    attribute.getDescription(),
167                    attribute.getGetter(),
168                    attribute.getSetter(),
169                    descriptor));
170            }
171    
172            List<ModelMBeanOperationInfo> operationInfos = new ArrayList<ModelMBeanOperationInfo>();
173    
174            for (OperationDescriptor operation : operations) {
175                Descriptor descriptor = new DescriptorSupport();
176                descriptor.setField("name", operation.getMethod().getName());
177                descriptor.setField("class", className);
178                descriptor.setField("descriptorType", "operation");
179                descriptor.setField("role", operation.getRole());
180    
181                operationInfos.add(new ModelMBeanOperationInfo("",
182                    operation.getMethod(),
183                    descriptor));
184    
185                // TODO: discover parameter names
186                // see http://jroller.com/eu/entry/using_asm_to_read_mathod
187            }
188    
189            return new ModelMBeanInfoSupport(className, null,
190                attributeInfos.toArray(new ModelMBeanAttributeInfo[attributeInfos.size()]),
191                new ModelMBeanConstructorInfo[0],
192                operationInfos.toArray(new ModelMBeanOperationInfo[attributeInfos.size()]),
193                new ModelMBeanNotificationInfo[0]);
194        }
195    
196        private static class OperationDescriptor
197        {
198            private final Method method;
199            private final String role;
200    
201            public OperationDescriptor(Method method, String role)
202            {
203                this.method = method;
204                this.role = role;
205            }
206    
207            public Method getMethod()
208            {
209                return method;
210            }
211    
212            public String getRole()
213            {
214                return role;
215            }
216        }
217    
218        private static class AttributeDescriptor
219        {
220            private Method getter = null;
221            private Method setter = null;
222            private final String description;
223            private final String name;
224    
225            private AttributeDescriptor(String name, String description)
226            {
227                this.name = name;
228                this.description = description;
229            }
230    
231            public String getName()
232            {
233                return name;
234            }
235    
236            public Method getGetter()
237            {
238                return getter;
239            }
240    
241            public void setGetter(Method getter)
242            {
243                this.getter = getter;
244            }
245    
246            public Method getSetter()
247            {
248                return setter;
249            }
250    
251            public void setSetter(Method setter)
252            {
253                this.setter = setter;
254            }
255    
256            public String getDescription()
257            {
258                return description;
259            }
260        }
