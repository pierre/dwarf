031    /**
032     * Describe a SchemaField in Goodwill.
033     * This is basically the union of a SchemaField and extra metadata for the SQL sink.
034     *
035     * @see com.ning.metrics.serialization.schema.SchemaField
036     */
037    public class GoodwillSchemaField
038    {
039        private static final ObjectMapper mapper = new ObjectMapper();
040    
041        public static final String JSON_THRIFT_FIELD_NAME = "name";
042        public static final String JSON_THRIFT_FIELD_TYPE = "type";
043        public static final String JSON_THRIFT_FIELD_ID = "position";
044        public static final String JSON_THRIFT_FIELD_DESCRIPTION = "description";
045    
046        /*
047         * When storing the thrifts in a database/datawarehouse, use these fields to describe
048         * the associated sql type and length.
049         * Given the large range of data types (Netezza, Oracle, MySQL, ...), no enforcement
050         * is performed (plain Strings).
051         */
052        public static final String JSON_THRIFT_FIELD_SQL_KEY = "sql";
053        public static final String JSON_THRIFT_FIELD_SQL_TYPE = "type";
054        public static final String JSON_THRIFT_FIELD_SQL_LENGTH = "length";
055        public static final String JSON_THRIFT_FIELD_SQL_SCALE = "scale";
056        public static final String JSON_THRIFT_FIELD_SQL_PRECISION = "precision";
057    
058        private final SchemaField schemaField;
059        private final String description;
060        private Sql sql;
061    
062        /**
063         * Jackson constructor
064         * <p/>
065         * {
066         * "name": "myField",
067         * "type": "string",
068         * "position": 1,
069         * "description": "string",
070         * "sql": {
071         * "type": "nvarchar",
072         * "length": 255,
073         * "scale": null,
074         * "precision": null
075         * }
076         *
077         * @param name        Schema field name
078         * @param type        Schema field type. This is not necessarily a Thrift type. @see SchemaFieldType
079         * @param id          field position
080         * @param description Short description of the field
081         * @param sql         SQL object (used by the sink)
082         */
083        @JsonCreator
084        @SuppressWarnings("unused")
085        public GoodwillSchemaField(
086            @JsonProperty(JSON_THRIFT_FIELD_NAME) String name,
087            @JsonProperty(JSON_THRIFT_FIELD_TYPE) String type,
088            @JsonProperty(JSON_THRIFT_FIELD_ID) short id,
089            @JsonProperty(JSON_THRIFT_FIELD_DESCRIPTION) String description,
090            @JsonProperty(JSON_THRIFT_FIELD_SQL_KEY) Sql sql
091        )
092        {
093            this(name, type, id, description, sql == null ? null : sql.getType(), sql == null ? null : sql.getLength(), sql == null ? null : sql.getScale(), sql == null ? null : sql.getPrecision());
094        }
095    
096        /**
097         * Manual constructor, typically used by Goodwill stores.
098         *
099         * @param name         Schema field name
100         * @param type         Schema field type. This is not necessarily a Thrift type. @see SchemaFieldType
101         * @param id           field position
102         * @param description  Short description of the field
103         * @param sqlType      SQL type (varchar, int, ...)
104         * @param sqlLength    SQL type length
105         * @param sqlScale     SQL type scale
106         * @param sqlPrecision SQL type precision
107         */
108        public GoodwillSchemaField(
109            String name,
110            String type,
111            short id,
112            String description,
113            String sqlType,
114            Integer sqlLength,
115            Integer sqlScale,
116            Integer sqlPrecision
117        )
118        {
119            if (name == null) {
120                throw new IllegalArgumentException("GoodwillSchemaField name can't be null");
121            }
122    
123            if ((sqlType == null || sqlType.equals("string")) && (sqlScale != null || sqlPrecision != null)) {
124                throw new IllegalArgumentException("Strings cannot have a scale or precision");
125            }
126    
127            this.schemaField = SchemaFieldType.createSchemaField(name, type, id);
128    
129            // Optional fields
130            sql = new Sql(sqlType, sqlLength, sqlScale, sqlPrecision);
131            this.description = description;
132        }
133    
134        public GoodwillSchemaField(SchemaField field)
135        {
136            this(field.getName(), field.getType().name(), field.getId(), null, null);
137        }
138    
139        public static GoodwillSchemaField decode(
140            String thriftItemJson
141        ) throws IOException
142        {
143            return mapper.readValue(thriftItemJson, GoodwillSchemaField.class);
144        }
145    
146        @JsonValue
147        @SuppressWarnings({"unchecked", "unused"})
148        public ImmutableMap toMap()
149        {
150            return new ImmutableMap.Builder()
151                .put(JSON_THRIFT_FIELD_NAME, getName())
152                .put(JSON_THRIFT_FIELD_TYPE, getType())
153                .put(JSON_THRIFT_FIELD_ID, getId())
154                .put(JSON_THRIFT_FIELD_DESCRIPTION, getDescription() == null ? "" : getDescription())
155                .put(JSON_THRIFT_FIELD_SQL_KEY, getSql() == null ? "" : getSql())
156                .build();
157        }
158    
159        /**
160         * Extra information for the SQL Sink
161         */
162        public static class Sql
163        {
164            private String type;
165            private Integer length;
166            private Integer scale;
167            private Integer precision;
168    
169            @JsonCreator
170            public Sql(
171                @JsonProperty(JSON_THRIFT_FIELD_SQL_TYPE) String type,
172                @JsonProperty(JSON_THRIFT_FIELD_SQL_LENGTH) Integer length,
173                @JsonProperty(JSON_THRIFT_FIELD_SQL_SCALE) Integer scale,
174                @JsonProperty(JSON_THRIFT_FIELD_SQL_PRECISION) Integer precision)
175            {
176                this.type = type;
177                this.length = length;
178                this.scale = scale;
179                this.precision = precision;
180            }
181    
182    
183            @JsonValue
184            @SuppressWarnings({"unchecked", "unused"})
185            public ImmutableMap toMap()
186            {
187                return new ImmutableMap.Builder()
188                    .put(JSON_THRIFT_FIELD_SQL_TYPE, getType() == null ? "" : getType())
189                    .put(JSON_THRIFT_FIELD_SQL_LENGTH, getLength() == null ? "" : getLength())
190                    .put(JSON_THRIFT_FIELD_SQL_SCALE, getScale() == null ? "" : getScale())
191                    .put(JSON_THRIFT_FIELD_SQL_PRECISION, getPrecision() == null ? "" : getPrecision())
192                    .build();
193            }
194    
195            public String getType()
196            {
197                return type;
198            }
199    
200            public Integer getLength()
201            {
202                return length;
203            }
204    
205            public Integer getScale()
206            {
207                return scale;
208            }
209    
210            public Integer getPrecision()
211            {
212                return precision;
213            }
214        }
215    
216        public String getName()
217        {
218            return schemaField.getName();
219        }
220    
221        public SchemaFieldType getType()
222        {
223            return schemaField.getType();
224        }
225    
226        public short getId()
227        {
228            return schemaField.getId();
229        }
230    
231        public Sql getSql()
232        {
233            return sql;
234        }
235    
236        public String getDescription()
237        {
238            return description;
239        }
240    
241        @Override
242        public String toString()
243        {
244            try {
245                return toJSON().toString();
246            }
247            catch (JsonGenerationException e) {
248                return "GoodwillSchemaField{" +
249                    JSON_THRIFT_FIELD_NAME + "='" + getName() + '\'' +
250                    ", " + JSON_THRIFT_FIELD_TYPE + "='" + getType() + '\'' +
251                    ", " + JSON_THRIFT_FIELD_ID + "=" + getId() +
252                    ", " + JSON_THRIFT_FIELD_SQL_TYPE + "='" + sql.type + '\'' +
253                    ", " + JSON_THRIFT_FIELD_SQL_LENGTH + "=" + sql.length +
254                    ", " + JSON_THRIFT_FIELD_SQL_SCALE + "=" + sql.scale +
255                    ", " + JSON_THRIFT_FIELD_SQL_PRECISION + "=" + sql.precision +
256                    ", " + JSON_THRIFT_FIELD_DESCRIPTION + "=" + description +
257                    '}';
258            }
259            catch (IOException e) {
260                return "GoodwillSchemaField{" +
261                    JSON_THRIFT_FIELD_NAME + "='" + getName() + '\'' +
262                    ", " + JSON_THRIFT_FIELD_TYPE + "='" + getType() + '\'' +
263                    ", " + JSON_THRIFT_FIELD_ID + "=" + getId() +
264                    ", " + JSON_THRIFT_FIELD_SQL_TYPE + "='" + sql.type + '\'' +
265                    ", " + JSON_THRIFT_FIELD_SQL_LENGTH + "=" + sql.length +
266                    ", " + JSON_THRIFT_FIELD_SQL_SCALE + "=" + sql.scale +
267                    ", " + JSON_THRIFT_FIELD_SQL_PRECISION + "=" + sql.precision +
268                    ", " + JSON_THRIFT_FIELD_DESCRIPTION + "=" + description +
269                    '}';
270            }
271        }

273        /**
274         * Create a JSON representation of the GoodwillSchemaField. It will always contain
275         * the name, type and position. Description and SQL attributes are however
276         * optional.
277         *
278         * @return JSONObject containing all fields
279         * @throws IOException             if a serialization exception occurs
280         * @throws JsonGenerationException if a serialization exception occurs
281         */
282        public ByteArrayOutputStream toJSON() throws IOException
283        {
284            ByteArrayOutputStream out = new ByteArrayOutputStream();
285            mapper.writeValue(out, this);
286            return out;
287        }

289        /**
290         * Pretty print the SQL type.
291         * TODO: add layer of abstraction, too Netezza specific
292         *
293         * @return a human readable representation of the SQL type
294         */
295        @SuppressWarnings("unused")
296        public String getFullSQLType()
297        {
298            String fullSQLType = null;
299    
300            if (sql.type == null) {
301                return null;
302            }
303            else if ((sql.type.equals("decimal")) || sql.type.equals("numeric")) {
304                if (sql.precision != null) {
305                    if (sql.scale != null) {
306                        fullSQLType = sql.type + "(" + sql.precision + ", " + sql.scale + ")";
307                    }
308                    else {
309                        fullSQLType = sql.type + "(" + sql.precision + ")";
310                    }
311                }
312            }
313            else {
314                if (sql.type.equals("nvarchar") || sql.type.equals("varchar")) {
315                    if (sql.length != null) {
316                        fullSQLType = sql.type + "(" + sql.length + ")";
317                    }
318                }
319            }
320    
321            if (fullSQLType == null) {
322                fullSQLType = sql.type;
323            }
324    
325            return fullSQLType;
326        }
