017    package com.ning.metrics.goodwill.access;
018    
019    import java.util.Collections;
020    import java.util.List;
021    import java.util.concurrent.ConcurrentHashMap;
022    import java.util.concurrent.ExecutionException;
023    import java.util.concurrent.Executors;
024    import java.util.concurrent.TimeUnit;
025    
026    public class CachingGoodwillAccessor extends Accessor
027    {
028        private static final int DEFAULT_CACHE_TIMEOUT_IN_SECONDS = 90;
029        private final Object cacheMonitor = new Object();
030    
031        private final ConcurrentHashMap<String, GoodwillSchema> knownSchemata = new ConcurrentHashMap<String, GoodwillSchema>();
032        private final GoodwillAccessor delegate;
033    
034        public CachingGoodwillAccessor(String host, int port)
035        {
036            this(host, port, DEFAULT_CACHE_TIMEOUT_IN_SECONDS);
037        }
038    
039        public CachingGoodwillAccessor(String host, int port, int cacheTimeoutInSeconds)
040        {
041            super(host, port);
042    
043            delegate = new GoodwillAccessor(host, port);
044    
045            Executors.newScheduledThreadPool(1).scheduleWithFixedDelay(new Runnable()
046            {
047                @Override
048                public void run()
049                {
050                    refreshSchemataCache();
051                }
052            }, 0, cacheTimeoutInSeconds, TimeUnit.SECONDS);
053        }
054    
055        /**
056         * Invalidate and refresh the cache
057         * <p/>
058         * This is blocking and returns when the cache has been updated
059         */
060        public void refreshSchemataCache()
061        {
062            try {
063                List<GoodwillSchema> schemata = delegate.getSchemata().get();
064    
065                // If Goodwill is down - keep the old cache around
066                if (schemata == null) {
067                    return;
068                }
069    
070                ConcurrentHashMap<String, GoodwillSchema> newSchemataCache = new ConcurrentHashMap<String, GoodwillSchema>();
071                for (GoodwillSchema schema : schemata) {
072                    newSchemataCache.put(schema.getName(), schema);
073                }
074    
075                synchronized (cacheMonitor) {
076                    knownSchemata.clear();
077                    knownSchemata.putAll(newSchemataCache);
078                }
079            }
080            catch (InterruptedException e) {
081                log.warn("Interrupted while refreshing the cache");
082                Thread.currentThread().interrupt();
083            }
084            catch (ExecutionException e) {
085                log.warn(String.format("Unable to refresh schemata cache: %s", e.getLocalizedMessage()));
086            }
087        }
088    
089        /**
090         * Given a schema name, get the associated GoodwillSchema. This method tries hard to find it, i.e. it will refresh the
091         * cache if the schema is not in the cache.
092         *
093         * @param schemaName name of the schema to find
094         * @return GoodwillSchema describing the schema
095         */
096        public GoodwillSchema getSchema(String schemaName)
097        {
098            GoodwillSchema schema = knownSchemata.get(schemaName);
099            if (schema == null) {
100                refreshSchemataCache();
101                schema = knownSchemata.get(schemaName);
102            }
103    
104            return schema;
105        }
106    
107        public List<GoodwillSchema> getSchemata()
108        {
109            return Collections.list(knownSchemata.elements());
110        }
111    
112        /**
113         * Close the underlying http client
114         */
115        public synchronized void close()
116        {
117            delegate.close();
118        }
119    }
