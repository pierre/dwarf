017    package com.ning.metrics.goodwill.access;
018    
019    import com.ning.http.client.AsyncCompletionHandler;
020    import com.ning.http.client.AsyncHttpClient;
021    import com.ning.http.client.Response;
022    import org.apache.log4j.Logger;
023    import org.codehaus.jackson.map.ObjectMapper;
024    import org.codehaus.jackson.type.TypeReference;
025    
026    import java.io.BufferedReader;
027    import java.io.IOException;
028    import java.io.InputStreamReader;
029    import java.util.HashMap;
030    import java.util.List;
031    import java.util.concurrent.Future;
032    
033    public class GoodwillAccessor
034    {
035        private static final Logger log = Logger.getLogger(GoodwillAccessor.class);
036    
037        private final String host;
038        private final int port;
039        private String url;
040    
041        private final AsyncHttpClient client = new AsyncHttpClient();
042    
043        public GoodwillAccessor(String host, int port)
044        {
045            this.host = host;
046            this.port = port;
047    
048            this.url = String.format("http://%s:%d/registrar", host, port);
049        }
050    
051        /**
052         * Get the Goodwill schema associated to a schema name.
053         * </p>
054         * Typical invocation:
055         * <pre>
056         * {@code
057         * try {
058         * GoodwillSchema type = accessor.getSchema("test").get();
059         * ...
060         * }
061         * catch (Exception e) {
062         * // Connection exception? Goodwill server down?
063         * }
064         * }
065         * </pre>
066         *
067         * @param schemaName schema name to lookup
068         * @return a future on the Schema
069         */
070        public Future<GoodwillSchema> getSchema(String schemaName)
071        {
072            try {
073                return client.prepareGet(String.format("%s/%s", url, schemaName)).addHeader("Accept", "application/json").execute(new AsyncCompletionHandler<GoodwillSchema>()
074                {
075                    @Override
076                    public GoodwillSchema onCompleted(Response response) throws Exception
077                    {
078                        if (response.getStatusCode() != 200) {
079                            return null;
080                        }
081    
082                        BufferedReader reader = new BufferedReader(new InputStreamReader(response.getResponseBodyAsStream()));
083                        ObjectMapper mapper = new ObjectMapper();
084    
085                        GoodwillSchema thrift = mapper.readValue(reader, GoodwillSchema.class);
086    
087                        reader.close();
088    
089                        return thrift;
090                    }
091    
092                    @Override
093                    public void onThrowable(Throwable t)
094                    {
095                        log.warn(t);
096                    }
097                });
098            }
099            catch (IOException e) {
100                log.warn(String.format("Error getting Schema list from %s:%d", host, port));
101                return null;
102            }
103        }
104    
105        /**
106         * Get all schemata.
107         * <p/>
108         * Use schemata, instead of schemas, which is closer to the original σχήματα.
109         *
110         * @return a future on a list of Schema
111         * @see #getSchema for an invocation example
112         */
113        public Future<List<GoodwillSchema>> getSchemata()
114        {
115            try {
116                return client.prepareGet(url).addHeader("Accept", "application/json").execute(new AsyncCompletionHandler<List<GoodwillSchema>>()
117                {
118                    @Override
119                    public List<GoodwillSchema> onCompleted(Response response) throws Exception
120                    {
121                        if (response.getStatusCode() != 200) {
122                            return null;
123                        }
124    
125                        BufferedReader reader = new BufferedReader(new InputStreamReader(response.getResponseBodyAsStream()));
126                        ObjectMapper mapper = new ObjectMapper();
127    
128                        HashMap<String, List<GoodwillSchema>> map = mapper.readValue(
129                            reader,
130                            new TypeReference<HashMap<String, List<GoodwillSchema>>>()
131                            {
132                            });
133                        List<GoodwillSchema> goodwillSchemata = map.get("types");
134    
135                        reader.close();
136    
137                        return goodwillSchemata;
138                    }
139    
140                    @Override
141                    public void onThrowable(Throwable t)
142                    {
143                        log.warn(t);
144                    }
145                });
146            }
147            catch (IOException e) {
148                log.warn(String.format("Error getting Schema list from %s:%d", host, port));
149                return null;
150            }
151        }
152    }
