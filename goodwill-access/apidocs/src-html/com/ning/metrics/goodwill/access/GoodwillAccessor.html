017    package com.ning.metrics.goodwill.access;
018    
019    import com.ning.http.client.AsyncCompletionHandler;
020    import com.ning.http.client.AsyncHttpClient;
021    import com.ning.http.client.AsyncHttpClientConfig;
022    import com.ning.http.client.Response;
023    import org.apache.log4j.Logger;
024    import org.codehaus.jackson.map.ObjectMapper;
025    import org.codehaus.jackson.type.TypeReference;
026    
027    import java.io.BufferedReader;
028    import java.io.IOException;
029    import java.io.InputStreamReader;
030    import java.util.HashMap;
031    import java.util.List;
032    import java.util.concurrent.Future;
033    
034    public class GoodwillAccessor
035    {
036        private static final Logger log = Logger.getLogger(GoodwillAccessor.class);
037    
038        private final String host;
039        private final int port;
040        private String url;
041    
042        private AsyncHttpClient client;
043    
044        public GoodwillAccessor(String host, int port)
045        {
046            this.host = host;
047            this.port = port;
048    
049            this.url = String.format("http://%s:%d/registrar", host, port);
050    
051            createHttpClient();
052        }
053    
054        private void createHttpClient()
055        {
056            // Don't limit the number of connections per host
057            // See https://github.com/ning/async-http-client/issues/issue/28
058            AsyncHttpClientConfig.Builder builder = new AsyncHttpClientConfig.Builder();
059            builder.setMaximumConnectionsPerHost(-1);
060            client = new AsyncHttpClient(builder.build());
061        }
062    
063        /**
064         * Get the Goodwill schema associated to a schema name.
065         * </p>
066         * Typical invocation:
067         * <pre>
068         * {@code
069         * try {
070         * GoodwillSchema type = accessor.getSchema("test").get();
071         * ...
072         * }
073         * catch (Exception e) {
074         * // Connection exception? Goodwill server down?
075         * }
076         * }
077         * </pre>
078         *
079         * @param schemaName schema name to lookup
080         * @return a future on the Schema
081         */
082        public Future<GoodwillSchema> getSchema(String schemaName)
083        {
084            try {
085                return client.prepareGet(String.format("%s/%s", url, schemaName)).addHeader("Accept", "application/json").execute(new AsyncCompletionHandler<GoodwillSchema>()
086                {
087                    @Override
088                    public GoodwillSchema onCompleted(Response response) throws Exception
089                    {
090                        if (response.getStatusCode() != 200) {
091                            return null;
092                        }
093    
094                        BufferedReader reader = new BufferedReader(new InputStreamReader(response.getResponseBodyAsStream()));
095                        ObjectMapper mapper = new ObjectMapper();
096    
097                        GoodwillSchema thrift = mapper.readValue(reader, GoodwillSchema.class);
098    
099                        reader.close();
100    
101                        return thrift;
102                    }
103    
104                    @Override
105                    public void onThrowable(Throwable t)
106                    {
107                        log.warn(t);
108                    }
109                });
110            }
111            catch (IOException e) {
112                log.warn(String.format("Error getting Schema list from %s:%d (%s)", host, port, e.getLocalizedMessage()));
113                return null;
114            }
115        }
116    
117        /**
118         * Get all schemata.
119         * <p/>
120         * Use schemata, instead of schemas, which is closer to the original σχήματα.
121         *
122         * @return a future on a list of Schema
123         * @see #getSchema for an invocation example
124         */
125        public Future<List<GoodwillSchema>> getSchemata()
126        {
127            try {
128                return client.prepareGet(url).addHeader("Accept", "application/json").execute(new AsyncCompletionHandler<List<GoodwillSchema>>()
129                {
130                    @Override
131                    public List<GoodwillSchema> onCompleted(Response response) throws Exception
132                    {
133                        if (response.getStatusCode() != 200) {
134                            return null;
135                        }
136    
137                        BufferedReader reader = new BufferedReader(new InputStreamReader(response.getResponseBodyAsStream()));
138                        ObjectMapper mapper = new ObjectMapper();
139    
140                        HashMap<String, List<GoodwillSchema>> map = mapper.readValue(
141                            reader,
142                            new TypeReference<HashMap<String, List<GoodwillSchema>>>()
143                            {
144                            });
145                        List<GoodwillSchema> goodwillSchemata = map.get("types");
146    
147                        reader.close();
148    
149                        return goodwillSchemata;
150                    }
151    
152                    @Override
153                    public void onThrowable(Throwable t)
154                    {
155                        log.warn(t);
156                    }
157                });
158            }
159            catch (IOException e) {
160                log.warn(String.format("Error getting Schema list from %s:%d (%s)", host, port, e.getLocalizedMessage()));
161                return null;
162            }
163        }
164    }
