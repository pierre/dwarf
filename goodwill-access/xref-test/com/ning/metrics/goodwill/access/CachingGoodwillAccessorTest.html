41  public class CachingGoodwillAccessorTest
42  {
43      private static final int CACHE_TIMEOUT_IN_SECONDS = 3; // 3 seconds
44      private static final String SCHEMA_NAME = "Schema";
45  
46      private int port;
47      private SocketConnector connector;
48      private Server server;
49      private Server errorServer;
50      private final AtomicInteger serversHits = new AtomicInteger(0);
51      private boolean jettyShouldBomb = false;

53      @BeforeClass(alwaysRun = true)
54      public void setUpGlobal() throws Exception
55      {
56          port = findFreePort();
57          connector = new SocketConnector();
58          connector.setHost("127.0.0.1");
59          connector.setPort(port);
60  
61          server = new Server()
62          {
63              @Override
64              public void handle(HttpConnection connection) throws IOException, ServletException
65              {
66                  serversHits.incrementAndGet();
67                  final Request request = connection.getRequest();
68                  final Response response = connection.getResponse();
69  
70                  if (jettyShouldBomb) {
71                      response.setStatus(500);
72                      request.setHandled(true);
73                      return;
74                  }
75                  else {
76                      response.setStatus(200);
77                  }
78  
79                  // Create a dummy schema
80                  List<GoodwillSchemaField> items = new ArrayList<GoodwillSchemaField>();
81                  items.add(new GoodwillSchemaField(SCHEMA_NAME, "STRING", (short) 0, null, "string", null, null, null));
82                  GoodwillSchema schema = new GoodwillSchema(SCHEMA_NAME, items);
83  
84                  // Create the GoodwillSchema store
85                  Map<String, List<GoodwillSchema>> payload = new HashMap<String, List<GoodwillSchema>>();
86                  List<GoodwillSchema> list = new ArrayList<GoodwillSchema>();
87                  list.add(schema);
88                  payload.put("types", list); // Hardcoded value in Goodwill
89  
90                  // Serialize the store and return it
91                  ByteArrayOutputStream out = new ByteArrayOutputStream();
92                  (new ObjectMapper()).writeValue(out, payload);
93                  out.close();
94  
95                  response.getWriter().print(out.toString());
96                  request.setHandled(true);
97              }
98          };
99          server.setThreadPool(new ExecutorThreadPool());
100         server.addConnector(connector);
101         connector.start();
102         server.start();
103     }

113     private int findFreePort() throws IOException
114     {
115         ServerSocket socket = null;
116 
117         try {
118             socket = new ServerSocket(0);
119 
120             return socket.getLocalPort();
121         }
122         finally {
123             if (socket != null) {
124                 socket.close();
125             }
126         }
127     }

130     @Test(groups = "slow")
131     public void testCachingRegistrar() throws Exception
132     {
133         CachingGoodwillAccessor accessor = new CachingGoodwillAccessor("127.0.0.1", port, CACHE_TIMEOUT_IN_SECONDS);
134 
135         // It will warm the cache on startup - give it some time
136         Thread.sleep(CACHE_TIMEOUT_IN_SECONDS * 1000 / 2);
137         Assert.assertEquals(serversHits.get(), 1);
138 
139         // The caching accessor will hit the server if the schema does not exist
140         GoodwillSchema schema = accessor.getSchema("Doesn'tExist");
141         Assert.assertNull(schema);
142         Assert.assertEquals(serversHits.get(), 2);
143 
144         // Try again
145         schema = accessor.getSchema("Doesn'tExist");
146         Assert.assertNull(schema);
147         Assert.assertEquals(serversHits.get(), 3);
148 
149         // Valid schema in the cache shouldn't trigger a hit
150         schema = accessor.getSchema(SCHEMA_NAME);
151         Assert.assertNotNull(schema);
152         Assert.assertEquals(serversHits.get(), 3); // No hit!
153 
154         // Test backend thread
155         Thread.sleep(CACHE_TIMEOUT_IN_SECONDS * 1000);
156         Assert.assertEquals(serversHits.get(), 4); // Refresh
157 
158         // We shouldn't care if Goodwill is down
159         jettyShouldBomb = true;
160 
161         // The good schema is cached
162         schema = accessor.getSchema(SCHEMA_NAME);
163         Assert.assertNotNull(schema);
164         Assert.assertEquals(serversHits.get(), 4);
165 
166         // The bad schemas are not
167         schema = accessor.getSchema("NewOne");
168         Assert.assertNull(schema);
169         Assert.assertEquals(serversHits.get(), 5);
170 
171         accessor.close();
172     }
